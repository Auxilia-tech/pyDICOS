31 bool WriteToProvidedBuffer()
32 {
33  //This example covers how to write to a pre-allocated buffer
34 
35  unsigned char *pMyBuffer(S_NULL); //The buffer to write DICOS data into
36  const SDICOS::S_UINT64 nMyBufferCapacityInBytes(10000); //Buffer capacity in bytes
37  SDICOS::S_UINT64 nDICOSDataSizeInBytes(0); //The size of the DICOS data within the buffer
38 
39  pMyBuffer = new unsigned char[nMyBufferCapacityInBytes]; //Pre-allocate buffer
40 
41  SDICOS::MemoryFile memfile; //The interface for reading and writing data from and to a buffer
42 
43  //Tell the interface it will be used for writing
44  //The passed in text can be anything
45  memfile.OpenWriting("Writing DICOS Data");
46 
47  //Tell the MemoryFile object that it does not own the buffer. This prevents
48  //the MemoryFile from reallocating the buffer if writing requires more space
49  //and deleting the buffer when the MemoryFile is closed.
50  memfile.SetMemoryPolicy(SDICOS::MemoryBuffer::enumPolicy_DoesNotOwnData);
51 
52  //Connect the interface to the buffer and provide its capacity
53  memfile.SetBuffer(pMyBuffer, nMyBufferCapacityInBytes);
54 
55  //Set the used buffer size to zero bytes
56  memfile.ChangeSize(0);
57 
58  //This functions returns the buffer's capacity: nMyBufferCapacityInBytes bytes
59  memfile.GetAllocatedBufferSize();
60 
61  //This function returns the size of the buffer used: 0 bytes
62  memfile.Length();
63 
64  SDICOS::CT ct; //The DICOS modality to write into the buffer. Works with any of the modalities (AIT2D, AIT3D, DX, CT, QR, TDR)
65  SDICOS::ErrorLog errorlog; //Track errors when writing
66 
67  //
68  // ... CT is initialized with data ...
69  //
70  InitCT(ct);
71 
72  bool bRetVal(true);
73 
74  //Write CT to buffer. Writing can fail if the buffer is too small.
75  if (!ct.Write(memfile, errorlog))
76  {
77  //Print error log if writing failed
78  std::cout << errorlog << std::endl;
79  bRetVal = false;
80  }
81  else
82  {
83  //Print out the size of the DICOS data within the buffer object
84  nDICOSDataSizeInBytes = memfile.Length();
85  std::cout << "DICOS data: " << nDICOSDataSizeInBytes << " bytes" << std::endl;
86  }
87 
88  //Closing the MemoryFile disconnects it from the buffer if it does not own the buffer; otherwise,
89  //it deletes the buffer
90  memfile.Close();
91 
92  delete[] pMyBuffer; //Deallocate memory
93  return bRetVal;
94 }

97 bool ReadFromProvidedBuffer()
98 {
99  //This example covers how to read from a provided buffer that contains
100  //DICOS data. The buffer, pMyBuffer, is assumed to have been filled with
101  //DICOS data and has a capacity defined by nMyBufferSizeInBytes. The DICOS
102  //data occupies a size of nMyDICOSDataSizeInBytes in that buffer.
103 
104  unsigned char *pMyBuffer(S_NULL); //The user provided buffer
105  const SDICOS::S_UINT64 nMyBufferCapacityInBytes(10000); //The buffer is allocated to his capacity
106  SDICOS::S_UINT64 nMyDICOSDataSizeInBytes(0);//DICOS data uses this amount of the buffer
107 
108  //
109  // ... Buffer is filled with DICOS data ...
110  //
111  pMyBuffer = new unsigned char[nMyBufferCapacityInBytes];
112  nMyDICOSDataSizeInBytes = InitCT(pMyBuffer, nMyBufferCapacityInBytes); //Size of DICOS data within buffer
113 
114  SDICOS::MemoryFile memfile; //The interface for reading and writing data from and to a buffer
115 
116  //Tell the interface it will be used for reading
117  //The passed in text can be anything
118  memfile.OpenReading("Reading DICOS DATA");
119 
120  //Set the buffer to read from and indicate the size of the DICOS data, not the size of the buffer
121  memfile.SetBuffer(pMyBuffer, nMyDICOSDataSizeInBytes);
122 
123  //Tell the MemoryFile object that it does not own the buffer. This prevents
124  //the MemoryFile from deleting the buffer when closed
125  memfile.SetMemoryPolicy(SDICOS::MemoryBuffer::enumPolicy_DoesNotOwnData);
126

149 bool WriteToBuffer()
150 {
151  SDICOS::CT ct; //The modality to write. Works with any of the modalities (AIT2D, AIT3D, DX, CT, QR, TDR)
152  SDICOS::ErrorLog errorlog;//Track errors when writing
153 
154  //
155  // ... CT has been initialized with data ...
156  //
157  InitCT(ct);
158 
159  SDICOS::MemoryFile memfile; //The interface for reading and writing data from and to a buffer
160 
161  //Open the memory file for writing
162  memfile.OpenWriting("Write");
163 
164  //Write to the MemoryFile. The MemoryFile allocates a buffer to fit the written DICOS data.
165  //If writing requires a larger capacity, the buffer is reallocated to a larger size
166  if (!ct.Write(memfile, errorlog))
167  {
168  std::cout << errorlog << std::endl;
169  return false;
170  }
171 
172  //Print the size of the DICOS data in the memory file
173  std::cout << "DICOS Data Size: " << memfile.Length() << " bytes" << std::endl;
174 
175  //Print the allocated buffer size
176  std::cout << "Buffer Capacity: " << memfile.GetAllocatedBufferSize() << " bytes" << std::endl;
177 
178  //Get the buffer object. The MemoryFile owns the data and will delete it when closed.
179  memfile.GetBuffer();
180 
181  //To recycle the MemoryFile for writing again without deallocating the
182  //buffer, call ChangeSize() and pass in zero bytes. This retains the allocated
183  //buffer, but sets the used size to zero bytes
184  memfile.ChangeSize(0);
185 
186  //Print MemoryFile info
187  std::cout << "Used Buffer Size: " << memfile.Length() << " bytes" << std::endl;
188  std::cout << "Buffer Capacity: " << memfile.GetAllocatedBufferSize() << " bytes" << std::endl;
189 
190  //Close the MemoryFile, which deletes the buffer
191  memfile.Close();
192  return true;
193 }
