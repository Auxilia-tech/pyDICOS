28 /// StartCreateTDRForBaggageSimpleNoThreat
29 /**
30  Create a Threat Detection Report (TDR) generated from an ATR device
31  for a piece of baggage with no Potential Threat Objects (PTO).
32  */
33 bool CreateNoThreatTDRForBaggageSimple()
34 {
35  /// Identifier for the TDR
36  SDICOS::S_INT32 TDRInstanceNumber = 1234;
37 
38  /// The SDICOS.TemplateTDR.TDR object represents a TDR for a single Object Of Inspection (OOI)
39  /// Create a Threat Detection Report (TDR) DICOS file for a piece of Baggage generated by an Automated Threat Recognition (ATR) Device.
40  SDICOS::TDR tdr(
41  SDICOS::ObjectOfInspectionModule::enumTypeBaggage, // OOI is a piece of Baggage
42  SDICOS::TDRTypes::ThreatDetectionReport::enumMachine, // TDR was generated by an ATR machine
43  TDRInstanceNumber // Numeric identifier for the TDR
44  );
45 
46  // Indicate the Date and Time TDR was created
47  SDICOS::DcsDate TDRCreationStartDate = SDICOS::DcsDate::Today();
48  SDICOS::DcsTime TDRCreationStartTime = SDICOS::DcsTime::Now();
49 
50  tdr.SetContentDateAndTime(TDRCreationStartDate,TDRCreationStartTime);
51 
52  // Set ATR Device information. Required by DICOS when TDR type is enumMachine
53  SDICOS::DcsLongString atrManufacturer("Alchemy");
54  SDICOS::DcsLongString atrVersion("0.999");
55 
56  //
57  // Indicate parameters used to the initialize and run the ATR algorithm.
58  // In this example, the ATR algorithm has two parameters.
59  //
60  SDICOS::Array1D<SDICOS::DcsLongString> ATR_Parameters(2);
61  ATR_Parameters[0] = "-random=true";
62  ATR_Parameters[1] = "-magic=true";
63  tdr.SetTDRTypeATR(atrManufacturer,atrVersion,ATR_Parameters);
64 
65  // Image scaling factor is required for TDR Type = ATR
66  // Indicate number of units on the 'human' that 1 unit in the image maps to.
67  tdr.SetImageScaleRepresentation(10.0);
68 
69  // Indicate that an alarm was triggered
70  tdr.SetAlarmDecision(SDICOS::TDRTypes::ThreatDetectionReport::enumClear);
71  SDICOS::DcsDate alarmDecisionDate(1944,6,6);
72  SDICOS::DcsTime alarmDecisionTime(6,30,0,0);
73  tdr.SetAlarmDecisionDateTime(alarmDecisionDate,alarmDecisionTime);
74 
75  // The TDR generation succeeded
76  tdr.SetAbortFlag(SDICOS::TDRTypes::ThreatDetectionReport::enumSuccess);
77 
78  //Add reference to a scan when no PTO is generated. For this example, dummy UIDs are generated.
79  const SDICOS::DcsUniqueIdentifier uidSopInstanceCT(SDICOS::DcsGUID::GenerateAsDecimalString()); //Create a dummy SOP Instance UID for a CT scan. Replace this with a real scan's SOP Instance UID.
80  const SDICOS::DcsUniqueIdentifier uidSopClassCT(SDICOS::SOPClassUID::GetCT());
81  tdr.AddReferencedSopInstance(uidSopInstanceCT, uidSopClassCT);
82 
83  // Write the TDR to a file, read it from the file and compare the results.
84  SDICOS::ErrorLog errorlog;
85  SDICOS::Folder tdrFolder("TDRFiles");
86  SDICOS::Filename tdrFilename(tdrFolder,"SimpleBaggageNoThreatTDR.dcs");
87 
88  if(!tdr.Write(tdrFilename,errorlog))
89  {
90  std::cerr << "Simple TDR Template Example unable to write DICOS File " << tdrFilename << std::endl;
91  std::cerr << errorlog << std::endl;
92  return false;
93  }
94  else
95  {
96  std::cout << "Wrote file to " << tdrFilename << std::endl;
97  SDICOS::TDR tdrRead;
98 
99  SDICOS::ErrorLog errorlog;
100  if(tdrRead.Read(tdrFilename,errorlog))
101  {
102  if(tdrRead == tdr)
103  {
104  std::cout << "Successfully read and compared TDR files" << std::endl;
105  return true;
106  }
107  else
108  {
109  std::cout << "TDR file loaded from " << tdrFilename << " does not match original." << std::endl;
110  return false;
111  }
112  }
113  else
114  {
115  std::cerr << "Unable to read DX file " << tdrFilename << std::endl;
116  std::cerr << errorlog << std::endl;
117  return false;
118  }
119  }
120  return true;
121 }

224  ptoMask.SetBit(w, h, d, value);
225

228  {
229  std::cerr << "Failed to validate threat bitmask (" << w << "," << h << "," << d << ")" << std::endl;

232  }
