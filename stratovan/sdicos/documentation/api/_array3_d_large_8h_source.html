284  void FreeMemory(IMemoryManager *pMemMgr)
285  {
286  MemoryBuffer mb;
287  CSliceClass **pArray2D = CParentClass::GetBuffer();
288  for (S_UINT32 n(0); n < CParentClass::GetSize(); ++n, ++pArray2D)
289  {
290  if (!(*pArray2D)->HasOwnership())
291  {
292  mb.SetMemoryPolicy(MemoryBuffer::enumPolicy_DoesNotOwnData);
293  mb.SetBuffer((unsigned char*)((*pArray2D)->GetBuffer()), (*pArray2D)->EstimateCoreFootprint());
294  pMemMgr->Deallocate(mb);
295  }
296  else
297  mb.SetMemoryPolicy(MemoryBuffer::enumPolicy_OwnsData);
298 
299  DELETE_POINTER(*pArray2D);
300  }
301 
302  // Delete debug slices
303  m_vDebugSlices.FreeMemory();
304 
305  // Delete pointers to the slices
306  CParentClass::FreeMemory();
307  m_nWidth = m_nHeight = 0;
308  }

383  public:
384  /// Construct empty Iterator that indicates iteration has ended
385  Iterator() : m_nCurrentSlice(ITERATION_END_VALUE), m_parray3D(S_NULL), m_psliceCurrent(S_NULL), m_Direction(ITERATION_DIRECTION_FORWARD) {}
386 
387  ///
388  /// Construct an Iterator that is ready for iteration over the given volume.
389  /// @param obj The Array3DLarge volume to iterate over
390  /// @param dir The direction that iteration should occur.
391  ///
392  Iterator(const CThisClass& obj, const ITERATION_DIRECTION dir = ITERATION_DIRECTION_FORWARD)
393  : m_nCurrentSlice(ITERATION_END_VALUE), m_parray3D(&obj), m_psliceCurrent(NULL), m_Direction(dir)
394  {
395  if(0 != obj.GetDepth())
396  {
397  if (ITERATION_DIRECTION_FORWARD == m_Direction)
398  {
399  m_nCurrentSlice = 0;
400  m_psliceCurrent = m_parray3D->CParentClass::GetBuffer() + m_nCurrentSlice;;
401  }
402  else if (ITERATION_DIRECTION_BACKWARD == m_Direction)
403  {
404  m_nCurrentSlice = obj.GetDepth() - 1;
405  m_psliceCurrent = m_parray3D->CParentClass::GetBuffer() + m_nCurrentSlice;;
406  }
407  else if (ITERATION_END == m_Direction)
408  {
409  m_nCurrentSlice = ITERATION_END_VALUE;
410  m_psliceCurrent = S_NULL;
411  }
412  }
413  }

445  /// The prefix operator++ increments the Iterator one step in the iteration direction.
446  /// ITERATION_DIRECTION_FORWARD: move to a slice with greater depth
447  /// ITERATION_DIRECTION_BACKWARD: move to a slice with lesser depth
448  ///
449  /// If the Iterator is at the iteration itEnd value operator++ does nothing.
450  ///
451  /// Dereferencing the Iterator return a pointer to the current slice or NULL.
452  /// @return a reference to the Iterator is returned
453  Iterator& operator++()
454  {
455  // early out if current slice for iteration is not valid
456  if(!m_parray3D || (m_nCurrentSlice < 0))
457  return (*this);
458 
459  if(ITERATION_DIRECTION_FORWARD == m_Direction)
460  {
461  ++m_nCurrentSlice;
462  // Volumes with > 2^31 slices are not supported
463  if (m_nCurrentSlice >= SDICOS::S_INT32(m_parray3D->GetDepth()))
464  {
465  m_nCurrentSlice = ITERATION_END_VALUE;
466  m_psliceCurrent = S_NULL;
467  }
468  else
469  ++m_psliceCurrent;
470  }
471  else
472  {
473  --m_nCurrentSlice;
474  if (ITERATION_END_VALUE < m_nCurrentSlice)
475  --m_psliceCurrent;
476  else
477  m_psliceCurrent = S_NULL;
478  }
479  return (*this);
480  }
481 
482  /// The prefix operator--() decrements the Iterator one step in the iteration direction.
483  /// ITERATION_DIRECTION_FORWARD: move to a slice with lesser depth
484  /// ITERATION_DIRECTION_BACKWARD: move to a slice with greater depth
485  ///
486  /// If the Iterator is at the iteration itEnd value operator-- does nothing.
487  /// @return a reference to the Iterator is returned
488  Iterator& operator--()
489  {
490  // early out if current slice for iteration is not valid
491  if(!m_parray3D || (m_nCurrentSlice < 0))
492  return (*this);
493 
494  if(ITERATION_DIRECTION_FORWARD == m_Direction)
495  {
496  --m_nCurrentSlice;
497  if (ITERATION_END_VALUE < m_nCurrentSlice)
498  --m_psliceCurrent;
499  else
500  m_psliceCurrent = S_NULL;
501  }
502  else
503  {
504  // for a reverse direction iterator operator-- advances one slice in depth
505  ++m_nCurrentSlice;
506  if (m_nCurrentSlice >= SDICOS::S_INT32(m_parray3D->GetDepth()))
507  {
508  m_nCurrentSlice = ITERATION_END_VALUE;
509  m_psliceCurrent = S_NULL;
510  }
511  else
512  ++m_psliceCurrent;
513  }
514  return (*this);
515  }

538  ///

545  return (*this);
546

565
