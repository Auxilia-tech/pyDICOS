32  //This example covers how to write to a pre-allocated buffer
33 
34  unsigned char *pMyBuffer(S_NULL); //The buffer to write DICOS data into
35  const SDICOS::S_UINT64 nMyBufferCapacityInBytes(10000); //Buffer capacity in bytes
36  SDICOS::S_UINT64 nDICOSDataSizeInBytes(0); //The size of the DICOS data within the buffer
37 
38  pMyBuffer = new unsigned char[nMyBufferCapacityInBytes]; //Pre-allocate buffer
39 
40  SDICOS::MemoryFile memfile; //The interface for reading and writing data from and to a buffer
41 
42  //Tell the interface it will be used for writing
43  //The passed in text can be anything
44  memfile.OpenWriting("Writing DICOS Data");
45 
46  //Tell the MemoryFile object that it does not own the buffer. This prevents
47  //the MemoryFile from reallocating the buffer if writing requires more space
48  //and deleting the buffer when the MemoryFile is closed.
49  memfile.SetMemoryPolicy(SDICOS::MemoryBuffer::enumPolicy_DoesNotOwnData);
50 
51  //Connect the interface to the buffer and provide its capacity
52  memfile.SetBuffer(pMyBuffer, nMyBufferCapacityInBytes);
53 
54  //Set the used buffer size to zero bytes
55  memfile.ChangeSize(0);
56 
57  //This functions returns the buffer's capacity: nMyBufferCapacityInBytes bytes
58  memfile.GetAllocatedBufferSize();
59 
60  //This function returns the size of the buffer used: 0 bytes
61  memfile.Length();
62 
63  SDICOS::AttributeManager manager; //Attribute manager to write
64  SDICOS::ErrorLog errorlog; //Track errors when writing
65 
66  //
67  // ... Attribute manager is initialized with data ...
68  //
69  InitAttributeManager(manager);
70 
71  bool bRetVal(true);
72 
73  //Write attribute manager to buffer. Writing can fail if the buffer is too small.
74  if (!SDICOS::DicosFile::Write(memfile, errorlog, manager, SDICOS::DicosFile::enumLittleEndianExplicit))
75  {
76  //Print error log if writing failed
77  std::cout << errorlog << std::endl;
78  bRetVal = false;
79  }
80  else

82  //Print out the size of the DICOS data within the buffer object

85  }
86

149 {
150  SDICOS::AttributeManager manager; //Attribute manager to write
151  SDICOS::ErrorLog errorlog;//Track errors when writing
152 
153  //
154  // ... Attribute manager has been initialized with data ...
155  //
156  InitAttributeManager(manager);
157 
158  SDICOS::MemoryFile memfile; //The interface for reading and writing data from and to a buffer
159 
160  //Open the memory file for writing
161  memfile.OpenWriting("Write");
162 
163  //Write to the MemoryFile. The MemoryFile allocates a buffer to fit the written DICOS data.
164  //If writing requires a larger capacity, the buffer is reallocated to a larger size
165  if (!SDICOS::DicosFile::Write(memfile, errorlog, manager, SDICOS::DicosFile::enumLittleEndianExplicit))
166  {
167  std::cout << errorlog << std::endl;
168  return false;
169  }
170 
171  //Print the size of the DICOS data in the memory file
172  std::cout << "DICOS Data Size: " << memfile.Length() << " bytes" << std::endl;
173 
174  //Print the allocated buffer size
175  std::cout << "Buffer Capacity: " << memfile.GetAllocatedBufferSize() << " bytes" << std::endl;
176 
177  //Get the buffer object. The MemoryFile owns the data and will delete it when closed.
178  memfile.GetBuffer();
179 
180  //To recycle the MemoryFile for writing again without deallocating the
181  //buffer, call ChangeSize() and pass in zero bytes. This retains the allocated
182  //buffer, but sets the used size to zero bytes
183  memfile.ChangeSize(0);
184 
185  //Print MemoryFile info
186  std::cout << "Used Buffer Size: " << memfile.Length() << " bytes" << std::endl;
187  std::cout << "Buffer Capacity: " << memfile.GetAllocatedBufferSize() << " bytes" << std::endl;
188 
189  //Close the MemoryFile, which deletes the buffer
190  memfile.Close();
191  return true;
192 }
