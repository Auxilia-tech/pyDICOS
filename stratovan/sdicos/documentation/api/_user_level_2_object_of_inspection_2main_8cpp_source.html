39 /**
40  * CreateWithObjectOfInspection shows how to use the Object of Inspection(OOI) API methods in IODCommon
41  * to create a CT DICOS file with OOI information.
42  *
43  * In this example all OOI attributes are set and optional attributes are indicated as such.
44  */
45 bool CreateWithObjectOfInspection()
46 {
47  bool bRes(true); // API methods return true/false to indicate success
48 
49  /// Create a CT Series for a cargo Object of Inspection (OOI)
50  /// For more details on using the User-Level API object SDICOS::CT see
51  /// the CTExample and UserCTExample
52  SDICOS::CT ct(
53  SDICOS::ObjectOfInspectionModule::enumTypeCargo,
54  SDICOS::CTTypes::CTImage::enumHighEnergy
55  );
56 
57  /// Acquisition time in seconds
58  ct.SetImageAcquisitionDuration(5.2);
59 
60  /// Add one section of volume data to the CT Series
61  /// Row and Column orientation vectors are (1,0,0) and (0,1,0)
62  /// CT Frame Type Pixel Data Characteristics value is set to ORIGINAL
63  /// CT Filter Type is set to NONE. CT Filter Material is not used.
64  SDICOS::Section * pSection = ct.AddSection();
65  pSection->GetPixelData().Allocate(SDICOS::Volume::enumUnsigned16Bit,128,64,100);
66 
67  pSection->SetFocalSpotSizeInMM(1.414f); /// Used nominal size of the focal spot in MM.
68  pSection->SetKVP(3.14f);
69  pSection->SetPositionInMM(10.0f, 20.0f, 15.0f); /// Volume data location in MM
70  pSection->SetSpacingInMM(1.0f, 2.0f, 10.0f); /// Column spacing, row spacing and slice spacing.
71 
72  // Clear the Volume data for the CT Series section
73  SDICOS::Volume& volume = pSection->GetPixelData();
74  volume.GetUnsigned16()->Zero(0xbeef);
75 
76  // Set object of inspection attributes available through the IODCommon base class
77 
78  // String identifier for the Object of Inspection
79  SDICOS::DcsLongString OOIId("123-345-eager-34t342-avf32");
80  bRes = ct.SetOOIID(OOIId) && bRes;
81 
82  // Name of system that created the OOIId
83  SDICOS::DcsLongString OOIIdAssigningAuthority("Checkpoint Inspection Handler");
84  bRes = ct.SetOOIIDAssigningAuthority((OOIIdAssigningAuthority)) && bRes;
85 
86  // What kind of id is being used
87  bRes = ct.SetOOIIDType(SDICOS::ObjectOfInspectionModule::IdInfo::enumBarcode) && bRes;
88 
89  // OOI Type is set in the constructor, but can also be set through the API
90  bRes = ct.SetOOIType(SDICOS::ObjectOfInspectionModule::enumTypeBioSample) && bRes;
91 
92  // For an OOIType SDICOS::ObjectofInspectionModule::enumPerson, gender is required
93  // bRes = ct.SetGender(SDICOS::ObjectOfInspectionModule::enumGenderFemale) && bRes;
94 
95  // Optional Attribute: mass of object in grams must be > 0.
96  bRes = !ct.SetOOIMassInGrams(-1) && bRes;
97  bRes = ct.SetOOIMassInGrams(2200) && bRes;
98 
99  // Optional Attribute: Multiple OOI Identifiers can be used to identify the same object
100  bRes = ct.AddAlternativeOOIId(
101  "Alternative OOIId",
102  "Alternative OOIId Issuer",
103  SDICOS::ObjectOfInspectionModule::IdInfo::enumText
104  ) && bRes;
105 
106  bRes = ct.AddAlternativeOOIId(
107  "Alternative OOIId 1",
108  "Alternative OOIId Issuer",
109  SDICOS::ObjectOfInspectionModule::IdInfo::enumRFID
110  ) && bRes;
111 
112  // An alternative ID can be removed
113  ct.DeleteAlternativeOOIId("Alternative OOIId");
114 
115  // Optional Attribute: The algorithm routing code is used to identify the threat detection algorithm
116  // that should be used to inspect this OOI. Specific types of OOI can require special handling
117  bRes = ct.SetAlgorithmRoutingCode("ALG_CODE", "ALG_SCHEME", "Algorithm Description") && bRes;
118 
119  // The Algorithm routing code can be removed
120  ct.DeleteAlgorithmRoutingCode();
121 
122  // Optional Attribute: OOI Type Description is a longer, human-readable description
123  bRes = ct.SetOOITypeDescriptor("Human readable description of OOI") && bRes;
124 
125  // The Type Descriptor can be removed
126  ct.DeleteOOITypeDescriptor();
127 
128  // Optional Attribute: Dimensions of OOI in meters. Values my be > 0
129  bRes = !ct.SetOOISizeInMeters(1.2f,0.3f,-1.0f) && bRes;
130  bRes = ct.SetOOISizeInMeters(1.2f,0.2f,0.75f) && bRes;
131 
132  // Dimensions can be removed
133  ct.DeleteOOISize();
134 
135  return bRes;
136 }
