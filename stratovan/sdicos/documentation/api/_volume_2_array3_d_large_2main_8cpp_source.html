48 /// \section Array3DLarge_ex1 Using the Array3DLarge class to set and get volume data using 2D slices.
49 /// Accessing 3D data in the Array3DLarge class.
50 ///
51 bool AccessingArray3DLargeData()
52 {
53  bool bRes(true);
54 
55  //
56  // <b>The SDICOS::Array3DLarge</b> class stores 3D Data for CT, DX, AIT etc.
57  // Internally it is implemented as a 1-D array of 2-D arrays so (X,Y) slices are contiguous blocks of memory.
58  //
59  SDICOS::Array3DLarge<SDICOS::S_UINT16> vnVolume;
60  SDICOS::S_UINT32 width(128),height(128),depth(100);
61  vnVolume.SetSize(width,height,depth);
62 
63  //
64  // The 3D array of CT data can be set in multiple ways
65  //
66  // Method 1: Accessing the array using 3-D indices.
67  //
68  SDICOS::S_UINT32 z;
69  for(z = 0; z < depth; ++z)
70  for(SDICOS::S_UINT32 h = 0; h < height; ++h)
71  for(SDICOS::S_UINT32 w = 0; w < width; ++w)
72  vnVolume[z][h][w] = (z+h+w);
73 
74  //
75  // Method 2: Accessing the data one plane at a time
76  // Note that the 3D memory is not a contiguous block. Slices are contiguous blocks.
77  //
78  for(z = 0; z < depth; ++z)
79  {
80  SDICOS::Array2D<SDICOS::S_UINT16> &zPlane = vnVolume[z];
81  for(SDICOS::S_UINT32 h = 0; h < height; ++h)
82  for(SDICOS::S_UINT32 w = 0; w < width; ++w) {
83  if((z+h+w) != zPlane[h][w]) {
84  bRes = false;
85  std::cout << "Error writing/reading volume data " << zPlane[h][w] << " " << (z+h+w) << std::endl;
86  }
87  zPlane[h][w] = (z+h+w); // If needed convert raw data to SDICOS::S_UINT16 cConvertData<type>(val);
88  }
89  }
90 
91  //
92  // Method 3: Access data explicitly as 2D planes
93  //
94  // The 3D volume is implemented as an array of 2D slices
95  // GetSlice() returns a pointer to an Array2D that stores a contiguous block
96  //
97  for(z = 0; z < depth; ++z)
98  {
99  SDICOS::Array2D<SDICOS::S_UINT16> * volSlice = vnVolume.GetSlice(z);
100 
101  // one-dimensional array that holds a plane of width*height dimensions
102  SDICOS::S_UINT16 * currentSlice = volSlice->GetBuffer();
103 
104  // Set data for a single slice as if it is a 1-D array.
105  for(SDICOS::S_UINT32 i = 0; i < (width*height); ++i) {
106  SDICOS::S_UINT16 value = z + (i/width) + (i%width);
107 
108  if(currentSlice[i] != value) {
109  std::cout << "Error writing/reading volume data " << currentSlice[i] << " " << value << std::endl;
110  bRes = false;
111  }
112 
113  currentSlice[i] = z+i;
114  }
115  }
116 
117  //
118  // Method 4: Copy data directly into the raw buffers of the array.
119  // Requires caller to setup data buffers in the correct format
120  //
121 
122  // Pointer to user data for a XY slice.
123  // This is for example purposes only.
124  SDICOS::S_UINT16 * userDataForSlice = new SDICOS::S_UINT16[width*height];
125  for(SDICOS::S_UINT32 i = 0; i < (width*height); ++i)
126  userDataForSlice[i] = SDICOS::S_UINT16(i);
127 
128  for(z = 0; z < depth; ++z)
129  {
130  // point to data buffer for Array2D object
131  SDICOS::S_UINT16 * destSlice = vnVolume[z].GetBuffer();
132 
133  // Copy user data into the volume slice.
134  // Alternatively, the user can
135  memcpy(destSlice, userDataForSlice, width*height*sizeof(SDICOS::S_UINT16));
136  }
137 
138  delete [] userDataForSlice;
139 
140  // Verify data
141  for(z = 0; z < depth; ++z)
142  for(SDICOS::S_UINT32 h = 0; h < height; ++h)
143  for(SDICOS::S_UINT32 w = 0; w < width; ++w)
144  {
145  if(vnVolume[z][h][w] != SDICOS::S_UINT16(h*width+w))
146  bRes = false;
147  }
148 
149  return bRes;
150 } // EndAccessingArray3DLargeData
