78  * Additions to Threat Sequence Items
79  */
80  {
81  SDICOS::TDRTypes::ThreatSequence::ThreatItem& threatItem = vThreatItems[0];
82 
83  /*
84  * Signal Generation Software Versions : Tag (0060, 7008) VM = 1 or more, VR = LO [optional]
85  * Defines version of software that converts original sensor data and transforms data for presentation or analysis.
86  * For 3D imaging devices this is the reconstruction software version used to generate the display
87  * image from the raw detector dataand for 2D screening devices it is the transformation software/algorithm.
88  */
89  {
90  //Add a version
91  threatItem.AddSignalGenerationSoftwareVersion("Version 41");
92 
93  //Get a reference to the array of software versions
94  SDICOS::Array1D<SDICOS::DcsLongString>& vstrVersions = threatItem.GetSignalGenerationSoftwareVersion();
95 
96  //Check if Signal Generation Software Versions is present
97  bool bIsPresent = threatItem.HasSignalGenerationSoftwareVersion();
98 
99  //Delete Signal Generation Software Versions
100  threatItem.DeleteSignalGenerationSoftwareVersions();
101  }
102 
103  /*
104  * Dynamic Screening Flag : Tag (0060, 7010) VR = CS [optional]
105  * Defines the inspection mode.
106  * Dynamic screening is an adaptive Risk Based Screening (RBS) security screening capability
107  * that provides personalized screening by adjusting screening parameters based on what is known about
108  * each passenger on the day of travel.It enables dynamic switching of automated detection algorithms for
109  * the screening of passengersand their belongings(including accessible property and checked baggage)
110  * based on passenger risk designations in a real-time fashion.
111  *
112  * Valid terms for Dynamic Screening Flag:
113  * BHS
114  * STIP
115  * OTHER
116  * NONE
117  */
118  {
119  //Set the Dynamic Screening Flag
120  threatItem.SetDynamicScreeningFlag(SDICOS::TDRTypes::ThreatSequence::ThreatItem::enumBHS);
121  threatItem.SetDynamicScreeningFlag(SDICOS::TDRTypes::ThreatSequence::ThreatItem::enumSTIP);
122  threatItem.SetDynamicScreeningFlag(SDICOS::TDRTypes::ThreatSequence::ThreatItem::enumOther);
123  threatItem.SetDynamicScreeningFlag(SDICOS::TDRTypes::ThreatSequence::ThreatItem::enumNone);
124 
125  //Get the Dynamic Screening Flag as an enum
126  SDICOS::TDRTypes::ThreatSequence::ThreatItem::DYNAMIC_SCREENING_FLAG nFlag = threatItem.GetDynamicScreeningFlagEnum();
127 
128  //Get the Dynamic Screening Flag as a string
129  SDICOS::DcsString strDSF = threatItem.GetDynamicScreeningFlagString();
130 
131  //Check if the Dynamic Screening Flag is present
132  bool bIsPresent = threatItem.HasDynamicScreeningFlag();
133 
134  //Delete the Dynamic Screening Flag
135  threatItem.DeleteDynamicScreeningFlag();
136  }
137  }

165  // Specifies method used summed the bounding boxes together

167  // Specifies method used the largest bounding box

171  // Specifies method used was none of the methods above

173  ptoRep.SetBoundingBoxResolution(SDICOS::TDRTypes::BaggageSpecificTDR::PTORepresentationSequenceItem::enumBBR_Other);
