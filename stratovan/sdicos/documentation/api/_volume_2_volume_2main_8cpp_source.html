41 /// Example showing how to use the SDICOS::Volume class used to store signed or unsigned 16-bit
42 /// volume data in an SDICOS::CT object using SDICOS:Section or an SDICOS::AIT3D object using SDICOS::SectionAIT.
43 //
44 /// \section Volume Use the SDICOS::Volume class to store signed or unsigned 8,16,32,64 bit and float data in SDICOS::Section or SDICOS::SectionAIT.
45 ///
46 bool VolumeExample()
47 {
48  bool bRes(true);
49 
50  ///
51  /// The Volume object stores either signed or unsigned 8,16,32,64 bit and float data using the SDICOS::Array3DLarge class.
52  /// It is designed to only store one kind of data at a time.
53  /// The interface for Volume contains utility methods for interfacing with the
54  SDICOS::Volume vol;
55 
56  // Explicitly allocate a float volume of the given size and delete any unsigned data
57  SDICOS::S_UINT32 width(123),height(234), depth(12);
58  vol.Allocate(SDICOS::Volume::enumFloat, width, height, depth);
59 
60  SDICOS::S_UINT32 sliceCount(0);
61 
62  // Use an iterator to access the slices in the volume and manipulate them
63  SDICOS::Volume::SliceIterator itSliceFloat = vol.Begin();
64  for(; itSliceFloat != vol.End(); ++ itSliceFloat, ++sliceCount)
65  {
66  // pSlice is a Array2D
67  SDICOS::Array2D<float> *pSlice = itSliceFloat.AsFloat();
68 
69  // Get a raw buffer to the slice's data
70  // For further examples for accessing volume data see the Example VolumeDataExample/VolumeDataExample.cpp
71  float *pSliceBuffer = pSlice->GetBuffer();
72 
73  memset(pSliceBuffer, 0xB0, vol.GetSliceSizeInBytes());
74  }
75 
76  bRes = (sliceCount == depth) && bRes;
77 
78  // Slices can be added individually
79  SDICOS::Array2D<float> *pSlice = vol.AddSliceFloat();
80  bRes = (vol.GetDepth() == (sliceCount+1)) && bRes;
81  depth = vol.GetDepth();
82 
83  // Slices can be added using an iterator that supports operator++() and operator*()
84  SDICOS::S_UINT32 numNewSlices(5);
85  SDICOS::Array2D<float> **ppSlices = new SDICOS::Array2D<float>*[numNewSlices+1];
86  for (SDICOS::S_UINT32 i(0); i < numNewSlices; ++i)
87  {
88  ppSlices[i] = new SDICOS::Array2D<float>(width, height);
89  ppSlices[i]->Zero(float(i));
90  }
91  ppSlices[numNewSlices] = NULL; //Add NULL to mark end of slices
92 
93  //The volume object, by default, is set to own all the slices and is responsible for deleting them.
94  //Adding slices in this way transfers the data from the array of slices to the Volume object
95  vol.AddSlices(ppSlices, &ppSlices[numNewSlices]);
96  bRes = (vol.GetDepth() == (numNewSlices + depth)) && bRes;
97 
98  //Delete slices. They no longer own the data and act as references to the data.
99  for (SDICOS::S_UINT32 i(0); i < numNewSlices; ++i)
100  DELETE_POINTER(ppSlices[i]);
101 
102  delete [] ppSlices; //Delete array of pointers to slices
103 
104  // The raw Array3DLarge object can be accessed
105  SDICOS::Array3DLarge<float> &volRef = *vol.GetFloat();
106  bRes = (volRef.GetDepth() == (numNewSlices+depth)) && bRes;
107 
108  // Slices can be added from an existing 3D array
109  {
110  SDICOS::S_UINT32 oldDepth(vol.GetDepth());
111  const SDICOS::S_UINT32 w(1), h(2), d(3);
112  SDICOS::Array3DLarge<float> newData;
113 
114  bRes = newData.SetSize(w,h,d) && bRes;
115  newData.Zero(-30789.0f);
116 
117  // Fails because slice dimensions do not match
118  bRes = !vol.AddSlices(newData) && bRes;
119 
120  SDICOS::S_UINT32 newDepth(123);
121 
122  // Call SetSize() first to allocate memory by 2D slices.
123  bRes = newData.SetSize(vol.GetWidth(),vol.GetHeight(), newDepth) && bRes;
124  newData.Zero(-31062.0f);
125 
126  // Setting the Array3DLarge (source) and Volume (destination) objects to OWNS_SLICES results in moving
127  // ownership of the slices when calling AddSlices().
128  //
129  // Setting the destination object to DOES_NOT_OWN_SLICES copies references to the slices.
130  // This means the slice data is shared between both objects and the source handles deleting
131  // the data.
132  newData.SetMemoryPolicy(SDICOS::MemoryPolicy::OWNS_SLICES); //Default memory policy
133  vol.SetMemoryPolicy(SDICOS::MemoryPolicy::OWNS_SLICES); //Default memory policy
134 
135  bRes = vol.AddSlices(newData) && bRes;
136  bRes = !newData.OwnsSlices(); //No longer owns data as ownership was moved to Volume object
137  bRes = SDICOS::MemoryPolicy::OWNS_SLICES == vol.GetMemoryPolicy(); //Volume has data ownership
138  bRes = (vol.GetDepth() == (oldDepth + newDepth)) && bRes;
139  }
140 
141  // Same as above with Unsigned data. The memory allocation policy is set so that the Array3DLarge<float> object owns the data
142  // instead of the SDICOS::Volume object.
143  {
144  // Volume that holds float data, but does not own the data
145  SDICOS::Volume vol32bit(SDICOS::Volume::enumUnsigned32Bit, SDICOS::MemoryPolicy::DOES_NOT_OWN_SLICES);
146 
147  // Create an Array3DLarge of unsigned slices and add those slices to the Volume object.
148  const SDICOS::S_UINT32 w(10), h(20), d(30);
149  SDICOS::Array3DLarge<SDICOS::S_UINT32> data(w,h,d, SDICOS::MemoryPolicy::OWNS_SLICES);
150  data.Zero(0xFEDC);
151 
152  bRes = vol32bit.AddSlices(data) && bRes;
153  bRes = (d == vol32bit.GetDepth()) && bRes;
154  }
155 
156  return bRes;
157 }
