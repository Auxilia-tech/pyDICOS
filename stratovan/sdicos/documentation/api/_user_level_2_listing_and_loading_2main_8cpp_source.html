52 int main(int argc, char* argv[])
53 {
54  SDICOS::DicosLoader dcsloader; // DICOS loading class
55  DicosLoadingCallback dcscallback; // Callback interface for passing loaded files
56  SDICOS::DicosFileListing dFileListing; // Finds DICOS files
57  SDICOS::Array1D<SDICOS::DicosFileListing::SopInstanceFiles> vFiles; // Organized DICOS files from SDICOS::DicosFileListing
58  const SDICOS::DicosFileListing::SopInstanceFiles *pSOP = NULL; // SOP instance reference
59  SDICOS::Folder folder; // Target folder to find DICOS files
60  SDICOS::Array1D<std::pair<SDICOS::Filename, SDICOS::ErrorLog> > vErrorLog; // List of DICOS files with errors
61 
62  if (1 == argc)
63  {
64  // Set the folder with DICOS files.
65  // When compiled, this file is at ..\TSA\DICOS\Examples\DicosListingAndLoadingExample\DicosListingAndLoadingExample.cpp
66  // And the data files are at ..\TSA\DICOS\Examples\DataFiles
67  SDICOS::Filename thisFile(__FILE__);
68  folder = thisFile.GetFullFolder();
69  folder.Set(folder.GetFullFolder() + "../../DataFiles");
70  std::cout << "Searching for DICOS files under " << folder.GetFullFolder() << std::endl;
71  }
72  else
73  {
74  folder.Set(std::string(argv[1]));
75  std::cout << "Searching for DICOS files under " << folder.GetFullFolder() << std::endl;
76  }
77 
78  // Find all DICOS files in the target folder and its sub folders. Any files DICOS files excluded from
79  // the listing due to errors will populate the third parameter.
80  // By default, this will only check files
81  // with no extension or a '*.dcs' extension. To list DICOS files with other extensions, pass a set
82  // of case sensitive extensions (ie. ".dicos") to the fourth parameter.
83  // To list non-DICOS files that share the same format, like DICOM, set the fifth parameter to 'true'.
84  if(!dFileListing.FindDicosFiles(folder, true, vErrorLog))
85  {
86  std::cout<<"\nNo DICOS files found in "<<folder.GetFullFolder()<<std::endl;
87  return 0;
88  }
89 
90  // Print out error log of DICOS files that failed to be included in the listing
91  for(SDICOS::S_UINT32 nError(0) ; nError < vErrorLog.GetSize() ; ++nError)
92  {
93  std::cout<<"Failed to list file \""<<vErrorLog[nError].first.GetName()<<"\"\n";
94  std::cout<<"Error log:\n"<<vErrorLog[nError].second<<std::endl;
95  }
96 
97  if (0 != vErrorLog.GetSize())
98  return 2;
99 
100  // Set the preferred callback handler. User-Level API is set by default.
101  // By default, this callback interface will only check files
102  // with no extension or a '*.dcs' extension. To load DICOS files with other extensions, call
103  // DicosLoadingCallback::IncludeFileExtension with a case sensitive extension (ie. ".dicos").
104  // To load non-DICOS files that share the same format, like DICOM, add the override
105  // function "virtual bool OnLoadOnlyDICOS()const" to the callback interface and set
106  // the function to return 'false'.
107  dcsloader.SetCallback(&dcscallback);
108 
109  // Set the preferred callback API. User-Level API is set by default.
110  dcsloader.PreferUserLevelAPI();
111 
112  // Get a list of all DICOS files organized in a hierarchy of OOI > Scan > Series > SOP Class
113  dFileListing.GetSopInstanceFiles(vFiles);
114 
115  if (0 == vFiles.GetSize())
116  {
117  return 3;
118  }
119 
120  // Print out information about each SOP instance found in the target folder
121  pSOP = vFiles.GetBuffer();
122  for(SDICOS::S_UINT32 nSopInstance(0) ; nSopInstance < vFiles.GetSize() ; ++nSopInstance, ++pSOP)
123  {
124  std::cout<<"\n-------------------";
125  std::cout<<"\nOOI ID: "<<pSOP->m_dsOOIID.Get();
126  std::cout<<"\nScan Instance UID: "<<pSOP->m_dsScanInstanceUID.Get();
127  std::cout<<"\nSeries Instance UID: "<<pSOP->m_dsSeriesInstanceUID.Get();
128  std::cout<<"\nSOP Class UID: "<<pSOP->m_dsSopClassUID.Get();
129  std::cout<<"\nSOP Instance UID: "<<pSOP->m_dsSopInstanceUID.Get();
130  std::cout<<"\nModality: "<<pSOP->m_dsModality.Get();
131  std::cout<<"\nNumber of Files: "<<pSOP->m_vFilenames.GetSize();
132  for(SDICOS::S_UINT32 nSopInstanceFile(0); nSopInstanceFile < pSOP->m_vFilenames.GetSize(); ++nSopInstanceFile)
133  std::cout<<"\n "<<pSOP->m_vFilenames[nSopInstanceFile].GetFullPath().c_str();
134  std::cout<<std::endl;
135  }
136 
137  // Example code showing how to load files by ids.
138  // The callback handler type is DicosLoadingCallback which is defined at the top of this file.
139  //
140  // Load all AIT3D DICOS files in SDICOS::DicosFileListing with
141  // OOI ID = "001"
142  // Scan Instance UID = "1.1.1",
143  // Series instance UID = "2.2.2".
144  if(dcsloader.Load(dFileListing, "001", "1.1.1", "2.2.2", SDICOS::SOPClassUID::GetAIT3D()))
145  {
146  std::cout << "\nSuccessfully loaded AIT3D DICOS files with OOI ID = 001, scan instance UID = 1.1.1 and series instance UID = 2.2.2" << std::endl;
147  }
148 
149  return 0;
150 }
