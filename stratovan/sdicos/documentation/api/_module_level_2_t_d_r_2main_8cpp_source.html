364

367  //

373  return false;

378  bRes = bRes && (tdr == tdr2);
379 
380  if(bRes)

383  }
384  else

387  }

389  {
390  std::cout << "Failed to validate Person-Specific TDR read from disk" << std::endl;
391  std::cout << errorlog << std::endl;
392  return false;
