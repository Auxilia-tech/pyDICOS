44 /**
45  * Convert a CT VTK File to a DICOS CT File with the User-Level API
46  * @return true/false
47  */
48 bool ConvertVTKToDICOSWithUserAPI(const char * filename)
49 {
50  const SDICOS::Filename filenameVTK(filename);
51 
52  /*
53  * CT Data is stored as a buffer of unsigned shorts
54  */
55  SDICOS::Array3DLarge<unsigned short> vnVolume;
56 
57  /*
58  * Origin and Spacing are stored in the SDICOS::CTTypes::CTMultiframeFunctionalGroupMacros
59  */
60  SDICOS::Point3D<float> ptOrigin;
61  SDICOS::Point3D<float> ptSpacing;
62 
63  if(!LoadVTKVolume(filenameVTK, vnVolume, ptOrigin, ptSpacing)) {
64  std::cout << "Failed to load VTK file " << filename << std::endl;
65  return false;
66  }
67 
68  /*
69  * The SDICOS::CT object is the user-level API object that represent the
70  * Computed Tomography (CT) Information Object Definition (IOD) in the DICOS Specification Section 7.
71  *
72  * Defined in "SDICOS/ModuleCT.h"
73  * Here we create a SDICOS::CT object and then initialize required DICOS attributes.
74  */
75 
76  /*
77  Construct a CT object given information about the Object of Inspection
78  All attributes not set below are given valid default values.
79  */
80  SDICOS::CT ctUser(
81  SDICOS::ObjectOfInspectionModule::enumTypeBaggage,
82  SDICOS::CTTypes::CTImage::enumHighEnergy,
83  SDICOS::CTTypes::CTImage::enumVolume,
84  SDICOS::ImagePixelMacro::enumMonochrome2
85  );
86 
87  /// Acquisition time in seconds. Set to FLT_MAX to indicate unknown time value.
88  ctUser.SetImageAcquisitionDuration(FLT_MAX);
89 
90  //
91  // Create a new SDICOS::Section object to contain the volume data and indicate
92  // that the new Section object now owns its slice data.
93  //
94  SDICOS::Section * pSection = ctUser.AddSection(vnVolume, SDICOS::MemoryPolicy::OWNS_SLICES);
95  vnVolume.SetMemoryPolicy(SDICOS::MemoryPolicy::DOES_NOT_OWN_SLICES);
96 
97  // These parameters are required
98  pSection->SetFocalSpotSizeInMM(focalSpotSizeInMM);
99  pSection->SetKVP(KVP);
100 
101  /*
102  * Set Pixel spacing in MM for X, Y, Z directions
103  */
104  pSection->SetSpacingInMM(ptSpacing.x, ptSpacing.y, ptSpacing.z);
105 
106  /*
107  * Set plane origin position.
108  */
109  pSection->SetPositionInMM(ptOrigin.x,ptOrigin.y,ptOrigin.z);
110 
111  /*
112  * Set plane orientation for the volume's top left position on a slice.
113  * Slice rows corresponds to X-axis and and slice columns correspond to Y-Axis
114  */
115  const SDICOS::Vector3D<float> vecRow(1, 0, 0);
116  const SDICOS::Vector3D<float> vecColumn(0, 1, 0);
117  pSection->SetPlaneOrientation(vecRow, vecColumn);
118 
119  const SDICOS::Filename outputFilename("CTFromVTKUserAPI.dcs");
120 
121  /*
122  * The ErrorLog class stores warning and errors from the validation process
123  */
124  SDICOS::ErrorLog errorlog;
125 
126  /*
127  * Writing the CT file validates the CT object and reports errors in the supplied errorlog.
128  */
129  if(!ctUser.Write(outputFilename, errorlog, SDICOS::DicosFile::enumLittleEndianExplicit))
130  {
131  SDICOS::Filename errorLogFilename("CTFromVTKUserAPI-ErrorLog.txt");
132  errorlog.WriteLog(errorLogFilename);
133  std::cout << "Failed to write CT File with User API " << outputFilename.GetFullPath() << std::endl;
134  std::cout << "ErrorLog written to " << errorLogFilename.GetFullPath() << std:: endl;
135  std::cout << errorlog << std::endl;
136  return false;
137  }
138 
139  errorlog.FreeMemory();
140 
141  SDICOS::AttributeManager manager;
142  if(SDICOS::DicosFile::Read(outputFilename, errorlog, manager))
143  {
144  manager.WriteToFileAsText("CTFromVTKUserAPI-Listings.txt");
145  }
146  else
147  {
148  std::cout << "Failed to read CT DICOS file " << outputFilename.GetFullPath() << std::endl;
149  std::cout << errorlog << std::endl;
150  return false;
151  }
152 
153  errorlog.FreeMemory();
154 
155  // Read CT using User Level API
156  SDICOS::CT ctUserRead;
157 
158  if(!ctUserRead.Read(outputFilename, errorlog))
159  {
160  std::cout << "Failed to read CT DICOS file " << outputFilename.GetFullPath() << std::endl;
161  return false;
162  }
163  else
164  {
165  if(ctUser == ctUserRead)
166  {
167  std::cout << "Successfully read and compared converted CT file" << std::endl;
168  }
169  else
170  {
171  std::cout << "Failed to read and compare converted CT file" << std::endl;
172  std::cout << errorlog << std::endl;
173  return false;
174  }
175  }
176 
177  return true;
178 } // EndConvertVTKToDICOSWithUserAPI

181 /**
182  * Convert a CT VTK File to a DICOS CT File with the Module-Level API
183  * This example explicitly sets all required DICOS attributes
184  * using the Module-Level API interface.
185  *
186  * In the User-Level API example above, the SDICOS::CT User-Level API object
187  * initializes required attributes to valid default values.
188  *
189  * @return true/false
190  */
191 bool ConvertVTKToDICOSWithModuleAPI(const char * filename)
192 {
193  /*
194  * The CT object is the high-level API object that represent the
195  * Computed Tomography (CT) Information Object Definition (IOD) in the DICOS Specification Section 7.
196  *
197  * #include "SDICOS/ModuleCT.h"
198  *
199  * All objects in the Stratovan DICOS Toolkit are declared in the SDICOS namespace.
200  */
201  SDICOS::CTModule ct;
202 
203  const SDICOS::Filename filenameVTK(filename);
204 
205  /*
206  * CT Data is stored as a buffer of unsigned shorts
207  */
208  SDICOS::Array3DLarge<unsigned short> vnVolume;
209 
210  /*
211  * Origin and Spacing are stored in the SDICOS::CTTypes::CTMultiframeFunctionalGroupMacros
212  */
213  SDICOS::Point3D<float> ptOrigin;
214  SDICOS::Point3D<float> ptSpacing;
215 
216  if(!LoadVTKVolume(filenameVTK, vnVolume, ptOrigin, ptSpacing))
217  return false;
218 
219  /*
220  * API methods generally return true/false to indicate
221  * success and failure.
222  */
223  bool bRes(true);
224 
225  /*
226  * The following code sets up all required attributes in the CT Image IOD (Section &, Table 14)
227  * This shows the minimum set of required attributes needed for a DICOS file to conform
228  * to the DICOS Specification.
229  */
230 
231  /*
232  * Set CT Series attributes represented by the CT Series Module (Table
233  */
234 
235  bRes = bRes && ct.GetSeries().SetAcquisitionStatus(SDICOS::GeneralSeriesModule::enumSuccessful);
236  bRes = bRes && ct.GetSeries().SetDateAndTime("20140114", "074417");
237  bRes = bRes && ct.GetSeries().SetInstanceUID("1.2.276.0.7230010.3.1.3.4126100761.8984.1407370732.880");
238 
239  /*
240  * The CT Image class represents the CT Image Module (Table 17)
241  *
242  * See "SDICOS/CTImage.h"
243  */
244  bRes = bRes && ct.GetCTImage().SetPixelData(vnVolume);
245 
246  /*
247  * DcsdDateTime object takes data as "YYYYMMDD" and time as "HHMMSS"
248  */
249  SDICOS::DcsDateTime strDT;
250  strDT.Set("20140114", "074417");
251  bRes = bRes && ct.GetCTImage().SetAcquisitionDateAndTime(strDT);
252  bRes = bRes && ct.GetCTImage().SetBurnedInAnnotation(false);
253 
254  /*
255  * See the documentation for SDICOS::ImagePixelMacro::PHOTOMETRIC_INTERPRETATION
256  */
257  bRes = bRes && ct.GetCTImage().SetPhotometricInterpretation(SDICOS::ImagePixelMacro::enumMonochrome2);
258 
259  /*
260  * The CT Image Type attribute contains 4 values.
261  */
262  bRes = bRes && ct.GetCTImage().SetImageTypeCT( SDICOS::CTTypes::CTImage::enumOriginal,
263  SDICOS::CTTypes::CTImage::enumPhotoelectric,
264  SDICOS::CTTypes::CTImage::enumVolume,
265  SDICOS::CTTypes::CTImage::enumNone);
266  bRes = bRes && ct.GetCTImage().SetAcquisitionDuration(30);
267  bRes = bRes && S_NULL != ct.GetCTImage().GetCommonImageDescription();
268  bRes = bRes && ct.GetCTImage().GetCommonImageDescription()->SetPixelPresentation(SDICOS::CommonImageDescription::enumPP_Monochrome);
269  bRes = bRes && ct.GetCTImage().GetCommonImageDescription()->SetVolumeBasedCalculationTechnique(SDICOS::CommonImageDescription::enumVC_None);
270  bRes = bRes && ct.GetCTImage().GetCommonImageDescription()->SetVolumetricProperties(SDICOS::CommonImageDescription::enumVP_Volume);
271 
272  /*
273  * The CTTypes::CTMultiframeFunctionalGroups class is the CT Specific implementation of the MultiframeFunctionalGroups (Table 109) module
274  * with the CT Specific Multiframe Functional Groups in Table 15
275  */
276  bRes = bRes && ct.GetMultiframeFunctionalGroups().SetNumberOfFrames(vnVolume.GetDepth());
277  bRes = bRes && ct.GetMultiframeFunctionalGroups().SetContentDateAndTime("20140114", "074417");
278  bRes = bRes && ct.GetMultiframeFunctionalGroups().SetInstanceNumber(1001);
279 
280  /*
281  * Passing true to the GetSharedFunctionalGroups(), GetPixelMeasurement(), GetPlaneOrientation(), GetPlanePosition()
282  * GetPixelValueTransformation(), GetImageFrameType() and GetXRayDetails()
283  * methods ensures that space is allocated for the PixelMeasurement object.
284  */
285  ct.GetMultiframeFunctionalGroups().AllocateSharedFunctionalGroups();
286  SDICOS::CTTypes::CTMultiframeFunctionalGroupMacros *pmfg = ct.GetMultiframeFunctionalGroups().GetSharedFunctionalGroups();
287  pmfg->AllocatePixelMeasurement();
288  bRes = bRes && pmfg->GetPixelMeasurement();
289 
290  /*
291  * Set Pixel spacing in MM for X and Y directions
292  */
293  bRes = bRes && pmfg->GetPixelMeasurement()->SetPixelSpacingInMM(ptSpacing.x, ptSpacing.y);
294 
295  /*
296  * Set slice thickness in MM (Z direction)
297  */
298  bRes = bRes && pmfg->GetPixelMeasurement()->SetSliceThicknessInMM(1.0f);
299 
300  /*
301  * Set plane orientation using cosines to define frame vectors.
302  */
303  pmfg->AllocatePlaneOrientation();
304  bRes = bRes && pmfg->GetPlaneOrientation()->SetImageOrientation(1,0,0, 0,1,0);
305 
306  /*
307  * Set plane per frame positions along the z-axis.
308  */
309  SDICOS::Array1D<SDICOS::CTTypes::CTMultiframeFunctionalGroupMacros> &vPerFrame = ct.GetMultiframeFunctionalGroups().GetPerFrameFunctionalGroups();
310 
311  for (SDICOS::S_UINT32 nFrame(0); nFrame < vPerFrame.GetSize(); ++nFrame)
312  {
313  SDICOS::CTTypes::CTMultiframeFunctionalGroupMacros &mfgm = vPerFrame[nFrame];
314  mfgm.AllocatePlanePosition();
315  bRes = bRes && mfgm.GetPlanePosition()->SetPositionInMM(ptOrigin.x, ptOrigin.y, ptOrigin.z + ptSpacing.z * float(nFrame));
316  }
317 
318  /*
319  * The PixelValueTransformation specifies a linear transform applied to the
320  * stored values to produce values in the output units.
321  */
322  pmfg->AllocatePixelValueTransformation();
323  bRes = bRes && pmfg->GetPixelValueTransformation()->SetRescaleIntercept(-1000);
324  bRes = bRes && pmfg->GetPixelValueTransformation()->SetRescaleSlope(1);
325 
326  /*
327  * String value indicating what the output units are after applying pixel transformation
328  */
329  bRes = bRes && pmfg->GetPixelValueTransformation()->SetRescaleType("HU");
330 
331  /*
332  * Same as as call to ct.GetCTImage().SetImageTypeCT()
333  */
334  pmfg->AllocateImageFrameType();
335  bRes = bRes && pmfg->GetImageFrameType()->SetFrameType( SDICOS::CTTypes::CTImage::enumOriginal,
336  SDICOS::CTTypes::CTImage::enumPhotoelectric,
337  SDICOS::CTTypes::CTImage::enumVolume,
338  SDICOS::CTTypes::CTImage::enumNone);
339 
340  /*
341  * The CommonImageDescription data is shared between the CTImage and CTMultiframeFunctionalGroupMacros
342  */
343  pmfg->GetImageFrameType()->GetCommonImageDescription() = *(ct.GetCTImage().GetCommonImageDescription());
344 
345  /*
346  * The CTXRayDetails module is required if the first value in the CT Image Type attribute is ORIGINAL or MIXED.
347  * See CTImage::SetImageTypeCT which is set above by calling SetImageTypeCT
348  */
349  pmfg->AllocateXRayDetails();
350  bRes = bRes && pmfg->GetXRayDetails()->SetFilterMaterial(SDICOS::CTTypes::CTXRayDetails::enumLead);
351  bRes = bRes && pmfg->GetXRayDetails()->SetFilterType(SDICOS::CTTypes::CTXRayDetails::enumNone);
352  bRes = bRes && pmfg->GetXRayDetails()->SetFocalSpotSizeInMM(0);
353  bRes = bRes && pmfg->GetXRayDetails()->SetKVP(0);
354 
355  /*
356  * The SDICOS::FrameOfReference class represents the Frame of Reference (Table 107)
357  * The Frame of Reference UID uniquely identifies a frame of reference for a series and
358  * is a grouping to indicate that multiple series in a scan have the same frame of reference.
359  */
360  bRes = bRes && ct.GetFrameOfReference().SetFrameOfReferenceUID("1.2.276.0.7230010.3.1.4.4126100761.8984.1407370740.886");
361 
362  /*
363  * The ObjectOfInspectionModule class represents Object Of Inspection Module (OOI) (Section 4.1, Table 10)
364  */
365  bRes = bRes && ct.GetObjectOfInspection().GetID().SetID("TSS4.000314");
366  bRes = bRes && ct.GetObjectOfInspection().GetID().SetIdAssigningAuthority("Issurer-001");
367  bRes = bRes && ct.GetObjectOfInspection().GetID().SetIdType(SDICOS::ObjectOfInspectionModule::IdInfo::enumText);
368  bRes = bRes && ct.GetObjectOfInspection().SetType(SDICOS::ObjectOfInspectionModule::enumTypeBaggage);
369 
370  /*
371  * The GeneralScanModule class represents the General Scan Module (Section 5 Table 12)
372  */
373  bRes = bRes && ct.GetGeneralScan().SetScanID("TSS4.314");
374  bRes = bRes && ct.GetGeneralScan().SetScanInstanceUID("1.2.276.0.7230010.3.1.2.4126100761.8984.1407370732.879");
375  bRes = bRes && ct.GetGeneralScan().SetStartDateAndTime("20140114", "074417");
376  bRes = bRes && ct.GetGeneralScan().SetType(SDICOS::GeneralScanModule::enumOperational);
377 
378  /*
379  * The GeneralEquipmentModule class represent the General Equipment Module (Table 70)
380  * Required information about the machine the scan was captured on.
381  */
382  bRes = bRes && ct.GetGeneralEquipment().SetCalibrationDateAndTime("20140225", "130532.435");
383  bRes = bRes && ct.GetGeneralEquipment().SetDeviceSerialNumber("DSN-001");
384  bRes = bRes && ct.GetGeneralEquipment().SetMachineAddress("Southfield, Michigan");
385  bRes = bRes && ct.GetGeneralEquipment().SetMachineID("Gremlin");
386  bRes = bRes && ct.GetGeneralEquipment().SetMachineLocation("1 Rust Bucket Way");
387  bRes = bRes && ct.GetGeneralEquipment().SetManufacturer("American Motors");
388  bRes = bRes && ct.GetGeneralEquipment().SetManufacturerModelName("GremlinX");
389  bRes = bRes && ct.GetGeneralEquipment().SetSoftwareVersion("1972");
390 
391  /*
392  * The SOPCommonModule class represents the SOP Common Module (Table 82)
393  * which defines attributes for identifying associated Service Object Pair (SOP) instances
394  */
395  bRes = bRes && ct.GetSopCommon().SetSopInstanceCreationDateAndTime("20140225", "150423.546");
396  bRes = bRes && ct.GetSopCommon().SetSopInstanceUID("1.2.276.0.7230010.3.1.4.4126100761.8984.1407370732.881");
397 
398  /*
399  * The CommonInstanceReferenceModule class represents the Common Instance Reference Module (Table 87)
400  * which describes the relationships between SOP Instances referenced between modules in the same instance.
401  * In this example, we have one scan with one series.
402  */
403  ct.GetCommonInstanceReference().AllocateReferencedSeries();
404  bRes = bRes && ct.GetCommonInstanceReference().GetReferencedSeries()->GetReferencedSeries().SetSize(1);
405 
406  /*
407  * Note that this series instance UID matches the one set above on the CT object
408  * with the call to ct.GetSeries().SetInstanceUID(...)
409  */
410  bRes = bRes && ct.GetCommonInstanceReference().GetReferencedSeries()->GetReferencedSeries()[0].SetSeriesInstanceUID("1.2.276.0.7230010.3.1.3.4126100761.8984.1407370732.880");
411  bRes = bRes && ct.GetCommonInstanceReference().GetReferencedSeries()->GetReferencedSeries()[0].GetReferencedInstances().SetSize(1);
412  bRes = bRes && ct.GetCommonInstanceReference().GetReferencedSeries()->GetReferencedSeries()[0].GetReferencedInstances()[0].SetClassUID(SDICOS::SOPClassUID::GetCT());
413  bRes = bRes && ct.GetCommonInstanceReference().GetReferencedSeries()->GetReferencedSeries()[0].GetReferencedInstances()[0].SetInstanceUID("1.2.276.0.7230010.3.1.4.4126100761.8984.1407370732.885");
414 
415  const SDICOS::Filename outputFilename("CTFromVTKModuleAPI.dcs");
416 
417  /*
418  * The ErrorLog class stores warning and errors from the validation process
419  */
420  SDICOS::ErrorLog errorlog;
421 
422  /*
423  * Writing the CT file validates the CT object and reports errors in the
424  * supplied errorlog. If this call
425  */
426  if(!ct.Write(outputFilename, SDICOS::DicosFile::enumLittleEndianExplicit, errorlog))
427  {
428  errorlog.WriteLog("CTFromVTKModuleAPI-ErrorLog.txt");
429  std::cout << "Failed to write CT File " << outputFilename.GetFullPath() << std::endl;

501 bool LoadVTKVolume(
502  const SDICOS::Filename &filename,
503  SDICOS::Array3DLarge<unsigned short> &arrayVolume,
504  SDICOS::Point3D<float> &ptOrigin,
505  SDICOS::Point3D<float> &ptSpacing)
506 {
507  bool bRes(true);
508 
509  SDICOS::File file;
510 
511  if(!file.OpenReading(filename))
512  return false;
513 
514  int nW(1), nH(1), nD(1) /*nNumDims(0), nDataType(0)*/;
515  VTK_SCALAR_TYPE nScalarType(enumVTK_Unknown);
516 
517  std::string str;
518  {
519  bRes = bRes && file.ReadWord(str);
520  if(!IsEqualNoCase(str, "#"))
521  return false;
522  bRes = bRes && file.ReadWord(str);
523  if(!IsEqualNoCase(str, "vtk"))
524  return false;
525  bRes = bRes && file.ReadWord(str);
526  if(!IsEqualNoCase(str, "DataFile"))
527  return false;
528  bRes = bRes && file.ReadWord(str);
529  if(!IsEqualNoCase(str, "Version"))
530  return false;
531  bRes = bRes && file.ReadLine(str);
532  if(!IsEqualNoCase(str, "3.0"))
533  return false;
534  }
535 
536  {
537  bRes = bRes && file.ReadLine(str);
538  if(!IsEqualNoCase(str, "vtk output"))
539  return false;
540  }
541 
542  {
543  bRes = bRes && file.ReadLine(str);
544  if(!IsEqualNoCase(str, "BINARY"))
545  return false;
546  }
547 
548  {
549  bRes = bRes && file.ReadLine(str);
550  if(!IsEqualNoCase(str, "DATASET STRUCTURED_POINTS"))
551  return false;
552  }
553 
554  {
555  bRes = bRes && file.ReadWord(str);
556  if(!IsEqualNoCase(str, "DIMENSIONS"))
557  return false;
558  bRes = bRes && file.ReadWord(str);
559  nW = atoi(str.c_str());
560  bRes = bRes && file.ReadWord(str);
561  nH = atoi(str.c_str());
562  bRes = bRes && file.ReadLine(str);
563  nD = atoi(str.c_str());
564  }
565 
566  {
567  bRes = bRes && file.ReadWord(str);
568  if(!IsEqualNoCase(str, "SPACING"))
569  return false;
570  bRes = bRes && file.ReadWord(str);
571  ptSpacing.x = (float)atof(str.c_str());
572  bRes = bRes && file.ReadWord(str);
573  ptSpacing.y = (float)atof(str.c_str());
574  bRes = bRes && file.ReadLine(str);
575  ptSpacing.z = (float)atof(str.c_str());
576  }
577 
578  {
579  bRes = bRes && file.ReadWord(str);
