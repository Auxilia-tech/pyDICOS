119 int main()
120 {
121  CustomQR customQR; //Our custom User-Level API QR class
122 
123  //Set values for the private data
124  customQR.m_fPrivateValue = 1.2f;
125  customQR.m_strPrivateText = "Sampled Sample Text";
126 
127  InitQR(customQR); //Initialize base class data
128 
129  SDICOS::MemoryFile memfile; //Buffer to write data into
130  SDICOS::ErrorLog errorlog; //Error log
131 
132  memfile.OpenReadingWriting("QR Memoryfile"); //Setup buffer to be written into
133 
134  //Write the QR data to the memory buffer
135  if (!customQR.Write(memfile, errorlog, SDICOS::DicosFile::enumLittleEndianExplicit))
136  {
137  std::cout << errorlog << std::endl;
138  return 1;
139  }
140 
141  memfile.SeekFront(); //Move to front of buffer
142 
143  /*
144  Read data into a new CustomQR object and compare with the original object
145  */
146  {
147  CustomQR customQRRead; //Custom QR used for reading
148 
149  //Read the data
150  if (!customQRRead.Read(memfile, errorlog))
151  {
152  std::cout << errorlog << std::endl;
153  return 2;
154  }
155 
156  //Private tags are marked as warnings in the errorlog
157  std::cout << errorlog << std::endl;
158 
159  //Compare both objects
160  if (customQR != customQRRead)
161  {
162  std::cout << "Data does not match!" << std::endl;
163  return 3;
164  }
165  }
166 
167  /*
168  The APIAllocators class provides a set of function pointers to allocate
169  custom classes.
170  */
171 
172  SDICOS::Utils::APIAllocators apiAllocators; //Set of custom allocator function pointers
173 
174  //Assign the custom allocator for each custom class
175  apiAllocators.m_pfnAllocateModuleQR = &CustomQRModule::AllocateCustom; //Module-Level API
176  apiAllocators.m_pfnAllocateUserQR = &CustomQR::AllocateCustom; //User-Level API
177 
178  /*
179  DicosLoader and DicosServer can use the allocator object
180  to ensure the correct class is allocated and passed to
181  the callback functions.
182  */
183 
184  SDICOS::DicosLoader dcsLoader;
185  dcsLoader.SetCustomAllocators(apiAllocators);
186 
187  SDICOS::Network::DcsServer dcsServer;
188  dcsServer.SetCustomAllocators(apiAllocators);
189 
190  /*
191  To use the default classes instead of the custom classes,
192  set the function pointer for that function pointer to S_NULL.
193  */
194  {
195  apiAllocators.m_pfnAllocateUserQR = S_NULL; //Only restores the Module-Level API to default
196  }
197 
198  //Update users of the allocator functions
199  dcsLoader.SetCustomAllocators(apiAllocators);
200  dcsServer.SetCustomAllocators(apiAllocators);
201 
202  /*
203  To reset all the allocators to use the default classes,
204  call SDICOS::Utils::APIAllocators::FreeMemory();
205  */
206  {
207  apiAllocators.FreeMemory();
208  }
209 
210  //Update users of the allocator functions
211  dcsLoader.SetCustomAllocators(apiAllocators);
212  dcsServer.SetCustomAllocators(apiAllocators);
213 
214  return 0;
215 }

437 {

462 bool CustomQR::ReadDerived_internal(const SDICOS::QRModule &qr, SDICOS::ErrorLog &errorlog)

473 
474  /*
475  Get data from CustomQRModule
476  */
477  {
478  //Get the float value
479  pCustomQRModule->m_attPrivateValue.GetValue(m_fPrivateValue);
480 
481  //Get the text value. Check if there is a value with SDICOS::AttributeString::GetVM().

483  {
484  //Get the value by reference using the a SDICOS::AttributeString::operator[], which
485  //returns an SDICOS::DcsString reference. SDICOS::DcsString::Get() returns the char*
486  //to the data.
487  m_strPrivateText = pCustomQRModule->m_attPrivateText[0].Get();
488  }
489  }
490 
491  return true;
