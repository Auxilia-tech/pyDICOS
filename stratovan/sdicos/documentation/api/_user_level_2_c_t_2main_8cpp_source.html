43  /// The SDICOS::CT object represents a single Series in a Scan in an OOI.
44  /// All required Date and Time attributes are set to SDICOS::DcsDate::Today() and SDICOS::DcsTime::Now()

48  SDICOS::CTTypes::CTImage::enumHighEnergy, // Data is from a high-energy scan. See Examples/Multiple_Energy_CT for how to combine High and Low energy.

61  /// Section position at (0,0,0)
62  /// XYZ voxel spacing (1,1,1)
63  /// Row and Column orientation vectors are (1,0,0) and (0,1,0) by default
64  /// CT Frame Type Pixel Data Characteristics value is set to ORIGINAL
65  /// CT Filter Type is set to NONE. CT Filter Material is not used.

149  bRes = (NULL != pSection2 && bRes);

165  {

167  }

173  bRes = (NULL != volumeIterator.AsSigned8()) && bRes;
174  }
175  }

177  {
178  for(volumeIterator = (*sectionIt)->GetPixelData().Begin() ; (volumeIterator != (*sectionIt)->GetPixelData().End()); ++volumeIterator, ++sliceCount)
179  {
180  bRes = (NULL != volumeIterator.AsUnsigned16()) && bRes;
181  }
182  }

184  {

186  {
187  bRes = (NULL != volumeIterator.AsSigned16()) && bRes;
188  }
189  }
190  else if (SDICOS::Volume::enumUnsigned32Bit == (*sectionIt)->GetPixelDataType())
191  {

193  {
194  bRes = (NULL != volumeIterator.AsUnsigned32()) && bRes;
195  }
196  }
197  else if (SDICOS::Volume::enumSigned32Bit == (*sectionIt)->GetPixelDataType())
198  {

200  {
201  bRes = (NULL != volumeIterator.AsSigned32()) && bRes;
202  }
203  }
204  else if (SDICOS::Volume::enumUnsigned64Bit == (*sectionIt)->GetPixelDataType())
205  {
206  for(volumeIterator = (*sectionIt)->GetPixelData().Begin() ; (volumeIterator != (*sectionIt)->GetPixelData().End()); ++volumeIterator, ++sliceCount)
207  {

213  for(volumeIterator = (*sectionIt)->GetPixelData().Begin() ; (volumeIterator != (*sectionIt)->GetPixelData().End()); ++volumeIterator, ++sliceCount)

232  return false;

237  SDICOS::Folder ctFolder("CTUserAPI/SimpleCT");
238  SDICOS::Filename ctFilename(ctFolder,"SimpleCT.dcs");
239 
240  // Two separate files will be created in the output folder because all sections in the CT do
241  // not have equal width and height volume data.
242  if(!ct.Write(ctFilename,errorlog,SDICOS::DicosFile::enumLittleEndianExplicit))
243  {
244  std::cerr << "Simple CT Template Example unable to write DICOS File " << ctFilename << std::endl;
245  std::cerr << errorlog << std::endl;
246  return false;

251 
252  //

254  // for a CT with a single section use this read method:
255  // ctRead.Read(ctFilename,errorlog)

257  SDICOS::DicosFileListing files;

260 
261  // Access files by ObjectOfInspection->Scan->Series
262  const SDICOS::DicosFileListing::SopInstance& fileInstance = files[0][0][0][0];
263

271  }
272  else
273  std::cerr <<"Simple CT Example original CT and read CT are equal. Number of Sections: " << ctRead.GetNumberOfSections() << std::endl;
274  }
275  else
276  {
277  std::cerr << "Failed to read CT Series from " << ctFilename << std::endl;
278  std::cerr << errorlog << std::endl;
279  return false;
280  }
281  }
282

311 // The position for Section 2 starts at -5000 (100 slices from Section 0 times 2.5 MM spacing)

314 ///
315 /// This example demonstrates the User Level API to create a CT DICOS file
316 /// with multiple volume sections that are saved as multiple DICOS files.
317 ///
318 bool CreateCTComplex()
319 {
320  ///
321  /// The SDICOS::CT object abstracts away the hierarchy of the DICOS specification
322  /// into a single API that contains methods for setting all of the attributes needed
323  /// for a DICOS CT object.
324  ///
325  /// The SDICOS::CT object represents a single Series in a Scan
326  /// in an Object of Inspection (OOI).
327  ///
328  /// Note that the Scan may contain multiple Series represented by
329  /// different CT instances and that the OOI may contain multiple scans.
330  ///
331  SDICOS::CT ct;
332 
333  //
334  // Where applicable, class methods in the Stratovan DICOS library return true/false to indicate success failure.
335  // In this example the return values of various methods are tested for success/failure but this is not a requirement for usage.
336  //
337  bool bRes(true);
338 
339  //
340  // Set CT attributes in the order that they are defined by the DICOS specification
341  // for the CT Information Object Definition (IOD)
342  //
343 
344  //
345  // Set Object of Inspection (OOI) Attributes)
346  //
347  bRes = SetObjectOfInspection(ct) && bRes;
348 
349  //
350  // Set Scan Module attributes. A Scan is a collection of one or more series of
351  // security objects grouped together to identify a theat.
352  // Each Scan is associated with one Object of Interest (OOI)
353  //
354 
355  // Set the UID to identity the Scan instance that the Series belongs to.
356  // If the Series is one of many in a Scan, this UID should be passed in to the
357  // user of the CTSeriesBase class. If not is can be auto-generated by calling
358  // the method CTSeroesBase::GenerateScanInstanceUID()
359  //
360  // A UID is a series of numeric components separated by the period "."
361  ct.SetScanInstanceUID("1.2.3.4.5");
362 
363  // Set the time and date that the Scan started
364  bRes = ct.SetScanStartDateAndTime(SDICOS::DcsDate::Today(),SDICOS::DcsTime::Now()) && bRes;
365  bRes = ct.SetScanType(SDICOS::GeneralScanModule::enumOperational) && bRes;
366 
367  // The Scan ID attribute is a human-readable ID for identifying the scan in a local context
368  bRes = ct.SetScanID("001") && bRes;
369 
370  //
371  // Set Series Module attributes. The CTSeriesBase object represents a single
372  // Series in the Scann.
373  // Note: The CT object forces the Series Modality attribute to "CT"
374  //
375 
376  // A UID is a series of numeric components separated by the period "."
377  bRes = ct.SetSeriesInstanceUID(SDICOS::DcsUniqueIdentifier::CreateGUID());
378 
379  // Set the date and time that this series started
380  bRes = ct.SetSeriesDateAndTime(SDICOS::DcsDate::Today(), SDICOS::DcsTime::Now()) && bRes;
381  bRes = ct.SetSeriesAcquisitionStatus(SDICOS::GeneralSeriesModule::enumSuccessful) && bRes;
382 
383  bRes = SetAcquisitionDeviceInformation(ct) && bRes;
384 
385  // Unique Identifier for this Service Object Pair (SOP) Instance
386  // In this example, the UID is set from a user supplied string.
387  SDICOS::DcsUniqueIdentifier strInstanceUID("1.2.3.4");
388  bRes = ct.SetSopInstanceUID(strInstanceUID) && bRes;
389 
390  // The Frame of Reference UID is used to uniquely identify a frame of reference for the images within a series
391  // or across multiple series in the same scan.
392  // Images that shared a Frame of Reference UID are spatially related.
393  // For this example, a random UID is simply generated. However, if multiple series share a Frame of Reference
394  // then this UID needs to match across them.
395  ct.GenerateFrameOfReferenceUID();
396 
397  //
398  // Set information about how the image data was gathered by the device.
399  //
400  // The Date and Time that data acquisition started
401  bRes = ct.SetImageAcquisitionDateAndTime(SDICOS::DcsDate(2014,12,3), SDICOS::DcsTime(22,23,4,0)) && bRes;
402 
403  // Indicates whether or not the image contains sufficient burned in annotation
404  // to identify the OOI and date the image was acquired.
405  // This could be "text" included as part of the CT data to identify objects or orientation.
406  // This value is set to false by default.
407  ct.SetBurnedInAnnotation(false);
408 
409  // Intended interpretation of pixel data.
410  // For CT these are:
411  // enumMonochrome1 "MONOCHROME1", Minimum pixel value is white
412  // enumMonochrome2 "MONOCHROME2", Minimum pixel value is black
413  bRes = ct.SetPhotometricInterpretation(SDICOS::ImagePixelMacro::enumMonochrome1) && bRes;
414 
415  // Information about the image and volume data
416  bRes = ct.SetImageType(SDICOS::CTTypes::CTImage::enumHighEnergy,SDICOS::CTTypes::CTImage::enumVolume) && bRes;
417 
418  // Acquisition Duration in Seconds
419  bRes = ct.SetImageAcquisitionDuration(1.414) && bRes;
420 
421  // The time and date the Data Creation was start.
422  // Note: this is the time pixel data was created not the time the data was
423  // acquired by the scanning device.
424  bRes = ct.SetContentDateAndTime(SDICOS::DcsDate::Today(),SDICOS::DcsTime::Now()) && bRes;
425 
426  // The Instance Number is another means for the user to identify this CT Series
427  // The default value is 0.
428  ct.SetInstanceNumber(4251);
429 
430  //
431  // The CT object represents a single Series in a Scan that can contain multiple Series.
432  // The volume data for a single Series can be contained in multiple,related volumes called Sections.
433  // Sections are independently allocated and can be manipulated in parallel.
434  // The two types of Sections are called related and independent.
435  //
436  // Note that in a typical deployed DICOS application, the CT Series will not be written to disk
437  // but will be written to another DICOS device either through a network connection or through
438  // shared memory.
439  //
440 
441  //
442  // Related Sections:
443  // For CT Data, a larger volume can be broken down into smaller sections along the Z-direction (Depth).
444  // These sections must share the same Width(W), Height(H).
445  // For example a 512x512x512 (WxHxD) volume can be broken down into three Sections
446  // of dimensions 512x512x100, 512x512x200 and 512x512x212.
447  //
448  // The data can be set for these Sections independently, but when the CT data is written to a file
449  // the related sections are combined into a single volume and the original Sections are not
450  // restored when they are read back in.
451  //
452 
453  //
454  // Independent Sections:
455  // Independent sections cannot be combined to form a larger volume. These are
456  // Sections that have different dimensions, spacing or location.
457  // Independent Sections are saved to separate DICOS files.
458  //
459  // For example, for a particular OOI and Scan, a Series could have a normal
460  // resolution CT volume for the whole OOI and a small higher-resolution volume
461  // for a sub-volume that is of particular interest.
462  // These are stored as independent sections.
463  //
464  SDICOS::Section * pSection = NULL;
465 
466  //
467  // This example illustrates a CT with two related sections and one independent section with different dimensions and spacing.
468  // The two related sections,which shared X,Y dimensions and X,Y,X spacing are 128x128x50 and 128x128x25. Together they form a
469  // 128x128x75 volume in a single file, but remain separate sections due to positioning. The independent section is 64x64x100
470  // and has different spacing and is placed in a separate file.
471  //
472  ct.SetNumberOfSections(3);
473 
474  for(SDICOS::S_UINT32 i = 0; i < 3; ++i)
475  {
476  pSection = ct.GetSectionByIndex(i);
477 
478  // Set data common to all Sections of the volume
479  // See documentation above where sectionSpacing array pSection defined for the meaning of these values.
480  pSection->SetSpacingInMM(sectionSpacing[i][SectionSpacingColumnIndex],sectionSpacing[i][SectionSpacingRowIndex],sectionSpacing[i][SectionSpacingSliceIndex]);
481  pSection->SetPositionInMM(position[i][0],position[i][1],position[i][2]);
482 
483  // Set slicing direction for slices in the volume data
484  // true: (Default) slicing direction is in the row X column direction
485  // false: slicing direction is in the -(row X column) direction
486  pSection->SetSlicingDirection(true);
487 
488  // Set image orientation based on the top left of each slice
489  // Pixel row along X-Axis (left-to-right)
490  // Pixel column along Y-Axis (top-to-bottom)
491  const SDICOS::Vector3D<float> vecRow(1, 0, 0);
492  const SDICOS::Vector3D<float> vecColumn(0, 1, 0);
493  pSection->SetPlaneOrientation(vecRow, vecColumn);
494 
495  // Set Data specific to CT Scans
496  pSection->SetImagePixelPresentation(SDICOS::CommonImageDescription::enumPP_Monochrome);
497  pSection->SetImageVolumeBasedCalculationTechnique(SDICOS::CommonImageDescription::enumVC_None);
498  pSection->SetImageVolumetricProperties(SDICOS::CommonImageDescription::enumVP_Volume);
499 
500  // Set parameters to transform stored pixel data
501  pSection->SetRescaleSlope(1.0f);
502  pSection->SetRescaleIntercept(0.0f);
503  pSection->SetRescaleType("HU"); // Set the units used by pixel data after applying rescaling
504 
505  // X-Ray absorbing material in the filter
506  pSection->SetFilterMaterial(SDICOS::CTTypes::CTXRayDetails::enumMixed);
507 
508  // Type of filter inserted into the XRay beam
509  // Wedge, Butterfly, Multiple, Flat, Shaped, None
510  pSection->SetFilterType(SDICOS::CTTypes::CTXRayDetails::enumFlat);
511 
512  // Used nominal size of the focal spot in MM.
513  pSection->SetFocalSpotSizeInMM(5);
514 
515  // Peak kilo voltage output of the x-ray generator
516  pSection->SetKVP(100.1f);
517  }
518 
519  //
520  // Set the Volume data associated with each of the Sections.
521  // See Examples/VolumeDataExample for more information about working with volumes.
522  //
523  SDICOS::Volume& v0 = ct.GetSectionByIndex(0)->GetPixelData();
524  v0.Allocate(SDICOS::Volume::enumUnsigned16Bit);
525 
526  SDICOS::Array3DLarge<SDICOS::S_UINT16> &v0Data = *v0.GetUnsigned16();
527  v0Data.SetSize(sectionDims[0][0],sectionDims[0][1],sectionDims[0][2]);
528 
529  // Set volume data by setting each XY plane
530  // See the example file Examples/VolumeDataExample/VolumeDataExample.cpp
531  for(SDICOS::S_UINT32 i = 0; i < v0Data.GetDepth(); ++i)
532  {
533  SDICOS::Array2D<SDICOS::S_UINT16>& xyPlane = v0Data[i];
534  SDICOS::S_UINT16 * pBuffer = xyPlane.GetBuffer();

679 bool VerifyWrittenDICOSFiles(
680  const SDICOS::CT& originalCT,
681  const SDICOS::Folder& ctFolder,
682  const SDICOS::Filename& ctFilename
683 )
684 {
685  bool bRes(true);
686 
687  std::cout << "Write CT DICOS file from template " << ctFilename << " " << originalCT.GetNumberOfFilesToCreate() << std::endl;
688 
689  //
690  // The SDICOS::DicosFileListing class searches and collects all DICOS files
691  // in a file system folder and hierarchy. Any files DICOS files excluded from
692  // the listing due to errors will populate the 'vErrorLog' parameter.
693  //
694  SDICOS::DicosFileListing files;
695  SDICOS::Array1D<std::pair<SDICOS::Filename, SDICOS::ErrorLog> > vErrorLog;
696  files.FindDicosFiles(ctFolder,false,vErrorLog);
697 
698  if( 0 == files.GetNumberOfObjectOfInspections() ||
699  0 == files[0].GetNumberOfScans() ||
700  0 == files[0][0].GetNumberOfSeries())
701  {
702  std::cerr << "Unable to find DICOS files for CT Template" << std::endl;
703  return false;
704  }
705 
706  //
707  // In this example, the SDICOS::DicosFileListing object is used to find the
708  // files generated for each independent CT Section added to the CT Series.
709  // There are two files generated in this example.
710  //
711  SDICOS::Array1D<SDICOS::DicosFileListing::FileInfo> filenames;
712  files.GetFileInfo(filenames);
713 
714  for(SDICOS::S_UINT32 i = 0; i < filenames.GetSize(); ++i)
715  {
716  std::cout << filenames[i].m_filename << std::endl;
717  }
718 
719  // One error log is created for each Section in the Series
720  SDICOS::Array1D< std::pair<SDICOS::Filename, SDICOS::ErrorLog> > vErrorlogs;
721  SDICOS::CT ctRead;
722 
723  // Access files by ObjectOfInspection->Scan->Series
724  const SDICOS::DicosFileListing::SopInstance& fileInstance = files[0][0][0][0];
725 
726  // Read the CT Data for the Series
727  if(!ctRead.Read(fileInstance, vErrorlogs))
728  {
729  std::cerr << "Unable to read CT DICOS files" << std::endl;
730  for(SDICOS::S_UINT32 i = 0; i < vErrorlogs.GetSize(); ++i)
731  {
732  std::cerr << vErrorlogs[i].first << std::endl << vErrorlogs[i].second << std::endl;
733  }
734 
735  return false;
736  }
737 
738  const SDICOS::S_UINT32 nNumSections(3);
739  std::cout << "Loaded CT DICOS Files. Number of Sections: " << ctRead.GetNumberOfSections() << std::endl;
740  bRes = (nNumSections == ctRead.GetNumberOfSections()) && bRes;
741  SDICOS::Section * sections[nNumSections];
742 
743  for (SDICOS::S_UINT32 n(0); n < nNumSections; ++n)
744  {
745  sections[n] = ctRead.GetSectionByIndex(n);
746  bRes = bRes && S_NULL != sections[n];
747  }
748 
749  // Check that volumes have the expected dimensions and spacing
750  for(SDICOS::S_UINT32 i = 0; i < nNumSections; ++i)
751  {
752  bool bSectionSpacingFound(false);
753  for(SDICOS::S_UINT32 j = 0; j < nNumSections; ++j)
754  {
755  if( SDICOS::IsEqual(sections[i]->GetColumnSpacing(),sectionSpacing[j][0]) &&
756  SDICOS::IsEqual(sections[i]->GetRowSpacing(), sectionSpacing[j][1]) &&
757  SDICOS::IsEqual(sections[i]->GetSliceSpacing(), sectionSpacing[j][2])
758  )
759  {
760  bSectionSpacingFound = true;
761  break;
762  }
763  }
764  bRes = bSectionSpacingFound && bRes;
765  bool bSectionDimsFound(false);
766  for(SDICOS::S_UINT32 j = 0; j < nNumSections; ++j)
767  {
768  if(sections[i]->GetWidth() == sectionDims[j][0] &&
769  sections[i]->GetHeight() == sectionDims[j][1] &&
770  sections[i]->GetDepth() == sectionDims[j][2])
771  {
772  bSectionDimsFound = true;
773  break;
774  }
775  }
776 
777  bRes = bSectionDimsFound && bRes;
778  }
779  std::cout << "Comparison of original and loaded CT files " << bRes << std::endl;
780 
781  return bRes;

784 void GenerateCTSection(SDICOS::CT& ct)
785 {
786  SDICOS::Section *psection = S_NULL;
787 
788  ct.SetNumberOfSections(1);
789  psection = ct.GetSectionByIndex(0);
790 
791  psection->SetFilterMaterial(SDICOS::CTTypes::CTXRayDetails::enumAluminum);
792  psection->SetFocalSpotSizeInMM(10);
793  psection->SetKVP(7000);
794 
795  SDICOS::Vector3D<float> vecRowOrientation;
796  SDICOS::Vector3D<float> vecColumnOrientation;
797 
798  vecRowOrientation.Set(1,0,0); // X-Axis points right
799  vecColumnOrientation.Set(0,1,0); // Y-Axis points down
800  // Z-Axis points into screen
801  psection->SetPlaneOrientation(vecRowOrientation, vecColumnOrientation);
802 
803  psection->SetPositionInMM(-125,-125,125);
804 
805  /// Physical Distance in the OOI between pixel centers. Measured in MM.
806  const float fColumnSpacing = 1.0f; // Distance in MM between pixel centers of two adjacent columns
807  const float fRowSpacing = 1.0f; // Distance in MM between pixel centers of two adjacent rows
808  const float fSliceSpacing = 1.0f; //Distance in MM between pixel centers of to adjacent slices
809  const float fSliceThickness = 1.0f; //Thickness of each slice in MM
810 
811  psection->SetSpacingInMM(fColumnSpacing, fRowSpacing, fSliceSpacing);
812  psection->SetSliceThickness(fSliceThickness);
813  psection->GetPixelData().Allocate(SDICOS::Volume::enumUnsigned16Bit, 256, 256, 256);
814  psection->GetPixelData().GetUnsigned16()->Zero(0);
815 
816  const SDICOS::Point3D<SDICOS::S_UINT16> ptCenter(125,125,125);
817 
818  // Make shapes
819  const SDICOS::S_UINT16 nWhite(0xFFFF), nGray256(0x0100), nGray2048(0x1000);
820  SDICOS::Array3DLarge<SDICOS::S_UINT16> &vVolume = *psection->GetPixelData().GetUnsigned16();
821  SDICOS::Point3D<SDICOS::S_UINT16> ptPos;
822 
823  // Rectangle on left-top-back(-X,-Y,+Z) corner of volume .
824  // Longest on Z axis, then Y-axis, then X-axis
825  SDICOS::S_UINT32 curSlice(0);
826 
827  for(ptPos.z = 0 ; ptPos.z < 200 ; ++ptPos.z, ++curSlice)
828  {
829  SDICOS::Array2D<SDICOS::S_UINT16>* pSlice = vVolume.GetSlice(curSlice);
830  for(ptPos.y = 0 ; ptPos.y < 20 ; ++ptPos.y)
831  {
832  SDICOS::S_UINT16 *pvoxel = pSlice->GetBuffer() + (ptPos.y * pSlice->GetWidth());
833  for(ptPos.x = 0 ; ptPos.x < 50 ; ++ptPos.x, ++pvoxel)
834  *pvoxel = nWhite;
835  }
836  }
837 
838  // Center block
839  for(curSlice = 0, ptPos.z = (ptCenter.z-40) ; ptPos.z < (ptCenter.z+40) ; ++ptPos.z, ++curSlice)
840  {
841  SDICOS::Array2D<SDICOS::S_UINT16>* pSlice = vVolume.GetSlice(curSlice);
842  for(ptPos.y = (ptCenter.y-40); ptPos.y < (ptCenter.y+40) ; ++ptPos.y)
843  {
844  ptPos.x = (ptCenter.x-40);
845  SDICOS::S_UINT16 *pvoxel = pSlice->GetBuffer() + (ptPos.y * pSlice->GetWidth()) + ptPos.x;
846  for(; ptPos.x < (ptCenter.x+40) ; ++ptPos.x, ++pvoxel)
847  *pvoxel = nGray256;
848  }
849  }
850 
851  // Small block to the right (+X) of the center block
852  for(curSlice = 0, ptPos.z = (ptCenter.z-5) ; ptPos.z < (ptCenter.z+5) ; ++ptPos.z, ++curSlice)
853  {
854  SDICOS::Array2D<SDICOS::S_UINT16>* pSlice = vVolume.GetSlice(curSlice);
855  for(ptPos.y = (ptCenter.y-5); ptPos.y < (ptCenter.y+5) ; ++ptPos.y)
856  {

858  SDICOS::S_UINT16 *pvoxel = pSlice->GetBuffer() + (ptPos.y * pSlice->GetWidth()) + ptPos.x;
859  for(; ptPos.x < (ptCenter.x+51) ; ++ptPos.x, ++pvoxel)
860  *pvoxel = nGray2048;
861  }
862  }
