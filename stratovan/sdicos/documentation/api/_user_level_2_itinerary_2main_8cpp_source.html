26 /**
27  * CreateCTWithItinerary shows how to use the itinerary API methods in IODCommon
28  * to create a CT DICOS file with itinerary information.
29  *
30  * In this example all itinerary attributes are set but in general
31  * the itinerary attributes are optional and can be left unset as needed.
32  */
33 bool CreateCTWithItinerary()
34 {
35  /// Create a CT Series for a person Object of Inspection (OOI)
36  /// For more details on using the User-Level API object SDICOS::CT see
37  /// the CTExample and UserCTExample
38  SDICOS::CT ct(
39  SDICOS::ObjectOfInspectionModule::enumTypePerson,
40  SDICOS::CTTypes::CTImage::enumHighEnergy,
41  SDICOS::CTTypes::CTImage::enumVolume,
42  SDICOS::ImagePixelMacro::enumMonochrome2 // Minimum pixel value is black
43  );
44  ct.EnableItinerary();
45  ct.SetGender(SDICOS::ObjectOfInspectionModule::enumGenderFemale);
46 
47  /// Acquisition time in seconds
48  ct.SetImageAcquisitionDuration(5.2);
49 
50  /// Add one section of volume data to the CT Series
51  /// Row and Column orientation vectors are (0,1,0) and (1,0,0)
52  /// CT Frame Type Pixel Data Characteristics value is set to ORIGINAL
53  /// CT Filter Type is set to NONE. CT Filter Material is not used.
54  ct.SetNumberOfSections(1);
55 
56  SDICOS::Section * s = ct.GetSectionByIndex(0);
57  s->SetFocalSpotSizeInMM(1.414f); /// Used nominal size of the focal spot in MM.
58  s->SetKVP(3.14f); /// Peak kilo voltage output of the x-ray generator
59  s->SetPositionInMM(10.0f, 20.0f, 15.0f); /// Volume data location in MM
60  s->SetSpacingInMM(1.0f, 2.0f, 10.0f); /// Column spacing, row spacing and slice thickness.
61 
62  // Set the Volume data for the CT Series section
63  SDICOS::Volume& volume = s->GetPixelData();
64  volume.Allocate(SDICOS::Volume::enumUnsigned16Bit,128,128,100);
65  SDICOS::Array3DLarge<SDICOS::S_UINT16>& sectionData = *volume.GetUnsigned16();
66 
67  SDICOS::S_UINT16 * userData = new SDICOS::S_UINT16[volume.GetWidth()*volume.GetHeight()];
68 
69  //Initialize with dummy values
70  memset(userData, 0, volume.GetWidth()*volume.GetHeight() * sizeof(SDICOS::S_UINT16));
71 
72  // Set volume data by setting each XY plane
73  for(SDICOS::S_UINT32 i = 0; i < volume.GetDepth(); ++i)
74  {
75  SDICOS::Array2D<SDICOS::S_UINT16>& xyPlane = sectionData[i];
76  SDICOS::S_UINT16 * pBuffer = xyPlane.GetBuffer();
77 
78  // The use can copy their data into pBuffer from another source or place it directly in pBuffer.
79  memcpy(pBuffer, userData, sizeof(SDICOS::S_UINT16)*xyPlane.GetWidth()*xyPlane.GetHeight());
80  }
81 
82  // Delete buffer
83  SDICOS::DELETE_ARRAY(userData);
84 
85  //
86  // Create an Itinerary for the Object of Inspection (OOI)
87  // An itinerary contains:
88  // - Identification information for the whole itinerary
89  // - One or more Route Segments with segment-specific information
90  //
91 
92  // Itinerary ID uniquely identifies the itinerary from a carrier, booking number etc.
93  SDICOS::DcsLongString itineraryID("123-345-567");
94  ct.SetItineraryID(itineraryID);
95 
96  // Indicate that the itineraryID came from an agent
97  // use enumCarrier to indicate that it came from the carrier handling the OOI
98  SDICOS::ItineraryModule::ITINERARY_ID_TYPE itineraryIDType(SDICOS::ItineraryModule::enumAgent);
99  ct.SetItineraryIDType(itineraryIDType);
100 
101  SDICOS::DcsLongString itineraryIDAssigningAuthority("Agent Smith");
102  ct.SetItineraryIDAssigningAuthority(itineraryIDAssigningAuthority);
103 
104  //
105  // Set up route segments in the itinerary
106  //
107  SDICOS::Array1D<SDICOS::ItineraryUser::ItineraryRouteSegment> vRouteSegments;
108 
109  SDICOS::DcsShortString routeSegmentID0("SFO to LAX");
110  SDICOS::ItineraryUser::ItineraryRouteSegment routeSegment0;
111 
112  routeSegment0.SetRouteSegmentID(routeSegmentID0);
113  routeSegment0.SetRouteID("Wormhole 1");
114  routeSegment0.SetInboundArrivalType(SDICOS::ItineraryModule::ItineraryRouteSegment::enumRailMetro);
115 
116  routeSegment0.SetCarrierID(SDICOS::DcsShortString("The Federation NCC1017A"));
117  routeSegment0.SetBoardingPassID(SDICOS::DcsShortString("Mamma Mia"));
118  routeSegment0.SetRouteSegmentStartLocationID(SDICOS::DcsShortString("SFO"));
119  routeSegment0.SetRouteSegmentEndLocationID(SDICOS::DcsShortString("LAX"));
120 
121  // Assigning authority for the Route Segment Start and End Location IDs.
122  routeSegment0.SetRouteSegmentLocationIdType(SDICOS::ItineraryModule::ItineraryRouteSegment::enumIATARouteSegmentLocation);
123 
124  SDICOS::DcsDateTime segmentStart(SDICOS::DcsDate::Today(),SDICOS::DcsTime::Now());
125  routeSegment0.SetRouteSegmentStartTime(segmentStart);
126 
127  SDICOS::DcsDateTime segmentEnd(SDICOS::DcsDate::Today(),SDICOS::DcsTime(12,15,0,0));
128  routeSegment0.SetRouteSegmentEndTime(segmentEnd);
129 
130  routeSegment0.SetAssignedLocation(SDICOS::DcsShortString("Seat A32"));
131  routeSegment0.SetInternationalRouteSegment(SDICOS::ItineraryModule::ItineraryRouteSegment::enumNo);
132 
133  vRouteSegments.Add(routeSegment0);
134 
135  // Set up a second route segment in the itinerary
136  SDICOS::DcsShortString routeSegmentID1 = "LAX to Pudong";
137  SDICOS::ItineraryUser::ItineraryRouteSegment routeSegment1;
138 
139  routeSegment1.SetRouteSegmentID(routeSegmentID1);
140  routeSegment1.SetRouteID("EVA 1025");
141  routeSegment1.SetInboundArrivalType(SDICOS::ItineraryModule::ItineraryRouteSegment::enumAir);
142 
143  routeSegment1.SetCarrierID(SDICOS::DcsShortString("NCC1017B"));
144  routeSegment1.SetBoardingPassID(SDICOS::DcsShortString("EVA123456"));
145  routeSegment1.SetRouteSegmentStartLocationID(SDICOS::DcsShortString("LAX"));
146  routeSegment1.SetRouteSegmentEndLocationID(SDICOS::DcsShortString("PUD"));
147 
148  // Assigning authority for the Route Segment Start and End Location IDs.
149  routeSegment1.SetRouteSegmentLocationIdType(SDICOS::ItineraryModule::ItineraryRouteSegment::enumIATARouteSegmentLocation);
150  routeSegment1.SetAssignedLocation(SDICOS::DcsShortString("Seat R10A"));
151  routeSegment1.SetInternationalRouteSegment(SDICOS::ItineraryModule::ItineraryRouteSegment::enumYes);
152 
153  vRouteSegments.Add(routeSegment1);
154 
155  ct.SetItineraryRouteSegments(vRouteSegments);
156 
157  SDICOS::ErrorLog errorlog;
158  SDICOS::Folder ctFolder("ItineraryExampleFiles");
159  SDICOS::Filename ctFilename(ctFolder,"CTWithItinerary.dcs");
160 
161  if(!ct.Write(ctFilename,errorlog,SDICOS::DicosFile::enumLittleEndianExplicit))
162  {
163  std::cerr << "Itinerary Example unable to write DICOS File " << ctFilename << std::endl;
164  std::cerr << errorlog << std::endl;
165  return false;
166  }
167  else
168  {
169  std::cout << "Itinerary Example wrote DICOS file " << ctFilename << std::endl;
170  SDICOS::CT ctRead;
171  if(ctRead.Read(ctFilename,errorlog))
172  {
173  if(ct != ctRead)
174  {
175  std::cerr << "Original CT and read CT are not equal" <<std::endl;
176  return false;
177  }
178 
179  // Check that itinerary data matches
180  bool bRes = (itineraryID == ctRead.GetItineraryID());
181  bRes = (itineraryIDAssigningAuthority == ctRead.GetItineraryIDAssigningAuthority()) && bRes;
182  bRes = (itineraryIDType == ctRead.GetItineraryIDType()) && bRes;
183 
184  SDICOS::Array1D<SDICOS::ItineraryUser::ItineraryRouteSegment> vRouteSegmentsRead;
185  bRes = ctRead.GetItineraryRouteSegments(vRouteSegmentsRead) && bRes;
186  bRes = (vRouteSegments == vRouteSegmentsRead) && bRes;
187 
188  if(!bRes)
189  {
190  std::cerr << "Failed to validate itinerary data from " << ctFilename << std::endl;
191  return false;
192  }
193  else
194  {
195  std::cerr << "Validated itinerary data from " << ctFilename << std::endl;
196  }
197  }
198  else
199  {
200  std::cerr << "Failed to read CT Series from " << ctFilename << std::endl;
201  std::cerr << errorlog << std::endl;
202  return false;
203  }
204  }
205 
206  return true;
207 }
208 ///EndCreateCTWithItinerary

211 {
212  bool bRes = CreateCTWithItinerary();
213  return (bRes ? 0 : 1);
214 }
