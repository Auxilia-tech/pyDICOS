32 /// Example of adding, finding, and removing attributes from
33 /// an attribute manager.
34 /// \code
35 void Add_Find_Remove_Attributes()
36 {
37  using namespace SDICOS;
38 
39  AttributeManager manager; ///< Attribute manager
40  AttributeManager::AttributePointer<AttributeString> attptrStr; ///< Pointer to AttributeString
41  AttributeManager::AttributePointer<AttributeString> attptrStrGet; ///< Pointer to AttributeString
42  Tag tag; ///< Tag
43 
44  tag.Set(0x1000, 0x0000); ///< Set tag for attribute to create
45 
46  /* Allocate an AttributeString object in the manager with the specified tag.
47  The third parameter, which is set to <b>true<\b>, indicates to allocate the attribute
48  if it is not found. */
49  manager.FindAttribute(tag, attptrStr, true);
50 
51  /* Set the size of the string array 1 and define the string as long string.
52  This creates an attribute with VM = 1 and VR = LO.*/
53  attptrStr->SetSize(1, AttributeString::enumLongString);
54 
55  (*attptrStr)[0] = "String"; /// Set the string to a value of "String"
56 
57  /* Find the attribute using an attribute pointer
58  The third parameter is not passed and left as default, which is <b>false<\b>.
59  The function will return <b>false<\b> is the attribute is not found. */
60  manager.FindAttribute(tag, attptrStrGet);
61 
62  manager.DeleteAttribute(tag); /// Delete the attribute
63 } // EndAdd_Find_Remove_Attributes

142 /// \section Sequences
143 /// Example of using sequences
144 /// \verbatim
145 void Sequences()
146 {
147  using namespace SDICOS;
148 
149  AttributeManager manager; ///< Attribute manager
150  AttributeManager::AttributePointer<AttributeSequence> attptrSQ; ///< Pointer to AttributeSequence
151  AttributeManager::AttributePointer<AttributeSequence> attptrSQsub; ///< Pointer to AttributeSequence
152  AttributeManager::AttributePointer<AttributeString> attptrStr; ///< Pointer to AttributeString
153  AttributeManager::AttributePointer<AttributeString> attptrStrGet; ///< Pointer to AttributeString
154  AttributeManager::AttributePointer<AttributeSignedInt16Bit> attptrSS; ///< Pointer to AttributeSignedInt16Bit
155  const Tag tagSQ(0x8000, 0x0002);
156  const Tag tagSQsub(0x2000, 0x0000);
157  const Tag tagSH(0x8001, 0x0000);
158  const Tag tagSS(0x8001, 0x0001);
159 
160  manager.FindAttribute(tagSQ, attptrSQ, true); ///< Allocate sequence
161 
162  /* Set number of sequence items to add.
163  Second parameter is the manager the sequence is allocated within. This is
164  only used to enable searching the the attribute manager when using <b>FindAttribute()<\b> with
165  sequence items.*/
166  attptrSQ->SetNumberOfItems(2, &manager);
167 
168  /* Each sequence item has the same functionality as an attribute manager */
169  (*attptrSQ)[0].FindAttribute(tagSS, attptrSS, true);
170  attptrSS->SetValue(5); ///< Set attribute value.
171 
172  /* Add attribute to the manager */
173  manager.FindAttribute(tagSH, attptrStr, true);
174  attptrStr->SetValue("string", AttributeString::enumShortString);
175 
176  /* Search for an attribute starting at a sequence item
177  Fourth parameter set to <b>true<\b> indicates to search parent attribute managers or parent sequence items
178  for the attribute if it is not found in the sequence item. In this case, it returns the attribute
179  found in <b>manager<\b>. */
180  (*attptrSQ)[0].FindAttribute(tagSH, attptrStrGet, false, true);
181 
182  /* Add a sequence to the sequence item */
183  (*attptrSQ)[1].FindAttribute(tagSQsub, attptrSQsub, true);
184 
185  /* Allocate 1 item within the sequence and set the parent sequence item as the parent manager. */
186  attptrSQsub->SetNumberOfItems(1, &(*attptrSQ)[1]);
187 
188  /* Finds the attribute in <b>manager<\b>. */
189  (*attptrSQsub)[0].FindAttribute(tagSH, attptrStrGet, false, true);
190 }///EndSequences
191 /// \endverbatim
