323  // Allocate a new buffer according to the capacity indicated by the current size
324  // and copy existing data to the new buffer if the new size is >= existing size.
325  // Adjust capacity accordingly.
326  bool Reallocate(const S_UINT32 nResizeAmount = 2, const bool bAllocateExtraCapacity = true)
327  {
328  CType * pOldData = m_pData;
329  const S_UINT32 nOldCapacity(m_nCapacity);
330 
331  // Adjust capacity so that at least nResizeAmount of space is available
332  // capacity = size * 1.5. Minimum nNewCapacity is 2.
333  const S_UINT32 nNewCapacity(bAllocateExtraCapacity ? CalcNewCapacity(SDICOS::Max(nResizeAmount, m_nSize)) : nResizeAmount);
334 
335  try
336  {
337  m_pData = new CType[nNewCapacity];
338  }
339  catch(std::bad_alloc&)
340  {
341  m_pData = NULL;
342  }
343 
344  if(m_pData)
345  {
346  try
347  {
348  m_nCapacity = nNewCapacity;
349 
350  // If the new capacity is >= old size, copy elements
351  if(nNewCapacity >= m_nSize)
352  {
353  for(S_UINT32 i = 0; i < m_nSize; ++i)
354  m_pData[i] = pOldData[i];
355  }
356  else
357  m_nSize = 0;
358 
359  if(pOldData)
360  DELETE_ARRAY( pOldData );
361 
362  m_bDeleteBuffer = true;
363  return true;
364  }
365  catch(...)
366  {
367  DELETE_ARRAY( m_pData );
368  m_pData = pOldData;
369  m_nCapacity = nOldCapacity;
370  return false;
371  }
372  }
373 
374  m_pData = pOldData;
375  return false;
376  }
