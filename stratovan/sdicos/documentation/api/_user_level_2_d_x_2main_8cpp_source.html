27 ///
28 /// Create a DX file that is intended for additional processing using the user-level API Object SDICOS::DX.
29 ///
30 bool CreateDXForProcessingSimple()
31 {
32  // Create a DX DICOS file for a Cargo Object of Inspection
33  // enumTypeCargo: OOI Type
34  // enumProcessing: The DICOS file is intended for further processing by ATR etc.
35  // enumOriginal: The DX Data is original data from the acquisition source
36  // enumMonochrome2: Indicates that low pixel values map to black.
37  SDICOS::DX dx(
38  SDICOS::ObjectOfInspectionModule::enumTypeCargo,
39  SDICOS::DXTypes::DXSeries::enumProcessing,
40  SDICOS::DXTypes::DXImage::enumOriginal,
41  SDICOS::ImagePixelMacro::enumMonochrome2
42  );
43 
44  SDICOS::S_UINT32 width(256), height(128);
45  SDICOS::Image2D& dxData = dx.GetXRayData();
46  dxData.Allocate(SDICOS::Image2D::enumUnsigned16Bit, width,height);
47 
48  // Data is in a 2D Array
49  SDICOS::Array2D<SDICOS::S_UINT16> &rawData = *dxData.GetUnsigned16();
50 
51  SDICOS::S_UINT16 count(0);
52  for(SDICOS::S_UINT32 row(0); row < dxData.GetHeight(); ++row)
53  {
54  for(SDICOS::S_UINT32 col(0); col < dxData.GetWidth(); ++col)
55  {
56  rawData[row][col] = count++;
57  }
58  }
59 
60  // Set other required attributes
61  dx.SetKVP(1.0f);
62  dx.SetImageOrientation(SDICOS::Vector3D<float>(1,0,0), SDICOS::Vector3D<float>(0,1,0));
63  dx.SetImagePosition(SDICOS::Point3D<float>(0,0,1));
64  dx.SetXRayTubeCurrent(2.5f);
65 
66  SDICOS::ErrorLog errorlog;
67  SDICOS::Folder dxFolder("DXFiles");
68  SDICOS::Filename dxFilename(dxFolder,"SimpleProcessingDX.dcs");
69 
70  if(!dx.Write(dxFilename,errorlog,SDICOS::DicosFile::enumLittleEndianExplicit))
71  {
72  std::cerr << "Simple DX Template Example unable to write DICOS File " << dxFilename << std::endl;
73  std::cerr << errorlog << std::endl;
74  return false;
75  }
76  else
77  {
78  std::cout << "Wrote file to " << dxFilename << std::endl;
79  SDICOS::DX dxRead;
80 
81  SDICOS::ErrorLog errorlog;
82  if(dxRead.Read(dxFilename,errorlog))
83  {
84  if(dxRead == dx)
85  {
86  std::cout << "Successfully read and compared DX files" << std::endl;
87  return true;
88  }
89  else
90  {
91  std::cout << "DX file loaded from " << dxFilename << " does not match original." << std::endl;
92  return false;
93  }
94  }
95  else
96  {
97  std::cerr << "Unable to read DX file " << dxFilename << std::endl;
98  std::cerr << errorlog << std::endl;
99  return false;
100  }
101  }
102 } /// EndCreateDXForProcessingSimple

105 /// Create a DX file intended for viewing.
106 /// DX Files for viewing require LUT information.
107 ///
108 bool CreateDXForPresentationSimple()
109 {
110  // Create a DX DICOS file for a Cargo Object of Inspection
111  // enumTypeBioSample: OOI Type
112  // enumPresentation: The DICOS file is intended for presentation at a viewing station
113  // enumOriginal: The DX Data is original data from the acquisition source
114  // enumMonochrome2: Indicates that low pixel values map to black.
115  SDICOS::DX dx(
116  SDICOS::ObjectOfInspectionModule::enumTypeBioSample,
117  SDICOS::DXTypes::DXSeries::enumPresentation,
118  SDICOS::DXTypes::DXImage::enumOriginal,
119  SDICOS::ImagePixelMacro::enumMonochrome2
120  );
121 
122  SDICOS::S_UINT32 width(256), height(128);
123  SDICOS::Image2D& dxData = dx.GetXRayData();
124  dxData.Allocate(SDICOS::Image2D::enumUnsigned16Bit,width,height);
125 
126  // Data is in a 2D Array
127  SDICOS::Array2D<SDICOS::S_UINT16> &rawData = *dxData.GetUnsigned16();
128 
129  SDICOS::S_UINT16 count(0);
130  for(SDICOS::S_UINT32 row(0); row < dxData.GetHeight(); ++row)
131  {
132  for(SDICOS::S_UINT32 col(0); col < dxData.GetWidth(); ++col)
133  {
134  rawData[row][col] = count++;
135  }
136  }
137 
138  // DX Files for presentation require a LUT.
139  // The example code demonstrates two ways to specify the LUT but only one is required.
140 
141  // LUT setup using Window Center and Width to define a linear transformation.
142  // The input range [734,1734]
143  dx.SetWindowCenterAndWidth(1234, 500);
144 
145  float windowCenter,windowWidth;
146  dx.GetWindowCenterAndWidth(windowCenter,windowWidth);
147  std::cout << "Center and Width " << windowCenter << " " << windowWidth << std::endl;
148 
149  // The explanation is optional
150  dx.SetWindowCenterAndWidthExplanation("Window Center and Width for test screen");
151 
152  // LUT Setup using an array of values
153  SDICOS::VoiLut::UnsignedLUT lut;
154  SDICOS::S_UINT32 lutSize = 512;
155  SDICOS::S_UINT16 lutMinValue = 1000;
156  lut.SetSize(lutSize);
157  for(unsigned int i(0); i < lutSize; ++i)
158  {
159  // made up LUT function for example purposes.
160  lut[i] = lutMinValue + (unsigned int)(sqrt(float(i)));
161  }
162 
163  // The LUT maps values in the input range [firstValueMapped, firstValueMaped+lutSize]
164  // to output range starting a lutMinValue.
165  // Values below firstValueMapped map to lutMinValue.
166  // Values above (firstValueMapped+lutSize) map to the last value in the lut.
167  SDICOS::S_UINT16 firstValueMapped = 2000;
168  dx.SetLUTData(lut,firstValueMapped);
169 
170  // Set other required attributes
171  dx.SetKVP(1.0f);
172  dx.SetImageOrientation(SDICOS::Vector3D<float>(1,0,0), SDICOS::Vector3D<float>(0,1,0));
173  dx.SetImagePosition(SDICOS::Point3D<float>(0,0,1));
174  dx.SetXRayTubeCurrent(2.5f);
175 
176  SDICOS::ErrorLog errorlog;
177  SDICOS::Folder dxFolder("DXFiles");
178  SDICOS::Filename dxFilename(dxFolder,"SimplePresentationDX.dcs");
179 
180  if(!dx.Write(dxFilename,errorlog))
181  {
182  std::cerr << "Simple DX Template Example unable to write DICOS File " << dxFilename << std::endl;
183  std::cerr << errorlog << std::endl;
184  return false;
185  }
186  else
187  {

217  /// Create a DX file using a color palette that is intended for additional processing using the user-level API Object SDICOS::DX.
218  ///
219 bool CreateDXWithColorPaletteForProcessingSimple()
220 {
221  // Create a DX DICOS file for a Cargo Object of Inspection
222  // enumTypeCargo: OOI Type
223  // enumProcessing: The DICOS file is intended for further processing by ATR etc.
224  // enumOriginal: The DX Data is original data from the acquisition source
225  // enumPaletteColor: Indicates that the a color palette is used.
226  SDICOS::DX dx(
227  SDICOS::ObjectOfInspectionModule::enumTypeCargo,
228  SDICOS::DXTypes::DXSeries::enumProcessing,
229  SDICOS::DXTypes::DXImage::enumOriginal,
230  SDICOS::ImagePixelMacro::enumPaletteColor);
231 
232  // Identity: Low to high values represent dark to light
233  // Inverse: Low to high values represent light to dark
234  if (!dx.SetPresentationLUTShape(SDICOS::GeneralImageModule::enumIdentity))
235  {
236  std::cout << "Invalid Presentation LUT Shape" << std::endl;
237  return false;
238  }
239 
240  //Dimensions of image
241  const SDICOS::S_UINT32 nWidth(256), nHeight(128);
242 
243  //Set up Red, Green, and Blue color palette
244  //These values are indexed into by the Pixel Data values
245  SDICOS::Array1D<SDICOS::S_UINT8> &vRed = dx.GetRedLUT();
246  SDICOS::Array1D<SDICOS::S_UINT8> &vGreen = dx.GetGreenLUT();
247  SDICOS::Array1D<SDICOS::S_UINT8> &vBlue = dx.GetBlueLUT();
248 
249  //Set all to same size. This example will use a single color for each row.
250  vRed.SetSize(nHeight);
251  vGreen.SetSize(nHeight);
252  vBlue.SetSize(nHeight);
253 
254  SDICOS::S_UINT8 *pnRed = vRed.GetBuffer();
255  SDICOS::S_UINT8 *pnGreen = vGreen.GetBuffer();
256  SDICOS::S_UINT8 *pnBlue = vBlue.GetBuffer();
257  SDICOS::S_UINT8 nColor(0);
258  for (SDICOS::S_UINT16 nIndex(0); nIndex < nHeight; ++nIndex, ++pnRed, ++pnGreen, ++pnBlue)
259  {
260  //Setting RGB to (nColor, nColor/2, nColor)
261  *pnRed = nColor;
262  *pnGreen = nColor/2;
263  *pnBlue = nColor;
264 
265  ++nColor;
266  }
267 
268  SDICOS::Image2D& imgPixelData = dx.GetXRayData();
269  imgPixelData.Allocate(SDICOS::Image2D::enumUnsigned16Bit, nWidth, nHeight);
270 
271  // Pixel Data is set of indices into the Red, Green, and Blue LUT's
272  SDICOS::Array2D<SDICOS::S_UINT16> &vIndexData = *imgPixelData.GetUnsigned16();
273 
274  for (SDICOS::S_UINT32 row(0); row < imgPixelData.GetHeight(); ++row)
275  {
276  for (SDICOS::S_UINT32 col(0); col < imgPixelData.GetWidth(); ++col)
277  {
278  vIndexData[row][col] = row; //Sets each column in a row to the same color
279  }
280  }
281 
282  // Set other required attributes
283  dx.SetKVP(1.0f);
284  dx.SetImageOrientation(SDICOS::Vector3D<float>(1,0,0), SDICOS::Vector3D<float>(0,1,0));
285  dx.SetImagePosition(SDICOS::Point3D<float>(0,0,1));
286  dx.SetXRayTubeCurrent(2.5f);
287 
288  SDICOS::ErrorLog errorlog;
289  SDICOS::Folder dxFolder("DXFiles");
290  SDICOS::Filename dxFilename(dxFolder, "SimpleColorPaletteDX.dcs");
291 
292  if (!dx.Write(dxFilename, errorlog, SDICOS::DicosFile::enumLittleEndianExplicit))
293  {
294  std::cerr << "Color Palette DX Example unable to write DICOS File " << dxFilename << std::endl;
295  std::cerr << errorlog << std::endl;
296  return false;
297  }
