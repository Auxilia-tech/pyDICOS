30 /*
31  * The XRay Equipment object is the module-level API object that represent the
32  * XRay Equipment in the DICOS Specification Section 14.
33  *
34  * #include "SDICOS/XRayEquipmentModule.h"
35  *
36  * All objects in the Stratovan DICOS Toolkit are declared in the SDICOS namespace.
37  */
38 bool CreateXRayEquipment(SDICOS::XRayEquipmentModule& xr)
39 {
40  /*
41  * High and low-level API methods generally return true/false to indicate
42  * success and failure.
43  */
44  bool bRes(true);
45 
46  /*
47  * XRay Equipment has one required Floating Point Single attribute.
48  */
49  float energy = 0;
50  bRes = bRes && xr.SetEffectiveEnergy(energy);
51 
52  /*
53  * XRay Equipment has one optional Short Text attribute.
54  */
55  DcsShortText st = "Short Text Test.";
56  bRes = bRes && xr.SetMaskCharacteristics(st);
57 
58  /*
59  * XRay Equipment has five optional Code String attributes.
60  * These attributes are represented by enums.
61  */
62 
63  // Type of G0
64  bRes = bRes && xr.SetTypeG0(XRayEquipmentModule::enumGratingG0);
65  bRes = bRes && xr.SetTypeG0(XRayEquipmentModule::enumMaskG0);
66 
67  // Type of G1
68  bRes = bRes && xr.SetTypeG1(XRayEquipmentModule::enumGratingG1);
69  bRes = bRes && xr.SetTypeG1(XRayEquipmentModule::enumOtherG1);
70 
71  // Type of G2
72  bRes = bRes && xr.SetTypeG2(XRayEquipmentModule::enumGratingG2);
73  bRes = bRes && xr.SetTypeG2(XRayEquipmentModule::enumOtherG2);
74 
75  // Reconstruction Algorithm for Phase Image
76  bRes = bRes && xr.SetReconAlgorithmPhaseImage(XRayEquipmentModule::enumFFT);
77  bRes = bRes && xr.SetReconAlgorithmPhaseImage(XRayEquipmentModule::enumModifiedFFT);
78  bRes = bRes && xr.SetReconAlgorithmPhaseImage(XRayEquipmentModule::enumOtherImage);
79 
80  // Reconstruction Algorithm for Dark Image
81  bRes = bRes && xr.SetReconAlgorithmDarkImage(XRayEquipmentModule::enumFFT);
82  bRes = bRes && xr.SetReconAlgorithmDarkImage(XRayEquipmentModule::enumModifiedFFT);
83  bRes = bRes && xr.SetReconAlgorithmDarkImage(XRayEquipmentModule::enumOtherImage);
84 
85  /*
86  * XRay Equipment has thirteen optional Floating Point Single attributes.
87  */
88 
89  // Distance from X-Ray source output to the G0 grating in CM.
90  bRes = bRes && xr.SetDistanceFromG0ToSourceInCentimeters(0.0f);
91  bRes = bRes && xr.SetDistanceFromG1ToSourceInCentimeters(1.0f);
92  bRes = bRes && xr.SetG0AbsorptionPeriod(3.0f);
93  bRes = bRes && xr.SetG1PhasePeriod(4.0f);
94  bRes = bRes && xr.SetG2AnalyzerPeriod(5.0f);
95  bRes = bRes && xr.SetVisibility(6.0f);
96  bRes = bRes && xr.SetAbsorptionXRDetectorSize(7.0f, 9.0f);
97  bRes = bRes && xr.SetAbsorptionReconstructionVoxelSize(9.0f, 10.0f);
98  bRes = bRes && xr.SetPhaseXRDetectorSize(11.0f, 12.0f);
99  bRes = bRes && xr.SetPhaseReconstructionVoxelSize(13.0f, 14.0f, 15.0f);
100  bRes = bRes && xr.SetDarkXRDetectorSize(16.0f, 17.0f);
101  bRes = bRes && xr.SetDarkReconstructionVoxelSize(18.0f, 19.0f, 20.0f);
102 
103  return bRes;
104 }

106 bool CreateDX()
107 {
108  /*
109  * The DXModule object is the module-level API object that represent the
110  * Digital X-Ray (CT) Information Object Definition (IOD) in the DICOS Specification Section 8.
111  *
112  * #include "SDICOS/ModuleDX.h"
113  *
114  * All objects in the Stratovan DICOS Toolkit are declared in the SDICOS namespace.
115  * The SDICOS::DXTypes namespace contains classes used to implement the DICOS modles in the DX IOD.
116  */
117  SDICOS::DXModule dx;
118 
119  //
120  // DX Data is stored as a buffer of unsigned shorts
121  //
122  // In this example, the DX data is stored as a 3D array which
123  // is implemented as a 1-D array of 2-D arrays so (X,Y) slices and contiguous blocks of memory.
124  //
125  Array3DLarge<unsigned short> vnVolume;
126 
127  // The DX Image only has 1 slice.
128  SDICOS::S_UINT32 width(128),height(128),depth(1);
129  vnVolume.SetSize(width,height,depth);
130 
131  // Buffer for volume slice that contains user data
132  SDICOS::S_UINT16 * sourceSlice = new SDICOS::S_UINT16[width*height];
133 
134  //Initialize with dummy data
135  memset(sourceSlice, 0, width*height*sizeof(SDICOS::S_UINT16));
136 
137  for(SDICOS::S_UINT32 z = 0; z < depth; ++z)
138  {
139  // 3D array is an array of 2D slices
140  // Access raw data for the slice
141  SDICOS::S_UINT16 * xyPlane = vnVolume[z].GetBuffer();
142 
143  // Set data for each element individually
144  for(SDICOS::S_UINT32 i = 0; i < (width*height); ++i)
145  xyPlane[i] = z;
146 
147  // Alternatively set slice data using memcpy
148  memcpy(xyPlane, sourceSlice, width*height*sizeof(SDICOS::S_UINT16));
149  }
150 
151  delete [] sourceSlice;
152 
153  /*
154  * Origin and Spacing are stored in the SDICOS::CTTypes::CTMultiframeFunctionalGroupMacros
155  */
156  Point3D<float> ptOrigin;
157  Point3D<float> ptSpacing;
158 
159  /// Indicate that this DX file is intended for viewing.
160  /// Per the DICOS Specification Section
161  /// A series of images intended for viewing by an observer, after application of any grayscale
162  /// transformations specified in the image object such as VOI LUT, shall have an Enumerated Value
163  /// of "FOR PRESENTATION"
164  ///
165  /// Sets DICOS SOP Class UID (0008, 0016) to "For Presentation" and
166  /// Presentation Intent Type (0008, 0068) to "FOR PRESENTATION"
167  dx.SetForPresentation();
168 
169  /*
170  * High and low-level API methods generally return true/false to indicate
171  * success and failure.
172  */
173  bool bRes(true);
174 
175  /*
176  * DcsdDateTime object takes data as "YYYYMMDD" and time as "HHMMSS"
177  */
178  SDICOS::DcsDateTime dateTime;
179  SDICOS::DcsDate date;
180  SDICOS::DcsTime time;
181  date.SetToday();
182  time.SetNow();
183  dateTime.Set(date,time);
184 
185  /*
186  * The DX Series class hold data for the Series Information Entity in the DX IOD.
187  * It includes Attributes for the DX Series Module nand the"General Series" module
188  *
189  * #include "SDICOS/GeneralSeriesModule.h"
190  * #include "SDICOS/DXSeries.h"
191  */
192  SDICOS::DXTypes::DXSeries& series = dx.GetSeries();
193  bRes = bRes && series.SetAcquisitionStatus(DXTypes::DXSeries::enumSuccessful);
194  bRes = bRes && series.SetDateAndTime(date,time);
195 
196  //
197  // Set the Unique ID of the Scan for the DX Series
198  // Create a unique identifier from a user defined string.
199  //
200  SDICOS::DcsUniqueIdentifier instanceUID("1.2.276.0.7230010.3.1.3.4126100761.8984.1407370732.884");
201  bRes = bRes && series.SetInstanceUID(instanceUID);
202 
203  /*
204  * The DX Image class represents the DX Image Module
205  *
206  * See "SDICOS/DXImage.h"
207  *
208  * Currently the data is copied from the given array into the
209  * CTImage module.
210  */
211  SDICOS::DXTypes::DXImage& image = dx.GetImage();
212  bRes = bRes && image.SetBurnedInAnnotation(true);
213  bRes = bRes && image.SetContentTimeAndDate(date,time);
214 
215  /*
216  * DX Image type has two mandatory values.
217  */
218  bRes = bRes && image.SetImageType( DXTypes::DXImage::enumOriginal,
219  DXTypes::DXImage::enumIntensity);
220 
221  /*
222  * Instance Number, is in the General Image Module
223  */
224  bRes = bRes && image.SetInstanceNumber(1);
225  bRes = bRes && image.SetPhotometricInterpretation(DXTypes::DXImage::enumMonochrome2);
226 
227  // Select slice 0 for this DX Image
228  bRes = bRes && image.SetPixelData(vnVolume[0]);
229  bRes = bRes && image.SetPixelIntensityRelationship(DXTypes::DXImage::enumLinear);
230  bRes = bRes && image.SetPixelIntensitySign(DXTypes::DXImage::enumLowValueLowIntensity);
231  bRes = bRes && image.SetRescaleType("HU");
232  bRes = bRes && image.SetPresentationLutShape(GeneralImageModule::enumIdentity);
233 
234  //
235  // VOI Lut object is required because the DX object is for presentation
236  // Presentation Type is "FOR PRESENTATION"
237  //
238  SDICOS::VoiLut& voiLut = image.GetVoiLut();
239  bRes = bRes && voiLut.SetWindowCenterAndWidth(2048, 4096);
240 
241  /*
242  * Note:
243  * The DXTypes::DXPositioning class that represents the DX Position Module
244  * is type 'U' User Optional and is not used in this example.
245  *
246  * The DXTypes::DXDetector class that represents the DX Detector Module
247  * is required but all attributes are Type 2 (Required but can be NULL) or Type 3 (Optional)
248  * so none of the attributes are explicitly set.
249  */
250 
251  /*
252  * The XRayGeneration class represents X-Ray Generation module
253  * Required information for how x-rays were generated during the x-ray image acquisition process.
254  */
255  SDICOS::DXTypes::XRayGeneration& xRayGeneration = dx.GetXRayGeneration();
256 
257  bRes = bRes && xRayGeneration.SetImageOrientation(1,0,0, 0,1,0);
258  bRes = bRes && xRayGeneration.SetImagePosition(ptOrigin.x,ptOrigin.y,ptOrigin.z);
259  bRes = bRes && xRayGeneration.SetKVP(0);
260  bRes = bRes && xRayGeneration.SetXRayTubeCurrent(0);
261 
262  /*
263  * The ObjectOfInspectionModule class represents Object Of Inspection Module (OOI)
264  */
265  SDICOS::ObjectOfInspectionModule& objectOfInspection = dx.GetObjectOfInspection();
266  SDICOS::ObjectOfInspectionModule::IdInfo& idInfo = objectOfInspection.GetID();
267 
268  bRes = bRes && idInfo.SetID("TSS4.000314");
269  bRes = bRes && idInfo.SetIdAssigningAuthority("TSA");
270  bRes = bRes && idInfo.SetIdType(ObjectOfInspectionModule::IdInfo::enumText);
271  bRes = bRes && objectOfInspection.SetType(ObjectOfInspectionModule::enumTypeBaggage);
272 
273  /*
274  * The GeneralScanModule class represents the General Scan Module
275  */
276  SDICOS::GeneralScanModule& generalScan = dx.GetGeneralScan();
277  bRes = bRes && generalScan.SetScanID("TSS4.314");
278 
279  // Create a randomized unique identifier
280  SDICOS::DcsUniqueIdentifier scanInstanceUID = SDICOS::DcsUniqueIdentifier::CreateGUID();
281  bRes = bRes && generalScan.SetScanInstanceUID(scanInstanceUID);
282 
283  bRes = bRes && generalScan.SetStartDateAndTime(date,time);
284  bRes = bRes && generalScan.SetType(GeneralScanModule::enumOperational);
285 
286  /*
287  * The GeneralEquipmentModule class represent the General Equipment Module
288  * Required information about the machine the scan was captured on.
289  */
290  SDICOS::GeneralEquipmentModule& generalEquipment = dx.GetGeneralEquipment();
291  bRes = bRes && generalEquipment.SetCalibrationDateAndTime("20140130", "130532.435");
292  bRes = bRes && generalEquipment.SetDeviceSerialNumber("007");
293  bRes = bRes && generalEquipment.SetMachineAddress("Davis, CA");
294  bRes = bRes && generalEquipment.SetMachineID("Enigma");
295  bRes = bRes && generalEquipment.SetMachineLocation("Dungeon");
296  bRes = bRes && generalEquipment.SetManufacturer("Stratovan");
297  bRes = bRes && generalEquipment.SetManufacturerModelName("Foo Bar");
298  bRes = bRes && generalEquipment.SetSoftwareVersion("2.718");
299 
300  /*
301  XRayEquipmentModule represents the X-Ray Equipment Module
302  Attribute related to the device's grating and detectors dusing X-Ray Image acquisition.
303  */
304  SDICOS::XRayEquipmentModule& XRayEquipment = dx.GetXRayEquipment();
305  bRes = bRes && CreateXRayEquipment(XRayEquipment);
306 
307  /*
308  * The SOPCommonModule class represents the SOP Common Module
309  * which defines attributes for identifying associated Service Object Pair (SOP) instances
310  */
311  SDICOS::SopCommonModule& sopCommon = dx.GetSopCommon();
312  bRes = bRes && sopCommon.SetSopInstanceCreationDateAndTime(date,time);
313 
314  SDICOS::DcsUniqueIdentifier sopInstanceUID;
315  sopInstanceUID.SetGUID();
316  bRes = bRes && sopCommon.SetSopInstanceUID(sopInstanceUID);
317 
318  /*
319  * The CommonInstanceReferenceModule class represents the Common Instance Reference Module
320  * which describes the relationships between SOP Instances referenced between modules in the same instance.
321  * In this example, we have one scan with one series.
322  */
323  SDICOS::CommonInstanceReferenceModule& commonInstance = dx.GetCommonInstanceReference();
324  commonInstance.AllocateReferencedSeries();
325 
326  SDICOS::ReferencedSeriesSequence * refSeriesSeq = commonInstance.GetReferencedSeries();
327  bRes = bRes && (S_NULL != refSeriesSeq);
328  bRes = bRes && refSeriesSeq->GetReferencedSeries().SetSize(1);
329 
330  if(refSeriesSeq)
331  {
332  /*
333  * Note that this series instance UID matches the one set above on the DX object
334  * with the call to dx.GetSeries().SetInstanceUID(...)
335  */
336  SDICOS::Array1D<SDICOS::ReferencedSeriesSequence::ReferencedSeries> & referencedSeries = refSeriesSeq->GetReferencedSeries();
337 
338  bRes = bRes && referencedSeries[0].SetSeriesInstanceUID(instanceUID);
339 
340  SDICOS::Array1D<SDICOS::SopInstanceReference>& referencedInstances = referencedSeries[0].GetReferencedInstances();
341  bRes = bRes && referencedInstances.SetSize(1);
342  bRes = bRes && referencedInstances[0].SetClassUID(SOPClassUID::GetDXPresentation());
343 
344  // Specify UID with user defined string
345  bRes = bRes && referencedInstances[0].SetInstanceUID("1.2.276.0.7230010.3.1.4.4126100761.8984.1407370732.885");
346  }
347 
348  const Filename filename("DXspimage.dcs");
349 
350  // Alternatively a Filename can be specified with a folder and name
351 #if 0
352  Folder folder("C:\temp");
353  Filename(folder,"DXspimage.dcs");
354 #endif
355 
356  /*
357  * The ErrorLog class stores warning and errors from the validation process
358  */
359  ErrorLog errorlog;
360 
361  /*
362  * Writing the DX file validates the DX object and reports errors in the
363  * supplied ErrorLog.
364  */
365  if(!dx.Write(filename, DicosFile::enumLittleEndianExplicit, errorlog))
366  {
367  errorlog.WriteLog("DXErrorLog.txt");
368  std::cerr << "Failed to write DX DICOS File " << filename << " Writing errors to DXErrorLog.txt" << std::endl;
369  std::cerr << errorlog << std::endl;
370  bRes = false;
371  }
372  else
373  {
374  std::cout << "Write DX DICOS File " << filename << std::endl;
375  }
376 
377  /*
378  * Reading the DICOS file
379  */
380  AttributeManager manager;
381  errorlog.FreeMemory();
382  if(DicosFile::Read(filename, errorlog, manager))
383  {
384  manager.WriteToFileAsText("DXspimageListings.txt");
385  std::cout << "Write Attribute Listing to DXspimageListings.txt" << std::endl;
386  }
387  else
388  {
389  std::cerr << "Failed to read as DICOS file " << filename << std::endl;
390  std::cerr << errorlog << std::endl;
391  bRes = false;
392  }
393 
394 
395  SDICOS::DXModule dxRead;
396  ErrorLog errorlogRead;
397 
398  if(!dxRead.Read(filename, errorlogRead))
399  {
400  std::cerr << "Failed to read as DX object " << filename << std::endl;
401  std::cerr << errorlogRead << std::endl;
402  bRes = false;
403  }
404  else
405  {
406  std::cout << "ReadDX from " << filename << std::endl;
407  bRes = bRes && dx == dxRead;
408  if(bRes)
409  {
410  std::cout << "Verified written and read DX files are equivalent" << std::endl;
411  }
412  else
413  {
414  std::cerr << "Failed to verify that written and read DX files are equivalent" << std::endl;
415  }
416  }
417 
418  return bRes;
419 }
