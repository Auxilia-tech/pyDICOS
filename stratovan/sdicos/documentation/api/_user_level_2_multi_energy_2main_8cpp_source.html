25 // This example creates CT objects for multiple energy levels and a 2D SP image
26 // and links a TDR to those CT objects.
27 // DICOS does not support holding multiple energy levels of the same volume
28 // in a single object. Each energy level of the same volume must be stored as a
29 // separate CT object.
30 //

45  SDICOS::CT ctSP; //CT object for 2D SP
46  SDICOS::TDR tdr; //TDR

59 
60  //Set same frame of reference

62 
63  //Date and time of volume acquisition
64  ctHighEnergy.SetImageAcquisitionDateAndTime(SDICOS::DcsDate::Today(), SDICOS::DcsTime::Now());
65  ctLowEnergy.SetImageAcquisitionDateAndTime(SDICOS::DcsDate::Today(), SDICOS::DcsTime::Now());
66  ctSP.SetImageAcquisitionDateAndTime(SDICOS::DcsDate::Today(), SDICOS::DcsTime::Now());

70  ctLowEnergy.SetImageAcquisitionDuration(5000);
71  ctSP.SetImageAcquisitionDuration(5000);
72 
73  //Low values in volume data correspond to low intensity values
74  ctHighEnergy.SetPhotometricInterpretation(SDICOS::ImagePixelMacro::enumMonochrome2);
75  ctLowEnergy.SetPhotometricInterpretation(SDICOS::ImagePixelMacro::enumMonochrome2);
76  ctSP.SetPhotometricInterpretation(SDICOS::ImagePixelMacro::enumMonochrome2);
77 
78  //Generate series UID's
79  ctHighEnergy.GenerateSeriesInstanceUID();
80  ctLowEnergy.GenerateSeriesInstanceUID();
81  ctSP.GenerateSeriesInstanceUID();

90  ctHighEnergy.SetScanID("ScanID"); //Scan ID
91  ctLowEnergy.SetOOIID(ctHighEnergy.GetOOIID());

156

158  tdr.SetProcessingStartTime(nPTOIdentifier, SDICOS::DcsDateTime(SDICOS::DcsDate::Today(), SDICOS::DcsTime::Now()));

163  tdr.SetBaggagePTODensityInGramsPerCC(nPTOIdentifier, 0.05f, nHighEnergyRepresentation);
164 
165  tdr.SetBaggagePTODetails(nPTOIdentifier, //ID for PTO
166  1000, //PTO mass in grams
167  5000, //PTO density in grams per CC
168  0); //Index for which PTO Representation Sequence Item
169  tdr.SetBaggagePTOVolumeInCC(nPTOIdentifier, 100.0f, 0); //PTO volume in CC
170 
171  //High Energy Values
172  {
173  tdr.SetThreatRegionOfInterest(nPTOIdentifier, //ID for PTO
174  SDICOS::Point3D<float>(0,0,0), //Voxel offset to PTO
175  SDICOS::Point3D<float>(10,10,10), //Voxel dimensions of PTO
176  SDICOS::Bitmap(), //Entire region marked as threat
177  nHighEnergyRepresentation); //High energy representation index
178 
179  tdr.SetBaggagePTOLocationDescription(nPTOIdentifier, "High Energy PTO", nHighEnergyRepresentation); //PTO description
180  }
181  //Low Energy Values
182  {
183  tdr.SetThreatRegionOfInterest(nPTOIdentifier, //ID for PTO
184  SDICOS::Point3D<float>(20,20,20), //Voxel offset to PTO
185  SDICOS::Point3D<float>(5,20,5), //Voxel dimensions of PTO
186  SDICOS::Bitmap(), //Entire region marked as threat
187  nLowEnergyRepresentation); //Low energy representation index
188 
189  tdr.SetBaggagePTOLocationDescription(nPTOIdentifier, "Low Energy PTO", nLowEnergyRepresentation); //PTO description
190  }
191  tdr.SetInstanceNumber(0); //Instance number
192  tdr.SetContentDateAndTime(SDICOS::DcsDate::Today(), SDICOS::DcsTime::Now()); //Date and time of content creation
193  tdr.SetTDRType(SDICOS::TDRTypes::ThreatDetectionReport::enumMachine); //TDR generated by ATR machine
194  tdr.SetTDRTypeATR("ATR Manufacturer", "ATR Version"); //ATR manufacturer and version
195 
196  //Relative scaling between the representative figure and the image stored in the PTO
197  tdr.SetImageScaleRepresentation(1);
198 
199  tdr.SetAlarmDecision(SDICOS::TDRTypes::ThreatDetectionReport::enumClear); //Alarm decision
200  tdr.SetAlarmDecisionDateTime(SDICOS::DcsDate::Today(), SDICOS::DcsTime::Now()); //Date and time of alarm decision
201  tdr.SetAbortFlag(SDICOS::TDRTypes::ThreatDetectionReport::enumSuccess); //No abort
202  tdr.SetTotalProcessingTimeInMS(2500); //Processing time for TDR
203 
204  ////////////////////////////
205  //Write files to disk

261 template<typename TYPE>
262 bool SetGeneralInformation(TYPE &mod)
263 {
264  bool bRes(true);
265  const SDICOS::DcsDate strDate(SDICOS::DcsDate::Today()); //Constant for today's date
266  const SDICOS::DcsTime strTime(SDICOS::DcsTime::Now()); //Constant for current time
267 
268  bRes = bRes && mod.SetContentDateAndTime(strDate, strTime); //Date and time of content acquisition
269  bRes = bRes && mod.SetOOIIDAssigningAuthority("OOI ID Assigning Authority"); //Name of OOI ID assigning authority
270  bRes = bRes && mod.SetOOIIDType(SDICOS::ObjectOfInspectionModule::IdInfo::enumRFID); //OOI ID corresponds to an RFID
271  bRes = bRes && mod.SetOOIType(SDICOS::ObjectOfInspectionModule::enumTypeBaggage); //OOI is baggage
272  bRes = bRes && mod.SetScanStartDateAndTime(strDate, strTime); //Date and time scan started
273  bRes = bRes && mod.SetScanType(SDICOS::GeneralScanModule::enumOperational); //Operational scan
274  bRes = bRes && mod.SetSeriesDateAndTime(strDate, strTime); //Date and time of series creation
275  bRes = bRes && mod.SetSeriesAcquisitionStatus(SDICOS::GeneralSeriesModule::enumSuccessful); //Acquisition was successfully captured
276  bRes = bRes && mod.SetDeviceCalibrationDateAndTime(strDate, strTime); //Date and time of device calibration
277  bRes = bRes && mod.SetDeviceSerialNumber("123456987"); //Device serial number
278  bRes = bRes && mod.SetMachineAddress("Machine Address"); //Machine address (Physical address)
279  bRes = bRes && mod.SetMachineLocation("Machine Location");//Machine location (Location on site)
280  bRes = bRes && mod.SetMachineID("Machine ID"); //Machine ID
281  bRes = bRes && mod.SetDeviceManufacturer("Device Manufacturer"); //Device manufacturer
282  bRes = bRes && mod.SetDeviceManufacturerModelName("Device Manufacturer Model Name"); //Model name
283  bRes = bRes && mod.SetDeviceSoftwareVersion("Device Software Version"); //Device's software version
284  bRes = bRes && mod.SetSopInstanceCreationDateAndTime(strDate, strTime); //Date and time of SOP instance creation
285 
286  return bRes;
287 }

305  secSP.SetSpacingInMM(1.0f,1.0f,0.0f);
306

308  secHigh.SetPositionInMM(0,0,0);
