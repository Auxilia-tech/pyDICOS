34 /// \class IClientAuthenticationCallback
35 /// \brief Callback interface for user name and passcode authentication. Functions called when a connecting client
36 /// presents a required user name and/or passcode. All functions are asynchronous as they are called from a
37 /// separate thread.
38 class IClientAuthenticationCallback
39 {
40 public:
41 
42  /// Called when the server must authenticate a user name
43  bool AuthenticateUserName(const SDICOS::Utils::AuthenticationData &ad)const;
44 
45  /// Called when the server must authenticate a user name and passcode
46  bool AuthenticateUserNameAndPasscode(const SDICOS::Utils::AuthenticationData &ad)const;
47 
48  /// Called when the server must authenticate the client's application name
49  bool AuthenticateClientApplicationName(const SDICOS::Utils::AuthenticationData &ad)const;
50 
51  /// Called when a client connects to a server (TCP/IP connection, not a DICOS session)
52  void ConnectedToClient(const SDICOS::Utils::SessionData &sd);
53  /// Called when a client disconnects from a server (TCP/IP connection, not a DICOS session)
54  void DisconnectedFromClient(const SDICOS::Utils::SessionData &sd);
55 
56  /// Called after client authentication
57  void DicosConnectionStarted(const SDICOS::Utils::SessionData &sd);
58  /// Called when the DICOS session ends
59  void DicosConnectionStopped(const SDICOS::Utils::SessionData &sd);
60 protected:
61 
62  /// Override to authenticate user name. Return true if the user name is accepted. Return false if user name is rejected.
63  /// An accepted user name allows the client to connect to the server and process received data. A rejected user name
64  /// prevents the client from connecting to the server.
65  /// If not overridden, function passes calls to OnAuthenticateUserNameAndPasscode with an empty passcode
66  virtual bool OnAuthenticateUserName(const SDICOS::Utils::AuthenticationData &ad)const;
67 
68  /// Override to authenticate user name. Return true if the user name and passcode is accepted. Return false if user name
69  /// and passcode is rejected. An accepted user name and passcode allows the client to connect to the server and
70  /// process received data. A rejected user name and passcode prevents the client from connecting to the server.
71  virtual bool OnAuthenticateUserNameAndPasscode(const SDICOS::Utils::AuthenticationData &ad)const;
72 
73  /// Override to authenticate the client's application name
74  /// Length Restriction: 16 characters maximum for application name
75  virtual bool OnAuthenticateClientApplicationName(const SDICOS::Utils::AuthenticationData &ad)const;
76 
77  /// Override to receive notification when a client has connected with the server (TCP/IP connection)
78  virtual void OnConnectedToClient(const SDICOS::Utils::SessionData &sd);
79  /// Override to receive notification when a client has disconnected from the server (TCP/IP connection)
80  virtual void OnDisconnectedFromClient(const SDICOS::Utils::SessionData &sd);
81 
82  /// Override to receive notification when a client has started a DICOS connection
83  virtual void OnDicosConnectionStarted(const SDICOS::Utils::SessionData &sd);
84  /// Override to receive notification when a client has stopped a DICOS connection
85  virtual void OnDicosConnectionStopped(const SDICOS::Utils::SessionData &sd);
86 };
