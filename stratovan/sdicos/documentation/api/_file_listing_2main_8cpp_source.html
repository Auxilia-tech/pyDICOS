25 // This example scans a folder and all its sub folders for DICOS files, retrieves all the files in a hierarchy,
26 // then prints the retrieved information to screen
27 //
28 #include "SDICOS/DICOS.h" // DICOS include file
29 #include <iostream>
30 
31 int main()
32 {
33  SDICOS::DicosFileListing dFileListing; // Finds DICOS files
34  SDICOS::Array1D<SDICOS::DicosFileListing::SopInstanceFiles> vFiles; // Organized DICOS files from SDICOS::DicosFileListing
35  const SDICOS::DicosFileListing::SopInstanceFiles *pSOP = NULL; // SOP instance reference
36  SDICOS::Folder folder; // Folder object
37  SDICOS::Array1D<std::pair<SDICOS::Filename, SDICOS::ErrorLog> > vErrorLog; // List of DICOS files with errors
38  bool bRes(true);
39 
40  // Set the folder with DICOS files
41  folder.GetCurrentWorkingFolder();
42  folder.Set(folder.GetFullFolder() + "../DataFiles");
43 
44  // Find all DICOS files in the target folder and its sub folders. Any files DICOS files excluded from
45  // the listing due to errors will populate the third parameter.
46  // By default, this will only check files
47  // with no extension or a '*.dcs' extension. To list DICOS files with other extensions, pass a set
48  // of case sensitive extensions (ie. ".dicos") to the fourth parameter.
49  // To list non-DICOS files that share the same format, like DICOM, set the fifth parameter to 'true'.
50  if(!dFileListing.FindDicosFiles(folder, true, vErrorLog))
51  {
52  std::cout<<"\nNo DICOS files found in "<<folder.GetFullFolder()<<std::endl;
53  return 0;
54  }
55 
56  // Print out error log of DICOS files that failed to be included in the listing
57  for(SDICOS::S_UINT32 nError(0) ; nError < vErrorLog.GetSize() ; ++nError)
58  {
59  std::cout<<"Failed to list file \""<<vErrorLog[nError].first.GetName()<<"\"\n";
60  std::cout<<"Error log:\n"<<vErrorLog[nError].second<<std::endl;
61  bRes = false;
62  }
63 
64  // Get a list of all DICOS files organized in a hierarchy of OOI > Scan > Series > SOP Class
65  dFileListing.GetSopInstanceFiles(vFiles);
66 
67  // Print out information about each SOP instance found in the target folder
68  pSOP = vFiles.GetBuffer();
69  for(SDICOS::S_UINT32 nSopInstance(0) ; nSopInstance < vFiles.GetSize() ; ++nSopInstance, ++pSOP)
70  {
71  std::cout<<"\n-------------------";
72  std::cout<<"\nOOI ID: "<<pSOP->m_dsOOIID.Get();
73  std::cout<<"\nScan Instance UID: "<<pSOP->m_dsScanInstanceUID.Get();
74  std::cout<<"\nSeries Instance UID: "<<pSOP->m_dsSeriesInstanceUID.Get();
75  std::cout<<"\nSOP Class UID: "<<pSOP->m_dsSopClassUID.Get();
76  std::cout<<"\nSOP Instance UID: "<<pSOP->m_dsSopInstanceUID.Get();
77  std::cout<<"\nModality: "<<pSOP->m_dsModality.Get();
78  std::cout<<"\nNumber of Files: "<<pSOP->m_vFilenames.GetSize();
79  for(SDICOS::S_UINT32 nSopInstanceFile(0); nSopInstanceFile < pSOP->m_vFilenames.GetSize(); ++nSopInstanceFile)
80  std::cout<<"\n "<<pSOP->m_vFilenames[nSopInstanceFile].GetFullPath().c_str();
81  std::cout<<std::endl;
82  }
83 
84  return (bRes ? 0 : 1);
85 }
