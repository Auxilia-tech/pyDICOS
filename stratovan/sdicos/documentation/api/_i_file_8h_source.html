295  template < typename TYPE > bool IFile::Read(std::vector< TYPE > &vData)
296  {
297  bool bRes(true);
298  S_UINT64 nLen;
299  vData.clear();
300 
301  bRes = bRes && Read(nLen);
302 
303  if(!bRes) return false;
304  if(0 == nLen) return true;
305 
306  vData.resize(unsigned(nLen));
307  for(S_UINT64 n(0) ; bRes && n < nLen ; ++n)
308  bRes = bRes && Read(vData[(size_t)n]);
309 
310  return bRes;
311  }

313  template < typename TYPE > bool IFile::Write(const std::vector< TYPE > &vData, const std::string &strDesc, FORMAT_HANDLE *phLenHandle, FORMAT_HANDLE *pHandle)
314  {
315  bool bRes(true);
316  const S_UINT64 nLen(vData.size());
317 
318  BeginSection(strDesc, pHandle);
319  bRes = Write(nLen, "Vector Length", phLenHandle) && bRes;
320  for(S_UINT64 n(0) ; n < nLen ; ++n)
321  bRes = Write(vData[(size_t)n], "Vector Item", NULL) && bRes;
322  EndSection((NULL != pHandle) ? *pHandle : 0);
323 
324  return bRes;
325  }
