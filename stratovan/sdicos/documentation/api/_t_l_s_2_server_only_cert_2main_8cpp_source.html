32 //
33 // This class derives from the callback interface class. Override callback functions to process DICOS
34 // data received by the DICOS server. The overridden functions should at least correspond to the API preference
35 // selected by the server. The function 'OnReceiveDicosFileError()', which receives errors that occurred
36 // during the transmission, and 'OnReceiveDicosFile(DicosData<AttributeManager> &, const ErrorLog &)', which is called
37 // if DcsServer::PreferTagAPI() is called or if an unknown modality is received, must always be overridden.
38 //
39 // Example:
40 // A call to DcsServer::PreferUserAPI() should have all the user-level API callback functions overridden in the
41 // callback interface class. Only the functions using the expected modalities should be overridden.
42 // Functions to override:
43 // -OnReceiveDicosFile(DicosData<SDICOS::CT> &, const ErrorLog &)
44 // -OnReceiveDicosFile(DicosData<SDICOS::DX> &, const ErrorLog &)
45 // -OnReceiveDicosFile(DicosData<SDICOS::AIT2D> &, const ErrorLog &)
46 // -OnReceiveDicosFile(DicosData<SDICOS::AIT3D> &, const ErrorLog &)
47 // -OnReceiveDicosFile(DicosData<SDICOS::QR> &, const ErrorLog &)
48 // -OnReceiveDicosFile(DicosData<SDICOS::TDR> &, const ErrorLog &)
49 //
50 // Example:
51 // A call to DcsServer::PreferModuleAPI() should have all the module-level API callback functions overridden in the
52 // callback interface class. Only the functions using the expected modalities should be overridden.
53 // Functions to override:
54 // -OnReceiveDicosFile(DicosData<SDICOS::CTModule> &, const ErrorLog &)
55 // -OnReceiveDicosFile(DicosData<SDICOS::DXModule> &, const ErrorLog &)
56 // -OnReceiveDicosFile(DicosData<SDICOS::AIT2DModule> &, const ErrorLog &)
57 // -OnReceiveDicosFile(DicosData<SDICOS::AIT3DModule> &, const ErrorLog &)
58 // -OnReceiveDicosFile(DicosData<SDICOS::QRModule> &, const ErrorLog &)
59 // -OnReceiveDicosFile(DicosData<SDICOS::TDRModule> &, const ErrorLog &)
60 //

136 bool ExampleServer::StartListeningWithSSL()
137 {
138  /*
139  Setup required DICOS Server parameters
140  */
141 
142  /// Set the server port and IP
143  m_DcsServer.SetIP("127.0.0.1");
144  m_DcsServer.SetPort(ExampleServer::GetServerPort());
145 
146  /// Set the application name. The application name is a 16 character maximum name associated with the server
147  /// Should be related to the application using the server
148  m_DcsServer.SetApplicationName(ExampleServer::GetServerApplicationName());
149 
150  /// Set client certificates as optional
151  /// This allows a TLS connection to be established between a server with verified certificates without the client to have certs
152  /// This must be called before DcsServer::SetSslCertificate, otherwise the server will still expect client certificates
153  m_DcsServer.SetTLSClientCertificateRequired(false);
154 
155  SDICOS::ErrorLog errorLog;
156 
157  /// Set the DICOS server's SSL certificate
158  if(false == m_DcsServer.SetSslCertificate(m_certFile, m_certPassword, m_certCommonName, errorLog))
159  {
160  std::cerr << "Failed to set DcsServer certificate " << errorLog << std::endl;
161  return false;
162  }
163  else{
164  std::cerr << "Set DcsServer certificate SSL=" << m_DcsServer.IsUsingSsl() << std::endl;
165  }
166 
167  /// Start the server listening
168  /// Takes *this as an argument to have access to all of the overridden IRecieveCallback functions
169  if(m_DcsServer.StartListening((*this)))
170  {
171  std::cerr << "DcsServer Listening at " << m_DcsServer.GetIP() << ":" << m_DcsServer.GetPort() << " UsingSsl=" << m_DcsServer.IsUsingSsl() << std::endl;
172  }
173  else
174  {
175  std::cerr << "Failed to start DcsServer at " << m_DcsServer.GetIP() << ":" << m_DcsServer.GetPort() << " UsingSsl=" << m_DcsServer.IsUsingSsl() << std::endl;
176  return false;
177  }
178  return true;
179 }

214  Setup required DICOS Client parameters
215  */
216  m_DcsClient.SetSourceApplication(SDICOS::DcsApplicationEntity("CLIENT"));
217  /// Destination application must be the same application name as the server's
218  m_DcsClient.SetDestinationApplication(ExampleServer::GetServerApplicationName());
219 
220  /// Set the server port and IP for the client to connect to

223  std::cerr << "Failed to set DcsClient's Server IP and Port" << std::endl;
224  return false;

228  std::cerr << "Set DcsClient's Server IP and Port" << std::endl;
