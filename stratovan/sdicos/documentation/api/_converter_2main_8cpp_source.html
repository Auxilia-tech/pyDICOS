126 /*
127 
128 Converter function uses the converter callback class to convert an older DICOS file to a V03 DICOS file
129 
130 */
131 bool ConvertToV03()
132 {
133  SDICOS::DicosLoader dcsLoader; //Variable for loading DICOS files with a provided callback class
134  Converter converter; //Converter callback class
135  SDICOS::Filename filename("../DataFiles/V02_Data/CT/CTUserAPI.dcs"); //Set the older DICOS file's name
136 
137  dcsLoader.SetCallback(&converter); //Set the callback class
138 
139  dcsLoader.PreferUserLevelAPI(); //Tell the loader to use the User-Level API callback functions (This is the default setting)
140 
141  return dcsLoader.Load(filename); //Load the file. The loader calls the overridden functions in the Converter class
142 }
