343 // Init attributes for use in examples
344 bool InitTDR(SDICOS::TDRModule &tdr)
345 {
346  bool bRes(true);
347 
348  // Utility method for setting the current date
349  SDICOS::DcsDate date;
350  date.SetToday();
351 
352  // Utility method for setting the current time
353  SDICOS::DcsTime time;
354  time.SetNow();
355 
356  // The DateTime object is a specific DICOS Value Representation (VR) that contains date and time
357  SDICOS::DcsDateTime dateTime;
358  dateTime.Set(date,time);
359  //
360  // unique identifier for the CT instance referenced by threat items
361  //
362  SDICOS::DcsUniqueIdentifier ctInstanceUID("1.2.276.0.7230010.3.1.4.4126100761.8984.1407370732.881");
363  //
364  // The Threat Sequence class represents the Threat Sequence Module (Table 62)
365  //
366  SDICOS::TDRTypes::ThreatSequence &ts = tdr.GetThreatSequence();
367 
368  //
369  // The ThreatSequence contains information about Potential Threat Objects (PTOs).
370  // The ThreatSequence::ThreatItem class stores the information for a specific PTO.
371  // The ThreatSequence contains one or more ThreatItems.
372  //
373  SDICOS::Array1D<SDICOS::TDRTypes::ThreatSequence::ThreatItem>& threatObject = ts.GetPotentialThreatObject();
374 
375  // for this example there is 1 threats
376  threatObject.SetSize(1);
377 
378  // Indicate that Item 0 is a piece of baggage
379  threatObject[0].EnableBaggageTDR(true);
380  bRes = bRes && threatObject[0].SetID(0);
381 
382  //
383  // The AssessmentSequence class contains one or more threat assessments for this threat object
384  // The Assessment Sequence is implemented as an array of AssessmentItem objects
385  // In this example, two threat assessments are provided for a explosive precursor
386  //
387  SDICOS::TDRTypes::AssessmentSequence& objectAssessmentSequence = threatObject[0].GetAssessment();
388  bRes = bRes && objectAssessmentSequence.GetAssessment().SetSize(2);
389  bRes = bRes && objectAssessmentSequence.GetAssessment()[0].SetThreatCategory(SDICOS::TDRTypes::AssessmentSequence::AssessmentItem::enumExplosive);
390  bRes = bRes && objectAssessmentSequence.GetAssessment()[0].SetAbilityAssessment(SDICOS::TDRTypes::AssessmentSequence::AssessmentItem::enumNoInterference);
391  bRes = bRes && objectAssessmentSequence.GetAssessment()[0].SetAssessmentFlag(SDICOS::TDRTypes::AssessmentSequence::AssessmentItem::enumThreat);
392  bRes = bRes && objectAssessmentSequence.GetAssessment()[1].SetThreatCategory(SDICOS::TDRTypes::AssessmentSequence::AssessmentItem::enumOther);
393  bRes = bRes && objectAssessmentSequence.GetAssessment()[1].SetAbilityAssessment(SDICOS::TDRTypes::AssessmentSequence::AssessmentItem::enumNoInterference);
394  bRes = bRes && objectAssessmentSequence.GetAssessment()[1].SetAssessmentFlag(SDICOS::TDRTypes::AssessmentSequence::AssessmentItem::enumThreat);
395 
396  // Set baggage specific attributes
397  {
398  // The BaggageSpecificTDR object stores information specific to baggage threats
399  SDICOS::TDRTypes::BaggageSpecificTDR& baggageTDR = threatObject[0].GetBaggageTDR();
400  // One representation for the PTO (Single-Energy CT).
401  // Each representation can signify information for another view or energy level.
402  baggageTDR.GetPTORepresentationSequenceItems().SetSize(1);
403  SDICOS::TDRTypes::BaggageSpecificTDR::PTORepresentationSequenceItem& seqItem = baggageTDR.GetPTORepresentationSequenceItems()[0];
404 
405  bRes = bRes && seqItem.SetMass(1321.2283935546875f);
406  bRes = bRes && seqItem.SetDensity(1.0f);
407 
408  // Indicate the original scan or scans associated with this TDR.
409  SDICOS::Array1D<SDICOS::SopInstanceReference> & referencedInstanceSequence = baggageTDR.GetPTORepresentationSequenceItems()[0].GetReferencedInstance();
410  bRes = bRes && referencedInstanceSequence.SetSize(1);
411  bRes = bRes && referencedInstanceSequence[0].SetClassUID(SDICOS::SOPClassUID::GetCT());
412  bRes = bRes && referencedInstanceSequence[0].SetInstanceUID(ctInstanceUID);
413 
414  // Indicate the geometry of the threat Region Of Interest
415  SDICOS::TDRTypes::CommonSpecificTDR::ThreatROI& threatRegionOfInterest = baggageTDR.GetPTORepresentationSequenceItems()[0].GetThreatROI();
416  bRes = bRes && threatRegionOfInterest.SetThreatRoiBase(101.0f, 141.0f, 57.0f);
417  bRes = bRes && threatRegionOfInterest.SetThreatRoiExtents(54.0f, 63.0f, 104.0f);
418  bRes = bRes && threatRegionOfInterest.GetThreatRoiBitmap().SetDims(54, 63, 104);
419  }
420 
421  // Set the Processing time for Item 0
422  bRes = bRes && threatObject[0].SetProcessingStartTime(dateTime);
423  bRes = bRes && threatObject[0].SetProcessingEndTime(dateTime);
424  // Total Processing time is in milliseconds
425  bRes = bRes && threatObject[0].SetTotalProcessingTime(0.0f);
426 
427  return bRes;
428 }
