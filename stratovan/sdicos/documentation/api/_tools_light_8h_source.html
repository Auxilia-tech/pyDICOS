506  /// Deletes items at the end of the list when reducing size. Appends items when increasing size.
507  ///
508  void SetSize(const S_UINT64 nNewSize)
509  {
510  // Same size
511  if(nNewSize == m_nNumItems)
512  return;
513 
514  // Delete all items
515  if(0 == nNewSize)
516  {
517  FreeMemory();
518  return;
519  }
520 
521  // Append items
522  if(nNewSize > m_nNumItems)
523  {
524  const S_UINT64 nNumItemsToAdd(nNewSize-m_nNumItems);
525 
526  for(S_UINT64 n(0) ; n < nNumItemsToAdd ; ++n)
527  Append(TYPE());
528

530  }
531 
532  // Delete items
533  const S_UINT64 nNumItemsToRemove(m_nNumItems-nNewSize);
534 
535  for(S_UINT64 n(0) ; n < nNumItemsToRemove ; ++n)
536  {
537  m_pLast = m_pLast->m_pPrev;
538  m_pLast->m_pNext->m_pPrev = S_NULL;
539  DELETE_POINTER(m_pLast->m_pNext);
