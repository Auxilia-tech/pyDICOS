108  virtual bool SetSize(S_UINT32 nWidth, S_UINT32 nHeight)
109  {
110  if((m_nWidth == nWidth) && (m_nHeight == nHeight))
111  return true;
112 
113  FreeMemory();
114 
115  m_nWidth = SDICOS::Max(S_UINT32(0), nWidth);
116  m_nHeight = SDICOS::Max(S_UINT32(0), nHeight);
117 
118  try
119  {
120  m_pBuffer = new CType[m_nWidth*m_nHeight];
121  m_ppData = new CType*[m_nHeight];
122  }
123  catch(const std::bad_alloc &)
124  {
125  m_nWidth = 0; // Memory allocation error...
126  m_nHeight = 0;
127  }
128  catch(...)
129  {
130  m_nWidth = 0; // Memory allocation error...
131  m_nHeight = 0;
132  }
133  if((NULL == m_pBuffer) || (NULL == m_ppData))
134  {
135  DELETE_ARRAY(m_pBuffer);
136  DELETE_ARRAY(m_ppData);
137  return false;
138  }
139 
140  for(S_UINT32 j = 0; j < m_nHeight; ++j)
141  m_ppData[j] = &(m_pBuffer[j*m_nWidth]);
142 
143  m_bDeleteBuffer = true;
144 
145  return true;
146  }
