25 // This example shows how to utilize the User-Level API in order to send a
26 // CT object over the network.
27 //
28 // The accompanying HostDaemon executable can be used to create a DICOS Server that
29 // listens for and accepts connections from DICOS client.
30 //
31 // To run the example first start the DICOS Server with an ip address and port and then
32 // run SimpleClient with the same ip address and port. If the transfer is successful, log
33 // messages should appear somewhere in /var/log depending upon how the system logging is setup.
34 //
35 // The log messages on a successful transfer should look like:
36 // HostDaemonExample[1705]: received dicos data via User-API CT. Id 1357197849.30379.8598.66202.21690000.Errors
37 //

59 int main(int argc, char * argv[])
60 {
61  for(int i = 0; i < argc; ++i)
62  {
63  if(0 == strcmp("-help", argv[i]) ||
64  (0 == strcmp("--help", argv[i])) ||
65  (0 == strcmp("/H", argv[i]))
66  )
67  {
68  PrintUsage();
69  exit(EXIT_SUCCESS);
70  }
71  }
72 
73  /* Parse command line args for port and IP address */
74  const char * ip(NULL);
75  SDICOS::S_INT32 port(0);
76  for(int i = 0; i < argc; ++i)
77  {
78  if(0 == strcmp("-ip", argv[i]))
79  {
80  if(argc > (i+1))
81  {
82  ip = argv[i+1];
83  }
84  }
85  else if (0 == strcmp("-port", argv[i]))
86  {
87  if(argc > (i+1))
88  {
89  port = atoi(argv[i+1]);
90  }
91  }
92  }
93 
94  int numErrors = 0;
95  if(NULL == ip)
96  {
97  std::cerr << "IP Address for listening must be specified." << std::endl;
98  ++numErrors;
99  }
100 
101  if(port <= 0) {
102  std::cerr << "Port for listening cannot be <= 0" << std::endl;
103  ++numErrors;
104  }
105 
106  if(numErrors)
107  {
108  PrintUsage();
109  return EXIT_FAILURE;
110  }
111 
112  // This example uses the User-Level API CT object to send data to the DICOS Server.
113  SDICOS::CT ct;
114  SDICOS::ErrorLog errorlog;
115 
116  // Initialize the CT object
117  InitCT(ct);
118 
119  // Send the CT object over the network to a server
120  // The SendOverNetwork method in the SDICOS::CT class and other User-Level API classes
121  // takes care of establishing the connection, sending the DICOS data, and closing the connection
122  if(!ct.SendOverNetwork( port, // Port
123  ip, // IP Address
124  "SrcApp", // Source Application Name. Name of application using calling this function.
125  "ExampleHost", // Destination Application Name. Name of application accepting the client's connection.
126  errorlog)) // Error log
127  {
128  //Print error log if the transmission fails
129  std::cout<<"Failed to send data across network:"<<std::endl;
130  std::cout<<errorlog<<std::endl;
131  return EXIT_FAILURE;
132  }
133  std::cout << "Send CT file succeeded." << std::endl;
134 
135  //
136  // Alternatively the Client implementation can create a DcsClient object
137  // and keep the connection open for sending multiple DICOS files
138  // This model can used for a persistent client application that generates DICOS files.
139  //
140  SDICOS::Network::DcsClient client;
141 
142  bool bRes(true); // client methods return true/false to indicate success and failure
143 
144  bRes = client.SetServerPortandIP(port,ip) && bRes;
145  bRes = client.SetSourceApplication("SrcApp") && bRes;
146  bRes = client.SetDestinationApplication("ExampleHost") && bRes;
147 
148  // For a client machine with multiple IP address, e.g. > 1 network card, this method is used to indicate which IP address to use for sending
149  //bRes = client.SetClientIP() && bRes;
150 
151  if(!bRes)
152  {
153  // print error message
154  return EXIT_FAILURE;
155  }
156 
157  //Connect to host
158  if(!client.ConnectToServer())
159  {
160  // handler error
161  return EXIT_FAILURE;
162  }
163 
164  // Send multiple CT files before closing connection
165  ct.SetSopInstanceUID(SDICOS::DcsUniqueIdentifier::CreateGUID());
166  bRes = ct.SendOverNetwork(client, errorlog) && bRes;
167 
168  if(!bRes)
169  {
170  // Handle failure
171  return EXIT_FAILURE;
172  }
173 
174  ct.SetSopInstanceUID(SDICOS::DcsUniqueIdentifier::CreateGUID());
175  bRes = ct.SendOverNetwork(client,errorlog) && bRes;
176 
177  if(!bRes) {
178  // Handle failure
179  return EXIT_FAILURE;
180  }
181 
182  client.DisconnectFromServer();
183 
184  return EXIT_SUCCESS;
185 }

188 void InitCT(SDICOS::CT &ct)
189 {
190  const SDICOS::DcsDate strDate(SDICOS::DcsDate::Today());
191  const SDICOS::DcsTime strTime(SDICOS::DcsTime::Now());
192  SDICOS::Vector3D<float> vecRowOrientation;
193  SDICOS::Vector3D<float> vecColumnOrientation;
194 
195  ct.FreeMemory();
196 
197  vecRowOrientation.Set(1,0,0);
198  vecColumnOrientation.Set(0,1,0);
199 
200  ct.SetFrameOfReferenceUID("4.4.6.6.0.0.8.8.7");
201  ct.SetImageAcquisitionDateAndTime(strDate, strTime);
202  ct.SetBurnedInAnnotation(false);
203  ct.SetPhotometricInterpretation(SDICOS::ImagePixelMacro::enumMonochrome2);
204  ct.SetImageType(SDICOS::CTTypes::CTImage::enumPhotoelectric);
205  ct.SetImageAcquisitionDuration(5000);
206  ct.SetContentDateAndTime(strDate, strTime);
207  ct.SetOOIID("10");
208  ct.SetOOIIDAssigningAuthority("OOI ID Assigning Authority");
209  ct.SetOOIIDType(SDICOS::ObjectOfInspectionModule::IdInfo::enumRFID);
210  ct.SetOOIType(SDICOS::ObjectOfInspectionModule::enumTypeBaggage);
211  ct.SetScanID("Scan ID");
212  ct.GenerateScanInstanceUID();
213  ct.SetScanStartDateAndTime(strDate, strTime);
214  ct.SetScanType(SDICOS::GeneralScanModule::enumOperational);
215  ct.GenerateSeriesInstanceUID();
216  ct.SetSeriesDateAndTime(strDate, strTime);
217  ct.SetSeriesAcquisitionStatus(SDICOS::GeneralSeriesModule::enumSuccessful);
218  ct.SetDeviceCalibrationDateAndTime(strDate, strTime);
219  ct.SetDeviceSerialNumber("123456987");
220  ct.SetMachineAddress("Machine Address");
221  ct.SetMachineLocation("Machine Location");
222  ct.SetMachineID("Machine ID");
223  ct.SetDeviceManufacturer("Device Manufacturer");
224  ct.SetDeviceManufacturerModelName("Device Manufacturer Model Name");
225  ct.SetDeviceSoftwareVersion("Device Software Version");
226  ct.SetSopInstanceCreationDateAndTime(strDate, strTime);
227  ct.GenerateSopInstanceUID();
228 
229  SDICOS::Section *psection = S_NULL;
230 
231  ct.SetNumberOfSections(1);
232  psection = ct.GetSectionByIndex(0);
233 
234  psection->SetFilterMaterial(SDICOS::CTTypes::CTXRayDetails::enumAluminum);
235  psection->SetFocalSpotSizeInMM(10);
236  psection->SetKVP(7000);
237  psection->SetPlaneOrientation(vecRowOrientation, vecColumnOrientation);
238  psection->SetPositionInMM(0,0,2000);
239  psection->SetSpacingInMM(1,1,1);
240  psection->GetPixelData().Allocate(SDICOS::Volume::enumUnsigned16Bit, 32, 32, 32);
241  psection->GetPixelData().GetUnsigned16()->Zero(20);
242 }
