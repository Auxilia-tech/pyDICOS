45 /// Sample code for incrementally adding slices to a volume
46 ///
47 /// \section Array3DLargeAddingSlicesExample Creating a 3D volume by incrementally adding slices
48 ///
49 /// Incrementally add slices to a volume. The slices can be owned by the Array3DLarge object
50 /// or by the user.
51 ///
52 bool Array3DLargeAddingSlicesExample()
53 {
54  bool bRes(true);
55  const SDICOS::S_UINT32 nWidth(197), nHeight(237), nDepth(50);
56 
57  //
58  // Construct the Array3DLarge object and indicate that it does not own the data
59  //
60  const SDICOS::MemoryPolicy::VOLUME_MEMORY_POLICY nMemoryPolicy = SDICOS::MemoryPolicy::DOES_NOT_OWN_SLICES;
61 
62  // A nDepth of 0 indicates that no slices are allocated at construction
63  SDICOS::Array3DLarge<SDICOS::S_INT32> volData(nWidth, nHeight, 0, nMemoryPolicy);
64 
65  const SDICOS::S_INT32 nInitValue(0XFADEC01B);
66  SDICOS::S_UINT32 nSliceCount(0);
67 
68  // Each slice is an SDICOS::Array2D object
69  SDICOS::Array2D<SDICOS::S_INT32> *pNewSlice = new SDICOS::Array2D<SDICOS::S_INT32>(nWidth, nHeight);
70 
71  pNewSlice->Zero(nInitValue); //Initialize each volume element with the same nValue
72 
73  // Add the slice to the volume. AddSlice returns true/false.
74  bRes = volData.AddSlice(pNewSlice) && bRes;
75 
76  // A slice with bad dimensions is rejected.
77  // The action taken when a bad slice is encountered is one of
78  // SDICOS::Array3DLargeBase::EARLY_OUT
79  // SDICOS::Array3DLargeBase::SKIP_AND_CONTINUE (applies only when adding multiple slices)
80  // SDICOS::Array3DLargeBase::ADD_DEBUG_SLICE
81  // The default action is EARLY_OUT
82  SDICOS::Array2D<SDICOS::S_INT32> vSliceDifferentDims(nWidth-1, nHeight-1);
83  vSliceDifferentDims.Zero(nInitValue);
84 
85  bool bAddResult = volData.AddSlice(&vSliceDifferentDims); // this call fails
86  bRes = (!bAddResult) && bRes;
87  bRes = (1 == volData.GetDepth()) && bRes;
88 
89  //
90  // Add a debug slice filled with a specific nValue.
91  // The debug slice is always owned by the Array3DLarge object
92  //
93  const SDICOS::Array3DLargeBase::ADD_SLICE_FAILURE_POLICY nAddSliceFailurePolicy = SDICOS::Array3DLargeBase::ADD_DEBUG_SLICE;
94  const SDICOS::S_INT32 nDebugFillValue(0xABCDEF);
95 
96  bAddResult = volData.AddSlice(&vSliceDifferentDims, nAddSliceFailurePolicy, &nDebugFillValue);
97 
98  // Failure to properly add the slice still returns false even if the debug slice is added
99  bRes = (!bAddResult) && bRes;
100  bRes = (2 == volData.GetDepth()) && bRes;
101 
102  //Check the first value of both slices
103  //volData[Slice][Row][Column]
104  bRes = volData[0][0][0] == nInitValue && bRes;
105  bRes = volData[1][0][0] == nDebugFillValue && bRes;
106 
107  if(!bRes)
108  {
109  std::cout << "Array3DLargeAddingSlicesExample failed to add slice with wrong dimensions" << std::endl;
110  std::cout << "added debug slice using failure policy ADD_DEBUG_SLICE." << std::endl;
111  }
112 
113  volData.FreeMemory(); //Delete references to slices
114  SDICOS::DELETE_POINTER(pNewSlice); //Delete slice
115 
116  //Reinitialize the volume to accept slices of nWidth x nHeight
117  volData.SetSize(nWidth, nHeight, 0);
118 
119  //Add the slice with incorrect dimensions, which results in adding a debug slice again
120  bAddResult = volData.AddSlice(&vSliceDifferentDims, nAddSliceFailurePolicy, &nDebugFillValue);
121 
122  //
123  // Allocated multiple slices to add to the volume
124  //
125  std::list< SDICOS::Array2D<SDICOS::S_INT32>* > listSlices;
126  for(SDICOS::S_UINT32 i(0); i < nDepth; ++i)
127  {
128  pNewSlice = new SDICOS::Array2D<SDICOS::S_INT32>(nWidth, nHeight);
129  pNewSlice->Zero(nInitValue + i);
130  listSlices.push_back(pNewSlice);
131  }
132 
133  const SDICOS::S_UINT32 nNewSliceCount = SDICOS::S_UINT32(volData.GetDepth() + listSlices.size());
134 
135  // Add a bad slice that will be ignored because the error policy is SKIP_AND_CONTINUE.
136  // All good slices will be added.
137  pNewSlice = new SDICOS::Array2D<SDICOS::S_INT32>(12,13);
138  listSlices.insert((++listSlices.begin()), pNewSlice);
139 
140  // Add slices to the volume using iterator bounds and ignore bad slices.
141  // Note that if bad slices are encountered AddSlices() will return false regardless of what
142  // the bad slice policy is set to
143  const bool bResAddSlicesWithBadSlice = volData.AddSlices(listSlices.begin(), listSlices.end(), SDICOS::Array3DLargeBase::SKIP_AND_CONTINUE);
144  bRes = !bResAddSlicesWithBadSlice && bRes;
145 
146  // Check that all slices have been added
147  bRes = (volData.GetDepth() == nNewSliceCount) && bRes;
148 
149  if(bRes)
150  std::cout << "Array3DLargeAddingSlicesExample successfully added " << (listSlices.size()-1) << " listSlices from a list using iteration." << std::endl;
151 
152  // Use an iterator to traverse the slices and test that slice data is correct.
153  SDICOS::S_UINT32 nDebugSliceCount(0);
154  SDICOS::S_INT32 nValidSliceCount(0);
155  SDICOS::Array3DLarge<SDICOS::S_INT32>::Iterator it = volData.Begin();
156 
157  for(nSliceCount = 0; it != volData.End(); ++it, ++nSliceCount)
158  {
159  const SDICOS::S_INT32 nValue = (**it)[0][0]; //Access first value of slice
160 
161  if(nValue == nDebugFillValue)
162  ++nDebugSliceCount;
163  else if(nValue == (nValidSliceCount + nInitValue))
164  ++nValidSliceCount;
165  else
166  {
167  bRes = false;
168  std::cout << "Array3DLargeAddingSlicesExample encountered bad nValue for slice " << nSliceCount << std::endl;
169  }
170  }
171 
172  bRes = (nValidSliceCount == SDICOS::S_INT32(nNewSliceCount - nDebugSliceCount));
173  bRes = (1 == nDebugSliceCount) && bRes;
174  bRes = (nSliceCount == nNewSliceCount) && bRes;
175 
176  std::list< SDICOS::Array2D<SDICOS::S_INT32>* >::iterator itSlice;
177 
178  //Delete slices
179  for (itSlice = listSlices.begin(); itSlice != listSlices.end(); ++itSlice)
180  SDICOS::DELETE_POINTER(*itSlice);
181 
182  listSlices.clear();
183 
184  return bRes;
185 } // EndArray3DLargeAddingSlices
