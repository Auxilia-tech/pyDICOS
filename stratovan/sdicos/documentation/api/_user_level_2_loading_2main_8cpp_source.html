32 // This class derives from the callback interface and is used by the

52 int main()
53 {
54  SDICOS::DicosLoader dcsloader; // DICOS loading class
55  DicosLoadingCallback dcscallback; // Callback interface for passing loaded files
56  SDICOS::Folder folder; // Folder object
57  int retvar=0;
58 
59  // Set the folder with DICOS files
60  folder.GetCurrentWorkingFolder();
61  folder.Set(folder.GetFullFolder() + "../DataFiles/V03_Data/");
62 
63  // Set the callback function to use after each file is loaded.
64  // By default, this callback interface will only check files
65  // with no extension or a '*.dcs' extension. To load DICOS files with other extensions, call
66  // DicosLoadingCallback::IncludeFileExtension with a case sensitive extension (ie. ".dicos").
67  // To load non-DICOS files that share the same format, like DICOM, add the override
68  // function "virtual bool OnLoadOnlyDICOS()const" to the callback interface and set
69  // the function to return 'false'.
70  dcsloader.SetCallback(&dcscallback);
71 
72  // Set the preferred callback API. User-Level API is set by default.
73  dcsloader.PreferUserLevelAPI();
74 
75  std::cout << "Loading files from folder " << folder.GetFullFolder().c_str() << std::endl;
76 
77  // Find all the DICOS files in the target folder and all sub folders
78  if(dcsloader.Load(folder, true))
79  {
80  std::cout<<"\nAll files loaded"<<std::endl;
81  } else {
82  std::cout<<"\nFailed to load all files"<<std::endl;
83  retvar=1;
84  }
85 
86  return retvar;
87 }

122 void DicosLoadingCallback::OnReceiveDicosFile(SDICOS::Utils::DicosData<SDICOS::CT> &ct, const SDICOS::ErrorLog &errorlog)
123 {
124  // This function is called after a CT DICOS file is loaded and the preferred API is set to User-Level API

126

134  ct.TakeOwnership(pct);
135  DELETE_POINTER(pct);
136 }
