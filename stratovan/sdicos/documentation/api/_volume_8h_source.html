395  /// Add slices using iterators similar to stl iterators.
396  ///
397  /// The dereferenced iterator must be pf type Array2D<TYPE>*, where TYPE is
398  /// the data of this object if it is already allocated.
399  ///
400  /// Allocates volume to slice type if volume is not allocated.
401  template<typename ITERATOR>
402  bool AddSlices(ITERATOR itStart, const ITERATOR &itEnd, const Array3DLargeBase::ADD_SLICE_FAILURE_POLICY nFailurePolicy = Array3DLargeBase::EARLY_OUT)
403  {
404  bool bRes(true);
405 
406  if (Array3DLargeBase::EARLY_OUT == nFailurePolicy)
407  {
408  for (; bRes && itStart != itEnd; ++itStart)
409  bRes = AddSlice(*itStart, nFailurePolicy);
410  }
411  else
412  {
413  for (; itStart != itEnd; ++itStart)
414  bRes = AddSlice(*itStart, nFailurePolicy) && bRes;
415  }
416 
417  return bRes;
418  }
