28 bool CreateAIT2DForProcessingSimple()
29 {
30  //
31  // Instantiate and initialize the AIT2D object.
32  // Attributes not set in the constructor call are set to valid defaults.
33  // See API documentation in SdcsTemplateAIT2D.h for default values.
34  // Object of Inspection Type (OOI Type) : Person
35  // AIT Device Type : Passive Millimeter Wave
36  // Presentation Intent Type : For Processing. Indicates data will undergo additional processing.
37  // Photometric Interpretation of Pixel Data : Monochrome2 (low pixel values map to low intensity)
38  //
39  SDICOS::AIT2D ait(
40  SDICOS::ObjectOfInspectionModule::enumTypePerson,
41  SDICOS::AIT2DTypes::AIT2DImage::enumPassiveMillimeterWave,
42  SDICOS::AIT2DTypes::AIT2DSeries::enumProcessing,
43  SDICOS::ImagePixelMacro::enumMonochrome2
44  );
45 
46  // OOI Type Person required gender
47  ait.SetGender(SDICOS::ObjectOfInspectionModule::enumGenderFemale);
48 
49  // AIT2D only supports unsigned 8-64 bit and float image data
50  SDICOS::S_UINT32 width(256), height(128);
51  SDICOS::Image2D& aitData = ait.GetData();
52 
53  // Allocate a contiguous block of memory for the AIT data.
54  // Data is stored as unsigned shorts (16-bit).
55  if(!aitData.Allocate(SDICOS::Image2D::enumUnsigned16Bit, width,height))
56  {
57  std::cerr << "Failed to allocate AIT data buffer" << std::endl;
58  return false;
59  }
60 
61  // Get reference to allocated unsigned data buffer.
62  SDICOS::Array2D<SDICOS::S_UINT16> &rawData = *aitData.GetUnsigned16();
63 
64  // Example for setting AIT data by accessing each element individually
65  SDICOS::S_UINT16 count(0);
66  for(SDICOS::S_UINT32 row(0); row < aitData.GetHeight(); ++row)
67  {
68  for(SDICOS::S_UINT32 col(0); col < aitData.GetWidth(); ++col)
69  {
70  rawData[row][col] = count++;
71  }
72  }
73 
74  // Buffer that holds user generated data.
75  // For example purposes only
76  SDICOS::S_UINT16 * userGeneratedData = new unsigned short[width*height];
77 
78  //Initialize with dummy data
79  memset(userGeneratedData, 0, width*height * sizeof(SDICOS::S_UINT16));
80 
81  // Set AIT data by getting a pointer to the raw data buffer
82  // allocated by the call to AllocateUnsigned.
83  // Data can be copied into this buffer or the buffer can be used directly
84  SDICOS::S_UINT16 * rawBuffer = rawData.GetBuffer();
85 
86  // Copy user generated into allocated buffer for image data.
87  memcpy(rawBuffer,userGeneratedData,width*height*sizeof(SDICOS::S_UINT16));
88 
89  // Delete buffer
90  SDICOS::DELETE_ARRAY(userGeneratedData);
91 
92  // Example for how to set the orientation of the person OOI with respect to AIT2D's Coordinate System.
93  // This attribute is not required.
94  SDICOS::Vector3D<float> personXAxis(-1,0,0), personYAxis(0,-1,0);
95  ait.SetPersonOrientation(personXAxis,personYAxis);
96 
97  SDICOS::ErrorLog errorlog;
98  SDICOS::Folder folder("AIT2DFiles");
99  SDICOS::Filename filename(folder,"SimpleProcessingAIT2D.dcs");
100 
101  if(!ait.Write(filename,errorlog,SDICOS::DicosFile::enumLittleEndianExplicit))
102  {
103  std::cerr << "Simple AIT2D Template Example unable to write DICOS File " << filename << std::endl;
104  std::cerr << errorlog << std::endl;
105  return false;
106  }
107  else
108  {
109  std::cout << "Wrote file to " << filename << std::endl;
110  SDICOS::AIT2D aitRead;
111 
112  SDICOS::ErrorLog errorlog;
113  if(aitRead.Read(filename,errorlog)) {
114  if(aitRead == ait) {
115  std::cout << "Successfully read and compared AIT2D files" << std::endl;
116  return true;
117  } else {
118  std::cout << "AIT2D file loaded from " << filename << " does not match original." << std::endl;
119  return false;
120  }
121  }
122  else {
123  std::cerr << "Unable to read AIT2D file " << filename << std::endl;
124  std::cerr << errorlog << std::endl;
125  return false;
126  }
127  }
128  return true;
129 } // EndCreateAIT2DForProcessingSimple

133  /// Create a AIT2D file using a color palette that is intended for additional processing using the user-level API Object SDICOS::DX.
134  ///
135 bool CreateAIT2DWithColorPaletteForProcessingSimple()
136 {
137  //
138  // Instantiate and initialize the AIT2D object.
139  // Attributes not set in the constructor call are set to valid defaults.
140  // See API documentation in SdcsTemplateAIT2D.h for default values.
141  // Object of Inspection Type (OOI Type) : Person
142  // AIT Device Type : Passive Millimeter Wave
143  // Presentation Intent Type : For Processing. Indicates data will undergo additional processing.
144  // Photometric Interpretation of Pixel Data : enumPaletteColor (Indicates that the a color palette is used.)
145  SDICOS::AIT2D ait(
146  SDICOS::ObjectOfInspectionModule::enumTypePerson,
147  SDICOS::AIT2DTypes::AIT2DImage::enumPassiveMillimeterWave,
148  SDICOS::AIT2DTypes::AIT2DSeries::enumProcessing,
149  SDICOS::ImagePixelMacro::enumPaletteColor);
150 
151  // Identity: Low to high values represent dark to light
152  // Inverse: Low to high values represent light to dark
153  if (!ait.SetPresentationLUTShape(SDICOS::GeneralImageModule::enumIdentity))
154  {
155  std::cout << "Invalid Presentation LUT Shape" << std::endl;
156  return false;
157  }
158 
159  //Dimensions of image
160  const SDICOS::S_UINT32 nWidth(256), nHeight(128);
161 
162  //Set up Red, Green, and Blue color palette
163  //These values are indexed into by the Pixel Data values
164  SDICOS::Array1D<SDICOS::S_UINT8> &vRed = ait.GetRedLUT();
165  SDICOS::Array1D<SDICOS::S_UINT8> &vGreen = ait.GetGreenLUT();
166  SDICOS::Array1D<SDICOS::S_UINT8> &vBlue = ait.GetBlueLUT();
167 
168  //Set all to same size. This example will use a single color for each row.
169  vRed.SetSize(nHeight);
170  vGreen.SetSize(nHeight);
171  vBlue.SetSize(nHeight);
172 
173  SDICOS::S_UINT8 *pnRed = vRed.GetBuffer();
174  SDICOS::S_UINT8 *pnGreen = vGreen.GetBuffer();
175  SDICOS::S_UINT8 *pnBlue = vBlue.GetBuffer();
176  SDICOS::S_UINT8 nColor(0);
177  for (SDICOS::S_UINT16 nIndex(0); nIndex < nHeight; ++nIndex, ++pnRed, ++pnGreen, ++pnBlue)
178  {
179  //Setting RGB to (nColor, nColor/2, nColor)
180  *pnRed = nColor;
181  *pnGreen = nColor / 2;
182  *pnBlue = nColor;
183 
184  ++nColor;
185  }
186 
187  SDICOS::Image2D& imgPixelData = ait.GetData();
188  imgPixelData.Allocate(SDICOS::Image2D::enumUnsigned16Bit, nWidth, nHeight);
189 
190  // Pixel Data is set of indices into the Red, Green, and Blue LUT's
191  SDICOS::Array2D<SDICOS::S_UINT16> &vIndexData = *imgPixelData.GetUnsigned16();
192 
193  for (SDICOS::S_UINT32 row(0); row < imgPixelData.GetHeight(); ++row)
194  {
195  for (SDICOS::S_UINT32 col(0); col < imgPixelData.GetWidth(); ++col)
196  {
197  vIndexData[row][col] = row; //Sets each column in a row to the same color
198  }
199  }
200 
201  SDICOS::ErrorLog errorlog;
202  SDICOS::Folder aitFolder("AIT2DFiles");
203  SDICOS::Filename aitFilename(aitFolder, "SimpleColorPaletteAIT2D.dcs");
204 
205  if (!ait.Write(aitFilename, errorlog, SDICOS::DicosFile::enumLittleEndianExplicit))
206  {
207  std::cerr << "Color Palette AIT2D Example unable to write DICOS File " << aitFilename << std::endl;
208  std::cerr << errorlog << std::endl;
209  return false;
210  }
211  else
212  {
213  std::cout << "Wrote file to " << aitFilename << std::endl;
214  SDICOS::AIT2D aitRead;
215 
216  SDICOS::ErrorLog errorlog;
217  if (aitRead.Read(aitFilename, errorlog))
218  {
219  if (aitRead == ait)
220  {
221  std::cout << "Successfully read and compared AIT2D files" << std::endl;
222  return true;
223  }
224  else
225  {
226  std::cout << "AIT2D file loaded from " << aitFilename << " does not match original." << std::endl;
227  return false;
228  }
229  }
230  else
231  {
232  std::cerr << "Unable to read DX file " << aitFilename << std::endl;
233  std::cerr << errorlog << std::endl;
234  return false;
235  }
236  }
237 } /// CreateAIT2DWithColorPaletteForProcessingSimple
