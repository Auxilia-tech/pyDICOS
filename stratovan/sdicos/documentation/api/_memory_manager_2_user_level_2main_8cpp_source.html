26 // This example shows how to use IMemoryManager to provide memory allocation and
27 // deallocation for the DICOS library's when loading with User-Level API.
28 //
29 #include "SDICOS/DICOS.h" //Header for DICOS
30 #include <iostream>
31 #include <map>

33 //
34 // This class derives from the IMemoryManager interface class and will handle allocating and
35 // deallocating memory for the DICOS library.
36 // This class preallocates a fixed number of buffers which are passed to the DICOS library for
37 // its pixel data.
38 // When the DICOS library needs to delete the allocated data, this class's deallocation function
39 // is called with the buffer passed as a parameter allowing it to be recycled without actually deleting it.
40 //
41 class CustomMemoryManager : public SDICOS::IMemoryManager
42 {
43 public:
44  SDICOS::Array1D< std::pair<bool, SDICOS::MemoryBuffer> > m_vBuffers; //Array of preallocated buffers
45  std::map<const unsigned char*, SDICOS::S_UINT32> m_mapUsedBuffers; //Simple tracking for which slices are used
46  const SDICOS::S_UINT64 m_nBufferSizeInBytes; //Size of each buffer
47 
48  //Constructor preallocates the buffers
49  CustomMemoryManager(const SDICOS::S_UINT64 m_nBufferSizeInBytes = (512 * 512 * 8), const SDICOS::S_UINT32 nNumBuffersToAllocated = 500);
50 
51  //Override from IMemoryManager to use preallocated buffer. This function will provide a buffer
52  //to the DICOS library and mark that buffer as 'used'.
53  virtual bool OnAllocate(SDICOS::MemoryBuffer &mbAllocate, const SDICOS::S_UINT64 nSizeInBytesToAllocate);
54 
55  //Override from IMemoryManager to deallocated buffer. This function removes the buffer from the 'used' list
56  //and allows it to be used again in OnAllocate()
57  virtual bool OnDeallocate(SDICOS::MemoryBuffer &mbDeallocate);
58 
59  //Override from IMemoryManager to set the memory policy for the buffer passed to the DICOS library.
60  //Since this class owns the data and the DICOS library is borrowing it, this function will be set
61  //to return MemoryBuffer::enumPolicy_DoesNotOwnData
62  virtual SDICOS::MemoryBuffer::MEMORY_POLICY OnGetSliceMemoryPolicy()const;
63 };

145 CustomMemoryManager::CustomMemoryManager(const SDICOS::S_UINT64 m_nBufferSizeInBytes, const SDICOS::S_UINT32 nNumBuffersToAllocated)
146  : m_nBufferSizeInBytes(m_nBufferSizeInBytes)
147 {
148  m_vBuffers.SetSize(nNumBuffersToAllocated); //Allocate a set number of buffers
149 
150  std::pair<bool, SDICOS::MemoryBuffer> *pBuffer(m_vBuffers.GetBuffer());
151 
152  //Initialize each buffer to the indicated size and mark them as available
153  for (SDICOS::S_UINT32 n(0); n < nNumBuffersToAllocated; ++n, ++pBuffer)
154  {
155  pBuffer->first = true;
156  pBuffer->second.GrowTo(m_nBufferSizeInBytes);
157  }
158 }

160 bool CustomMemoryManager::OnAllocate(SDICOS::MemoryBuffer &mbAllocate, const SDICOS::S_UINT64 nSizeInBytesToAllocate)
161 {
162  //Check if the requested buffer size is too large. This will tell the DICOS library to
163  //allocate a buffer itself.
164  if (m_nBufferSizeInBytes <= nSizeInBytesToAllocate)
165  {
166  return false;
167  }
168 
169  //Find an available buffer
170  std::pair<bool, SDICOS::MemoryBuffer> *pBuffer(m_vBuffers.GetBuffer());
171  for (SDICOS::S_UINT32 n(0); n < m_vBuffers.GetSize(); ++n, ++pBuffer)
172  {
173  if (pBuffer->first)
174  {
175  std::cout << "Allocated slice " << n << std::endl; //Print out the index of the buffer
176 
177  //Even though the buffer allocated may be larger, only state the requested size
178  //has been provided.
179  pBuffer->first = false; //Indicate buffer is no longer available
180  mbAllocate.SetBuffer(pBuffer->second.GetData(), nSizeInBytesToAllocate);
181  m_mapUsedBuffers[pBuffer->second.GetData()] = n; //Track the used memory buffer
182 
183  //Return true to tell the DICOS library this class provided a buffer of the requested size
184  return true;
185  }
186  }
187 
188  //No more buffers available, return false so the DICOS library handls allocation
189  return false;
190 }

192 bool CustomMemoryManager::OnDeallocate(SDICOS::MemoryBuffer &mbDeallocate)
193 {
194  //If memory policy does match, then the buffer was not allocated by this class
195  if (GetSliceMemoryPolicy() != mbDeallocate.GetMemoryPolicy())
196  return false; //Returning false allows the DICOS library to delete the buffer based on its memory policy
197 
198  std::map<const unsigned char*, SDICOS::S_UINT32>::iterator it;
199 
200  //Check if the buffer was provided by this class
201  it = m_mapUsedBuffers.find(mbDeallocate.GetData());
202  if (m_mapUsedBuffers.end() == it)
203  return false; //Returning false allows the DICOS library to delete the buffer based on its memory policy
204 
205  std::cout << "Deallocated slice " << it->second << std::endl; //Print out the index of the buffer
206 
207  m_vBuffers[it->second].first = true; //Mark the buffer as available for use in OnAllocate()
208  m_mapUsedBuffers.erase(it); //Remove buffer from the 'used' list
209 
210  //Returning true tells the DICOS library this class has deallocated the buffer
211  return true;
212 }

214 SDICOS::MemoryBuffer::MEMORY_POLICY CustomMemoryManager::OnGetSliceMemoryPolicy()const
215 {
216  //Tell the DICOS library it does not own the data. This prevents the DICOS library
217  //from deleting the preallocated buffers.
218  return SDICOS::MemoryBuffer::enumPolicy_DoesNotOwnData;
