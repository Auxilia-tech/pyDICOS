40 {
41  /// Create an AIT3D DICOS file for a person
42  /// The Object of Inspection Type (OOI) is automatically set to person
43  /// The AIT Device Type is Active Millimeter Wave.
44  /// The pixel photometric interpretation is Monochrome2 indicating that lower values map to lower intensities.
45  /// The Image description attributes default to a monochrome volume with original data.
46  ///
47  /// The Object of Inspection Type (OOI) is one of:
48  /// enumTypeBioSample
49  /// enumTypeCargo
50  /// enumTypeBaggage
51  /// enumTypeAnimal
52  /// enumTypeOther
53  /// enumTypePerson
54  ///
55  /// The AIT Device Type is one of:
56  /// enumPassiveMillimeterWave
57  /// enumActiveMillimeterWave
58  /// enumTerahertzImaging
59  ///
60  /// The Photometric Interpretation is one of:
61  /// enumMonochrome1
62  /// enumMonochrome2
63  /// enumPaletteColor
64  SDICOS::AIT3D ait(
65  SDICOS::ObjectOfInspectionModule::enumGenderFemale,
66  SDICOS::AIT3DTypes::AIT3DImage::enumActiveMillimeterWave,
67  SDICOS::ImagePixelMacro::enumMonochrome2
68  );
69 
70 
71  // Set the time that the pixel data was created.
72  ait.SetPixelDataCreationDateAndTime(SDICOS::DcsDate::Today(),SDICOS::DcsTime::Now());
73 
74  // Indicate when the data was originally acquired
75  SDICOS::DcsDate acquisitionDate(1941,12,7);
76  SDICOS::DcsTime acquisitionTime(7,48,0,0);
77  ait.SetAcquisitionDateTime(acquisitionDate,acquisitionTime);
78 
79  // Add one 3D Volume Section.
80  // Image Type and Image description are the same as the AIT3DSeriesBase
81  SDICOS::SectionAIT *pSection = ait.AddSection();
82  pSection->GetPixelData().Allocate(SDICOS::Volume::enumUnsigned16Bit, 128,100,400); // 3D Array of data allocated by 2D slices
83 
84  // Set slice spacing
85  pSection->SetSpacingInMM(10.0f,15.0f,20.0f);
86 
87  // Set Slice Position
88  pSection->SetPositionInMM(100,150,200);
89 
90  // Set plane orientation for each slice
91  // X-Axis goes from person's right hand to left hand
92  // Y-Axis goes from person's posterior to anterior
93  // Z-Axis goes from person's foot to head
94  const SDICOS::Vector3D<float> vecRow(1, 0, 0);
95  const SDICOS::Vector3D<float> vecColumn(0, 1, 0);
96  pSection->SetPlaneOrientation(vecRow, vecColumn);
97 
98  // Get reference to the Volume Data
99  SDICOS::Array3DLarge<SDICOS::S_UINT16>& sectionData = *pSection->GetPixelData().GetUnsigned16();
100 
101  // This buffer is allocated for example purposes only
102  SDICOS::S_UINT16 * userData = new SDICOS::S_UINT16[pSection->GetSliceSize()];
103 
104  //Initialize with dummy data
105  memset(userData, 0, pSection->GetSliceSize() * sizeof(SDICOS::S_UINT16));
106 
107  // Set volume data by setting each XY plane
108  for(SDICOS::S_UINT32 i = 0; i < pSection->GetDepth(); ++i)

110  SDICOS::Array2D<SDICOS::S_UINT16>& xyPlane = sectionData[i];
111  SDICOS::S_UINT16 * pBuffer = xyPlane.GetBuffer();
112

114  memcpy(pBuffer, userData, sizeof(SDICOS::S_UINT16)*pSection->GetSliceSize());
115  }

118  SDICOS::DELETE_ARRAY(userData);

336 /**
337  * Verify that the DICOS files written by CreateAIT3DMultiVolume() can be read and compared.
338  * @param original The original AIT object
339  * @param folder Folder that contains the DICOS files for each section. No other files should be in this folder
340  * @param baseFilename Files for each separate DICOS files are written as baseFilename0000, baseFilename0001 etc
341  * @return true/false
342  */
343 bool VerifyWrittenDICOSFiles(
344  const SDICOS::AIT3D& original,
345  const SDICOS::Folder& folder,
346  const SDICOS::Filename& baseFilename
347 )
348 {
349  bool bRes(true);
350 
351  std::cout << "Write AIT3D DICOS file from template " << baseFilename << " " << original.GetNumberOfFilesToCreate() << std::endl;
352 
353  //
354  // The SDICOS::DicosFileListing class searches and collects all DICOS files
355  // in a file system folder and hierarchy. Any files DICOS files excluded from
356  // the listing due to errors will populate the 'vErrorLog' parameter.
357  //
358  SDICOS::DicosFileListing files;
359  SDICOS::Array1D<std::pair<SDICOS::Filename, SDICOS::ErrorLog> > vErrorLog;
360  files.FindDicosFiles(folder,false,vErrorLog);
361 
362  if( 0 == files.GetNumberOfObjectOfInspections() ||
363  0 == files[0].GetNumberOfScans() ||
364  0 == files[0][0].GetNumberOfSeries())
365  {
366  std::cerr << "Unable to find DICOS files for AIT3D Template" << std::endl;
367  return false;
368  }
369 
370  //
371  // In this example, the SDICOS::DicosFileListing object is used to find the
372  // files generated for each independent AIT3D Section added to the AIT3D Series.
373  // There are two files generated in this example and they exist in the same folder
374  //
375  SDICOS::Array1D<SDICOS::DicosFileListing::FileInfo> filenames;
376  files.GetFileInfo(filenames);
377 
378  for(SDICOS::S_UINT32 i = 0; i < filenames.GetSize(); ++i)
379  {
380  std::cout << filenames[i].m_filename << std::endl;
381  }
382 
383  // One error log is created for each Section in the Series
384  SDICOS::Array1D< std::pair<SDICOS::Filename, SDICOS::ErrorLog> > vErrorlogs;
385  SDICOS::AIT3D aitRead;
386 
387  // Access files by ObjectOfInspection->Scan->Series
388  const SDICOS::DicosFileListing::SopInstance& fileSopInstance = files[0][0][0][0];
389 
390  // Read the CT Data for the Series
391  if(!aitRead.Read(fileSopInstance, vErrorlogs))
392  {
393  std::cerr << "Unable to read AIT3D DICOS files" << std::endl;
394  for(SDICOS::S_UINT32 i = 0; i < vErrorlogs.GetSize(); ++i)
395  {
396  std::cerr << vErrorlogs[i].first << std::endl << vErrorlogs[i].second << std::endl;
397  }
398  return false;
399  }
400  else
401  {
402  std::cout << "Loaded AIT3D DICOS Files. Number of Sections: " << aitRead.GetNumberOfSections() << std::endl;
403  bRes = (2 == aitRead.GetNumberOfSections()) && bRes;
404  SDICOS::SectionAIT * sections[2];
405 
406  // When a AIT3D object is loaded, the Section numbering starts at 0.
407  sections[0] = aitRead.GetSectionByIndex(0);
408  sections[1] = aitRead.GetSectionByIndex(1);
409 
410  bRes = (sections[0] && sections[1]) && bRes;
411 
412  // Check that volumes have the expected dimensions and spacing
413  for(SDICOS::S_UINT32 i = 0; i < 2; ++i)
414  {
415  bool bSectionSpacingFound(false);
416  for(SDICOS::S_UINT32 j = 0; j < 2; ++j)
417  {
418  if(sections[i]->GetColumnSpacing() == sectionSpacing[j][0] &&
419  sections[i]->GetRowSpacing() == sectionSpacing[j][1] &&
420  sections[i]->GetSliceSpacing() == sectionSpacing[j][2])
421  {
422  bSectionSpacingFound = true;
423  break;
424  }
425  }
426  bRes = bSectionSpacingFound && bRes;
427  bool bSectionDimsFound(false);
428  for(SDICOS::S_UINT32 j = 0; j < 2; ++j)
429  {
430  if(sections[i]->GetWidth() == sectionDims[j][0] &&
431  sections[i]->GetHeight() == sectionDims[j][1] &&
432  sections[i]->GetDepth() == sectionDims[j][2])
433  {
434  bSectionDimsFound = true;
435  }
436  }
437 
438  bRes = bSectionDimsFound && bRes;
439  }
440  std::cout << "Comparison of original and loaded AIT files " << bRes << std::endl;
441 
442  }
443  return bRes;
444 }
